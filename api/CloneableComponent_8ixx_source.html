39 /**
40 * @brief CRTP base class for components that support cloning.
41 *
42 * @details CloneableComponent uses the Curiously Recurring Template Pattern (CRTP)
43 * to provide automatic clone() implementation for derived component types.
44 * The derived class must have a copy constructor.
45 *
46 * Example usage:
47 * ```cpp
48 * class HealthComponent : public CloneableComponent<HealthComponent> {
49 * int health_ = 100;
50 * public:
51 * HealthComponent() = default;
52 * HealthComponent(const HealthComponent&) = default;
53 * };
54 *
55 * auto clone = original.clone();
56 * ```
57 *
58 * @tparam Derived The derived component type (CRTP pattern).
59 */
60 template<typename Derived>
61 class CloneableComponent : public Component, public Cloneable {
62
63 public:
64
65 /**
66 * @brief Creates a deep copy of the derived component.
67 *
68 * @details Uses the copy constructor of the Derived type to create
69 * a new instance wrapped in a unique_ptr.
70 *
71 * @return A unique_ptr owning the cloned component.
72 */
73 [[nodiscard]] std::unique_ptr<Component> clone() const override {
74 return std::make_unique<Derived>(static_cast<const Derived&>(*this));
75 }
