17 /**
18 * @class CollisionComponent
19 * @brief Defines collision layer membership and interaction masks for a GameObject.
20 *
21 * @details CollisionComponent enables layer-based collision filtering by assigning
22 * each entity to a specific layer and defining which other layers it can interact with.
23 * Two separate bitmasks control trigger (non-physical) or solid (physical) collision
24 * generation.
25 *
26 * Trigger collisions are asymmetric: entity A can detect entity B without B detecting A.
27 * Solid collisions should be treated symmetrically by the collision system.
28 *
29 * The collision reporter flag controls whether this entity can emit collision events.
30 * Non-reporter entities participate in collision detection but do not generate events.
31 *
32 * Example usage:
33 * ```cpp
34 * constexpr uint32_t LAYER_PLAYER = 1 << 0;
35 * constexpr uint32_t LAYER_ENEMY = 1 << 1;
36 * constexpr uint32_t LAYER_BULLET = 1 << 2;
37 *
38 * auto& collision = gameObject.add<CollisionComponent>(LAYER_PLAYER);
39 * collision.setTriggerCollisionMask(LAYER_ENEMY | LAYER_BULLET);
40 * collision.setSolidCollisionMask(LAYER_ENEMY);
41 * ```
42 */
43 class CollisionComponent : public Component {
