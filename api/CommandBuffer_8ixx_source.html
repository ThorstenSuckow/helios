37 /**
38 * @brief Queue for deferred execution of game commands.
39 *
40 * @details The `CommandBuffer` provides a queue for game commands, decoupling their
41 * creation from their execution. Commands are buffered along with their target
42 * `GameObject` identifiers (`Guid`) and are executed in a single batch when `flush()`
43 * is called against a `GameWorld`.
44 *
45 * Typical usage pattern:
46 * ```cpp
47 * helios::engine::runtime::messaging::command::CommandBuffer cmdBuffer;
48 * helios::engine::runtime::world::GameWorld world;
49 *
50 * // Queue commands during input processing
51 * cmdBuffer.add(player.guid(), std::make_unique<MoveCommand>(direction, speed));
52 * cmdBuffer.add(enemy.guid(), std::make_unique<AttackCommand>(target));
53 *
54 * // Execute all commands at end of frame
55 * cmdBuffer.flush(world); // Executes and clears buffer
56 * ```
57 *
58 * @note Commands targeting non-existent GameObjects are silently skipped during flush().
59 * This behavior will be logged once the logging system is available.
60 */
61 class CommandBuffer {

122 /**
123 * @brief Registers a dispatcher for a specific TargetedCommand type.

125 * @details When flush() processes a command of type T, it will route the

128 * @tparam T The concrete TargetedCommand type to dispatch.
129 *

204 /**
205 * @brief Adds a command targeting a GameObject identified by Guid.
206 *
207 * @param guid The unique identifier of the target GameObject.
208 * @param command The command to execute. Ownership is transferred to the buffer.
209 *
210 * @pre command must not be nullptr.
211 *
212 * @note If command is nullptr, the call is silently ignored (will be logged in future).
213 * @note This overload is useful when the GameObject instance is not directly available,
214 * such as in networked scenarios or when processing serialized commands.
215 */
216 template<typename T, typename... Args>
217 requires std::is_base_of_v<TargetedCommand, T>
218 void add(const helios::engine::ecs::EntityHandle& entityHandle, Args&& ...args) {
219
220 const std::type_index key{ typeid(T) };
221 auto cmd = std::make_unique<T>(std::forward<Args>(args)...);
222
223 targetedCommandBuffer_.push_back(TargetedCommandProxy{key, entityHandle, std::move(cmd)});
224 }

226 /**
227 * @brief Adds a world command to the buffer.
228 *
229 * @tparam T The concrete WorldCommand type to create.
230 * @tparam Args Constructor argument types for T.
231 *
232 * @param args Arguments forwarded to the T constructor.
233 *
234 * @details WorldCommands operate on the entire GameWorld rather than a
235 * specific GameObject. They are executed before TargetedCommands during flush().
236 */
237 template<typename T, typename... Args>
238 requires std::is_base_of_v<WorldCommand, T>
239 void add(Args&& ...args) {
240
241 const std::type_index key{ typeid(T) };
242 auto cmd = std::make_unique<T>(std::forward<Args>(args)...);
243
244 worldCommandBuffer_.push_back(WorldCommandProxy{key, std::move(cmd)});

247 /**
248 * @brief Executes all buffered commands against the GameWorld and clears the buffer.
249 *
250 * @param gameWorld The world containing target GameObjects.
251 *
252 * @return Reference to this CommandBuffer for method chaining.
253 *
254 * @details Execution proceeds in two phases:
255 * 1. **WorldCommands** are processed first. For each command, if a dispatcher
256 * is registered for its type, the command is routed via accept(). Otherwise,
257 * execute() is called directly.
258 * 2. **TargetedCommands** are processed second. Each command's target GameObject
259 * is resolved by Guid. If found and a dispatcher exists, the command is routed
260 * via accept(). Otherwise, execute() is called directly.
261 *
262 * @note Commands targeting non-existent GameObjects (i.e., find() returns nullptr)
263 * are skipped and a warning is logged.
264 * @note All commands are cleared after execution, regardless of success or failure.
265 *
266 * @warning Command execution should be noexcept. If execute() throws, the flush
267 * operation aborts and remaining commands are not executed.
268 */
269 CommandBuffer& flush(helios::engine::runtime::world::GameWorld& gameWorld) {
270
271 // World commands are processed first
272 for (auto& worldCommandProxy : worldCommandBuffer_) {
273
274 auto it = worldCommandDispatchers_.find(worldCommandProxy.typeIdx);
275
276 if (it != worldCommandDispatchers_.end()) {
277 worldCommandProxy.worldCommand->accept(gameWorld, *(it->second));
278 } else {
279 worldCommandProxy.worldCommand->execute(gameWorld);
280 }
281 }
282
283 // Targeted commands are processed second
284 for (auto& targetedCommandProxy : targetedCommandBuffer_) {
285
286 auto gameObject = gameWorld.find(targetedCommandProxy.entityHandle);
287
288 if (!gameObject) {
289 logger_.warn("GameObject with entityHandle not found, skipping command");
290 continue;
291 }
292
293 auto it = targetedCommandDispatchers_.find(targetedCommandProxy.typeIdx);
294
295 if (it != targetedCommandDispatchers_.end()) {
296 targetedCommandProxy.targetedCommand->accept(*gameObject, *(it->second));
297 } else {
298 targetedCommandProxy.targetedCommand->execute(*gameObject);
299 }
300 }
301
302 clear();
303 return *this;
304 }
305
306 /**
307 * @brief Clears all buffered commands without executing them.
308 *
309 * @return Reference to this CommandBuffer for method chaining.
310 *
311 * @note This destroys all buffered commands. Use this to discard commands
312 * without executing them (e.g., when reverting to a previous game state).
313 */
314 CommandBuffer& clear() {
315 targetedCommandBuffer_.clear();
316 worldCommandBuffer_.clear();
317 return *this;
318 }
