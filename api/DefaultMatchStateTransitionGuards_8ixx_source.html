22 /**

24 *
25 * @details Guard functions are predicates that determine whether a state
26 * transition is allowed. They are evaluated before a transition occurs and
27 * can block it by returning false.
28 *
29 * All guards are static functions that take an UpdateContext and a
30 * MatchStateTransitionRequest, returning true if the transition is allowed.
31 *
32 * @see MatchStateTransitionRequest
33 * @see MatchStateManager

39 /**
40 * @brief Guard that allows transition only if the player is active.
41 *
42 * @param updateContext The current frame's update context.
43 * @param transitionRequest The requested state transition.
44 *
45 * @return True if the player entity exists and is active.
46 */
47 static bool isPlayerActive(

49 const StateTransitionRequest<MatchState> transitionRequest

52 return playerOpt.has_value() && playerOpt->isActive();

54 }
