35 /**
36 * @brief Scheduler that evaluates spawn rules and produces spawn plans.
37 *
38 * @details SpawnScheduler is the central coordinator for rule-based spawning.
39 * It maintains a collection of spawn rules, evaluates them each frame, and
40 * produces ScheduledSpawnPlan instances for rules whose conditions are met.
41 *
42 * ## Workflow
43 *
44 * 1. **Registration:** Rules are added via `addRule()` with their profile ID
45 * 2. **Evaluation:** `evaluate()` is called each frame by GameObjectSpawnSystem
46 * 3. **Scheduling:** Rules that pass conditions produce ScheduledSpawnPlans
47 * 4. **Draining:** `drainScheduledPlans()` returns and clears pending plans
48 * 5. **Commit:** `commit()` is called when spawns complete to update state
49 *
50 * Example:
51 * ```cpp
52 * SpawnScheduler scheduler;
53 * scheduler.addRule(enemyProfileId, std::make_unique<TimerSpawnRule>(
54 * ruleId, 2.0f, 3 // Every 2s, spawn 3
55 * ));
56 *
57 * // In game loop
58 * scheduler.evaluate(updateContext);
59 * auto plans = scheduler.drainScheduledPlans();
60 * ```
61 *
62 * @see SpawnRule
63 * @see ScheduledSpawnPlan
64 * @see GameObjectSpawnSystem
65 */

112 scheduledSpawnPlans_.clear();

168 if (spawnRule->spawnRuleId() == spawnRuleId) {
169
170 auto it = spawnRuleStates_.find(spawnRuleId);
171 assert(it != spawnRuleStates_.end() && "Unexpected missing spawnRuleState");
172
173 spawnRule->commit(it->second, spawnCount);
