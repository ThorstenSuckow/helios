16 /**
17 * @class FpsMetrics
18 * @brief Aggregates and analyzes frame timing data over a rolling window.
19 *
20 * FpsMetrics collects FrameStats data from multiple frames and calculates
21 * average FPS and frame times. It maintains a configurable history buffer
22 * for smoothing out momentary fluctuations in frame rate.
23 *
24 * This class is particularly useful for debugging, profiling, and displaying
25 * performance metrics in development tools or debug overlays.
26 *
27 * @par Usage Example:
28 * @code
29 * helios::tooling::FpsMetrics metrics;
30 * metrics.setHistorySize(120);
31 *
32 * // In game loop:
33 * helios::engine::FrameStats stats = framePacer.sync();
34 * metrics.addFrame(stats);
35 *
36 * float fps = metrics.getFps();
37 * float avgFrameTime = metrics.getFrameTimeMs();
38 * @endcode
39 */
40 class FpsMetrics {
41
42 /**
43 * @brief Stores a rolling history of frame statistics.
44 *
45 * `history_` holds the full timing data (work time, wait time, total frame time)
46 * for a specified number of recent frames. This buffer is used to compute
47 * average frame rate and frame time metrics over a configurable window.
48 *
49 * The use of a `std::deque` allows efficient addition and removal
50 * of frame timing records while maintaining the order of events.
51 */
52 std::deque<helios::engine::FrameStats> history_;
53
54 /**
55 * @brief Size of the history buffer for storing frame statistics.
56 *
57 * Determines the maximum number of frames for which the frame timing data
58 * is retained to calculate performance metrics such as average FPS and
59 * frame times. A larger history size smooths out short-term variations
60 * but may reduce responsiveness to sudden changes in performance.
61 *
62 * This variable is initialized to a default value of 60 but can be adjusted
63 * based on the application's requirements for performance analysis.
64 */
65 size_t historySize_ = 60;
66
67 /**
68 * @brief Stores the calculated average frames per second (FPS).
69 *
70 * The `avgFps_` variable holds the computed average FPS value based on the
71 * accumulated frame timing data in `history_`. It is recomputed lazily when
72 * one of the query functions (e.g. `getFps()`) is called after new frames
73 * have been added.
74 */
75 float avgFps_ = 0.0f;
76
77 /**
78 * @brief Stores the average frame time in milliseconds.
79 *
80 * Represents the total frame time (including work and wait) averaged over
81 * the frames currently stored in the history buffer. The value is stored
82 * in milliseconds and recomputed lazily when queried via the public getters
83 * after new frames have been added.
84 */
85 float avgFrameTime_ = 0.0f;
86
87 /**
88 * @brief Tracks the most recent frame's work time in milliseconds.
89 *
90 * Represents the duration of work performed during the last processed frame.
91 * This value is derived from the most recent `FrameStats` in the history and
92 * expressed in milliseconds. It is updated when the metrics are recomputed
93 * (lazy evaluation triggered by the getters).
94 */
95 float lastWorkTime_ = 0.0f;
96
97 /**
98 * @brief Tracks the idle time in milliseconds for the most recent frame.
99 *
100 * Represents the duration the system waited (idle time) during the last
101 * processed frame. This value is derived from the most recent `FrameStats`
102 * in the history and expressed in milliseconds. It is updated when the
103 * metrics are recomputed (lazy evaluation triggered by the getters).
104 */
105 float lastWaitTime_ = 0.0f;
106
107 /**
108 * @var frameCount_
109 * @brief Tracks the total number of frames processed since initialization.
110 *
111 * The frameCount_ variable accumulates the count of frames that have been
112 * added via `addFrame()`. It starts at 0 and increments with each call,
113 * serving as a global counter independent of the current history size.
114 */
115 unsigned long long frameCount_ = 0;
116
117 /**
118 * @brief Indicates whether cached metrics need to be recomputed.
119 *
120 * Set to true whenever new frame data is added or the history changes in a
121 * way that invalidates the cached values. The next call to one of the public
122 * query functions will trigger a recomputation in `update()`.
123 */
124 bool needsUpdate_ = true;
125
126 /**
127 * @brief Recomputes cached metrics if needed.
128 *
129 * If `needsUpdate_` is true and the history is non-empty, this function
130 * updates `lastWorkTime_`, `lastWaitTime_`, `avgFrameTime_` and `avgFps_`
131 * based on the current contents of `history_` and then clears the flag.
132 *
133 * If the history is empty or the cache is already up to date, this function
134 * returns immediately without modifying any values.
135 *
136 * @note This function is safe to call in hot paths and does not throw under
137 * normal conditions; it is therefore marked `noexcept`.
138 */
139 void update() noexcept {
140 if (!needsUpdate_ || history_.empty()) {
141 return;
142 }
143
144 const auto& stats = history_.back();
145
146 lastWorkTime_ = stats.workTime * 1000.0f;
147 lastWaitTime_ = stats.waitTime * 1000.0f;
148
149 float sumMs = 0.0f;
150 for (const auto& s : history_) {
151 sumMs += s.totalFrameTime * 1000.0f;
152 }
153
154 avgFrameTime_ = sumMs / static_cast<float>(history_.size());
155 avgFps_ = (avgFrameTime_ > 0.0001f) ? (1000.0f / avgFrameTime_) : 0.0f;
156
157 needsUpdate_ = false;
158 }

162 /**
163 * @brief Adds a frame's statistics to the metrics history.
164 *
165 * Processes the provided FrameStats, updates the rolling history,
166 * and marks cached values as dirty so they will be recomputed on
167 * the next query. Old frames are automatically removed when the
168 * history buffer exceeds the configured size. Additionally, the
169 * `frameCount_` member is increased by one for each call.
170 *
171 * @param stats Frame statistics to add to the history.
172 *
173 * @note When used together with FramePacer, a typical usage pattern is:
174 * @code
175 * helios::engine::FrameStats stats = framePacer.sync();
176 * metrics.addFrame(stats);
177 * @endcode
178 * This should usually be called once per frame at the end of the
179 * frame loop, after timing information for the current frame has
180 * been measured.
181 */
182 void addFrame(const helios::engine::FrameStats& stats) {
183
184 needsUpdate_ = true;
185
186 // Maintain history for averaging
187 history_.push_back(stats);
188 if (history_.size() > historySize_) {
189 history_.pop_front();
190 }
191
192 frameCount_++;
193 }
194
195 /**
196 * @brief Sets the size of the frame history buffer.
197 *
198 * Determines how many frames are used for calculating averages.
199 * Larger values produce smoother metrics but respond slower to changes.
200 *
201 * @param size Number of frames to keep in history (default: 60).
202 *
203 * @note Common values: 30 for roughly half-second average at 60 FPS,
204 * 60 for roughly one-second average at 60 FPS.
205 */
206 void setHistorySize(size_t size) {
207 historySize_ = size;
208 while (history_.size() > historySize_) {
209 history_.pop_front();
210 }
211 needsUpdate_ = true;
212 }
