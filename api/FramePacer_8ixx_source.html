19 /**
20 * @class FramePacer
21 * @brief Controls and maintains a target frame rate through precise timing and pacing.
22 *
23 * The FramePacer class utilizes a `Stopwatch` to measure frame execution time and
24 * introduces necessary sleep delays to maintain a consistent target frame rate.
25 * It returns detailed timing statistics via the `FrameStats` structure upon synchronization.
26 *
27 * @details The pacing mechanism helps in achieving smoother frame delivery by minimizing
28 * jitter, although strict adherence depends on the OS scheduler's precision.
29 *
30 * @note If `targetFps` is set to 0.0f, no frame pacing is performed, and frames
31 * run as fast as the hardware and OS allow (unlocked framerate).
32 *
33 * @par Usage Example:
34 * ```cpp
35 * auto stopwatch = std::make_unique<helios::util::time::Stopwatch>();
36 * FramePacer pacer(std::move(stopwatch));
37 * pacer.setTargetFps(60.0f);
38 *
39 * while (running) {
40 * pacer.beginFrame();
41 * // ... game logic and rendering ...
42 * FrameStats stats = pacer.sync();
43 * }
44 * ```
45 */
46 class FramePacer {
47 /**
48 * @brief The stopwatch used for high-resolution time measurement.
49 */
50 std::unique_ptr<helios::util::time::Stopwatch> stopwatch_;

53 * @brief The target frame rate in Frames Per Second (FPS).
54 */

82 /**
83 * @brief Retrieves the current target frame rate.
84 *
85 * Returns the frame rate target that has been set for pacing, expressed in
86 * Frames Per Second (FPS). If the target frame rate is set to 0.0f, frame pacing
87 * is disabled, and the frame rate is unlocked.
88 *
89 * @return The target frame rate in FPS.
90 *
91 * @note A return value of 0.0f indicates that the frame pacing mechanism is not active.

94 return targetFps_;
95 }
96
97 /**
98 * @brief Marks the beginning of a new frame.
99 *
100 * Starts the internal stopwatch to begin measuring the frame's work time.
101 * This method must be called at the very beginning of each frame cycle,
102 * before any game logic, physics, or rendering operations.
103 */
104 void beginFrame() {
105 stopwatch_->start();
106 }
107
108 /**
109 * @brief Synchronizes frame timing and returns frame statistics.
110 *
111 * Measures the elapsed work time since `beginFrame()`. If a target FPS is set
112 * and the work time is less than the target frame duration, this method
113 * sleeps the current thread to meet the target timing.
114 *
115 * @return A `FrameStats` structure containing the total frame time (including wait),
116 * the actual work time (CPU processing), and the wait time (idle).
117 *
118 * @note If `targetFps` is 0.0f or the frame took longer than the target duration,
119 * no sleeping occurs, and `waitTime` in the returned stats will be 0.0f.
120 *
121 * @todo Implement hybrid spinning for the last millisecond of the wait time
122 * to improve timing precision and mitigate OS scheduler wake-up latency.
123 */
124 [[nodiscard]] FrameStats sync() {
125 float workTime = stopwatch_->elapsedSeconds();
126
127 float waitTime = 0.0f;
128 float totalTime = workTime;
129
130 if (targetFps_ > 0.0f) {
131 float targetTime = 1.0f / targetFps_;
132 if (targetTime > workTime) {
133 auto requestedWaitTime = targetTime - workTime;
134 auto sleepDuration = std::chrono::duration<float>(requestedWaitTime);
135 std::this_thread::sleep_for(sleepDuration);
136 // In a simple sleep model, we assume we slept for approximately the needed time.
137 // Real-world drift correction would re-measure the stopwatch here.
138 totalTime = stopwatch_->elapsedSeconds();
139 // This accounts for the overhead introduced by OS thread scheduling.
140 waitTime = totalTime-workTime;
141 }
142 }
143
144 return FrameStats{ totalTime, workTime, waitTime };
145 }
146 };
