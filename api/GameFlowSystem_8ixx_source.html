30 /**
31 * @brief System that drives automatic game state transitions.
32 *
33 * GameFlowSystem monitors the current GameState and issues StateCommands
34 * to advance the state machine when appropriate. It tracks the previous
35 * state and transition ID to avoid emitting duplicate commands.
36 *
37 * Typical flow: Undefined -> Start -> Title.
38 *
39 * @see GameState
40 * @see GameStateTransitionId
41 * @see StateCommand

57 /**
58 * @brief Updates the game flow and emits state transition commands.
59 *
60 * Checks the current GameState from the session and compares it with
61 * the previously observed state. If a transition is required, a
62 * StateCommand is added to the command buffer.
63 *
64 * @param updateContext The update context providing session and command buffer access.
65 */
66 void update(helios::engine::runtime::world::UpdateContext& updateContext) noexcept override {
67
68 auto& session = updateContext.session();
69
70 auto& commandBuffer = updateContext.commandBuffer();
71 const auto gameState = session.state<GameState>();
72 auto gameStateTransitionId = session.stateTransitionId<GameState>();
73
74 if (gameState != GameState::Undefined && prevMatchState_ == gameState && prevGameStateTransitionId_ == gameStateTransitionId) {
75 return;
76 }
77
78 prevMatchState_= gameState;
79 prevGameStateTransitionId_ = gameStateTransitionId;
80
81 switch (gameState) {
82
83 case GameState::Undefined: {
84 commandBuffer.add<StateCommand<GameState>>(
85 StateTransitionRequest<GameState>(gameState, GameStateTransitionId::StartRequested)
86 );
87 break;
88 }
89
90 case GameState::Start: {
91 commandBuffer.add<StateCommand<GameState>>(
92 StateTransitionRequest<GameState>(gameState, GameStateTransitionId::TitleRequested)
93 );
94 break;
95 }
96
97
98 default:
99 break;
100 }
101 }
102 };
