24 /**
25 * @typedef GameLoopEventBus
26 * @brief Double-buffered event bus for decoupled inter-system communication.
27 *
28 * @details GameLoopEventBus is a type alias for TypeIndexedDoubleBuffer configured
29 * with a dedicated index space for game loop events. It enables systems to publish
30 * events during their update phase, which become available for consumption after
31 * the buffer swap at frame boundaries.
32 *
33 * Events are pushed during system updates and read in subsequent phases or frames,
34 * enabling temporal decoupling between event producers and consumers.
35 *
36 * Example usage:
37 * ```cpp
38 * // Define an event type
39 * struct CollisionEvent {
40 * EntityId a;
41 * EntityId b;
42 * vec3f contactPoint;
43 * };
44 *
45 * // In collision detection system
46 * eventBus.push<CollisionEvent>(entityA, entityB, contact);
47 *
48 * // At frame boundary (in game loop)
49 * eventBus.swapBuffers();
50 *
51 * // In damage system (reads events from previous frame)
52 * for (const auto& evt : eventBus.read<CollisionEvent>()) {
53 * applyDamage(evt.a, evt.b);
54 * }
55 * ```
56 *
57 * @see helios::core::data::TypeIndexedDoubleBuffer
58 */
59 using GameLoopEventBus = helios::core::data::TypeIndexedDoubleBuffer<
60 helios::core::data::TypeIndexer<GameLoopEventBusGroup>
