34 /**
35 * @brief Central orchestrator for the game update cycle.
36 *
37 * @details
38 * The GameLoop manages the execution of game systems across three distinct phases:
39 * Pre, Main, and Post. Each phase can contain multiple passes, and each pass can
40 * have a configurable commit point for fine-grained synchronization control.
41 *
42 * ## Ownership
43 *
44 * The GameLoop owns:
45 * - **Three phases** (Pre, Main, Post), each containing passes with registered systems.
46 * - **A CommandBuffer** for deferred command execution.
47 * - **Three event buses** for different propagation scopes:
48 * - `phaseEventBus_`: Events readable in the next phase.
49 * - `passEventBus_`: Events readable in subsequent passes (within the same phase).
50 * - `frameEventBus_`: Events readable in the next frame.
51 *
52 * ## Commit Points
53 *
54 * Commit points allow systems to specify when commands should be flushed, managers
55 * should process their requests, and pass-level events should be synchronized.
56 * This enables deterministic ordering and fine-grained control over the update cycle.
57 *
58 * ## Frame Lifecycle
59 *
60 * ```
61 * ┌─────────────────────────────────────────────────────────────┐
62 * │ FRAME N │
63 * ├─────────────────┬─────────────────┬─────────────────────────┤
64 * │ PRE PHASE │ MAIN PHASE │ POST PHASE │
65 * │ (Input, Cmd) │ (Physics, AI) │ (Cleanup, Sync) │
66 * ├─────────────────┼─────────────────┼─────────────────────────┤
67 * │ phaseCommit() │ phaseCommit() │ phaseCommit() │
68 * │ │ │ frameEventBus_.swap() │
69 * └─────────────────┴─────────────────┴─────────────────────────┘
70 * ```
71 *
72 * @see Phase
73 * @see Pass
74 * @see CommitPoint
75 * @see CommandBuffer
76 * @see PassCommitListener
77 */
78 class GameLoop : public helios::engine::runtime::gameloop::PassCommitListener {

161 /**
162 * @brief Commits pass-level state based on the specified CommitPoint flags.
163 *
164 * @details Called after each pass that has a commit point configured. The
165 * CommitPoint flags determine which synchronization actions are performed:
166 *
167 * - **PassEvents:** Swaps pass event bus buffers, making events pushed via
168 * `UpdateContext::pushPass()` readable in subsequent passes.
169 * - **FlushCommands:** Executes pending commands from the CommandBuffer.
170 * - **FlushManagers:** Processes manager requests (e.g., spawning from pools).
171 *
172 * The order is: PassEvents → FlushCommands → FlushManagers.
173 * Commands must be flushed before managers to ensure spawn requests are
174 * generated before being processed.
175 *
176 * @param commitPoint The flags specifying which actions to perform.
177 * @param updateContext The current update context.
178 *
179 * @see CommitPoint
180 * @see Pass::addCommitPoint()
181 * @see UpdateContext::pushPass()
182 * @see UpdateContext::readPass()
183 */
184 void onPassCommit(
185 const CommitPoint commitPoint,
186 helios::engine::runtime::world::UpdateContext& updateContext) noexcept override {
187
188
189 if ((commitPoint & CommitPoint::PassEvents) == CommitPoint::PassEvents) {
190 passEventBus_.swapBuffers();
191 }
192
193 // commands must be executed before Managers
194 if ((commitPoint & CommitPoint::FlushCommands) == CommitPoint::FlushCommands) {
195 commandBuffer_.flush(updateContext.gameWorld());
196 }
197
198 if ((commitPoint & CommitPoint::FlushManagers) == CommitPoint::FlushManagers) {
199 updateContext.gameWorld().flushManagers(updateContext);
200 }
201
202 }
203
204 /**
205 * @brief Commits phase-level events and flushes commands and managers.
206 *
207 * Called after each phase completes. This method:
208 * 1. Swaps phase event bus buffers, making events readable in the next phase.
209 * 2. Clears pass event buffers for the new phase.
210 * 3. Flushes the command buffer, executing deferred commands.
211 * 4. Flushes managers, allowing to process any request generated by the commands.
212 *
213 * @param gameWorld Reference to the game world.
214 * @param updateContext The current update context.
215 *
216 * @see UpdateContext::pushPhase()
217 * @see UpdateContext::readPhase()
218 */
219 void phaseCommit(
220 helios::engine::runtime::world::GameWorld& gameWorld,
221 helios::engine::runtime::world::UpdateContext& updateContext) {
222 phaseEventBus_.swapBuffers();
223 passEventBus_.clearAll();
224
225 // command buffer generates request for managers, so this comes first
226 commandBuffer_.flush(updateContext.gameWorld());
227
228 // managers process requests
229 gameWorld.flushManagers(updateContext);
230 }

310 mainPhase_.init(gameWorld);
311 mainPhase_.addPassCommitListener(this);
312 break;

323 /**
324 * @brief Executes one full frame update across all phases.
325 *
326 * @details Iterates through Pre, Main, and Post phases, updating all registered
327 * systems and committing events and commands after each phase. The frame lifecycle:
328 *
329 * 1. **Pre Phase:** Input processing, command generation, preparation
330 * 2. **Main Phase:** Core gameplay logic, physics, collision detection
331 * 3. **Post Phase:** Cleanup, synchronization, rendering preparation
332 *
333 * After each phase, `phaseCommit()` is called to:
334 * - Swap phase event buffers (events become readable in next phase)
335 * - Clear pass event buffers
336 * - Flush command buffer
337 * - Flush managers
338 *
339 * After the Post phase, the frame event bus is swapped, making frame-level
340 * events readable in the next frame.
341 *
342 * @param gameWorld Reference to the game world.
343 * @param deltaTime Time elapsed since the last frame in seconds.
344 * @param inputSnapshot Snapshot of the current input state.
345 * @param viewportSnapshots Snapshots of viewports registered with an id.
346 *
347 * @pre init() must have been called before the first update.
348 *
349 * @see Phase
350 * @see phaseCommit()
351 * @see UpdateContext
352 */
353 void update(
354 helios::engine::runtime::world::GameWorld& gameWorld,
355 float deltaTime,
356 const helios::input::InputSnapshot& inputSnapshot,
357 std::span<const helios::rendering::ViewportSnapshot> viewportSnapshots
358 ) noexcept {
359
360 assert(initialized_ && "GameLoop not initialized");
361
362 auto updateContext = helios::engine::runtime::world::UpdateContext(
363 commandBuffer_,
364 gameWorld,
365 deltaTime,
366 phaseEventBus_,
367 passEventBus_,
368 frameEventBus_,
369 inputSnapshot,
370 viewportSnapshots
371 );
372
373
374 // gameloop phases
375 for (auto phase : {helios::engine::runtime::gameloop::PhaseType::Pre,
376 helios::engine::runtime::gameloop::PhaseType::Main,
377 helios::engine::runtime::gameloop::PhaseType::Post}) {
378
379 switch (phase) {
380 case helios::engine::runtime::gameloop::PhaseType::Pre:
381 prePhase_.update(updateContext);
382 phaseCommit(gameWorld, updateContext);
383 break;
384 case helios::engine::runtime::gameloop::PhaseType::Main:
385 mainPhase_.update(updateContext);
386 phaseCommit(gameWorld, updateContext);
387
388 break;
389 case helios::engine::runtime::gameloop::PhaseType::Post:
390 postPhase_.update(updateContext);
391 phaseCommit(gameWorld, updateContext);
392 frameEventBus_.swapBuffers();
393 break;
394 }
395
396
397 }
398 }
