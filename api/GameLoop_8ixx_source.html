30 /**
31 * @brief Central orchestrator for the game update cycle.
32 *
33 * The GameLoop manages the execution of game systems across three distinct phases:
34 * Pre, Main, and Post. Each phase can contain multiple passes, and each pass can
35 * have a configurable commit point for fine-grained synchronization control.
36 *
37 * The GameLoop owns:
38 * - Three phases (Pre, Main, Post), each containing passes with registered systems.
39 * - A CommandBuffer for deferred command execution.
40 * - Two event buses: one for phase-level and one for pass-level event propagation.
41 *
42 * Commit points allow systems to specify when commands should be flushed, managers
43 * should process their requests, and pass-level events should be synchronized.
44 * This enables deterministic ordering and fine-grained control over the update cycle.
45 *
46 * @see Phase
47 * @see Pass
48 * @see CommitPoint
49 * @see CommandBuffer
50 */

129 /**
130 * @brief Commits pass-level state based on the specified CommitPoint flags.
131 *
132 * @details Called after each pass that has a commit point configured. The
133 * CommitPoint flags determine which synchronization actions are performed:
134 *
135 * - **PassEvents:** Swaps pass event bus buffers, making events pushed via
136 * `UpdateContext::pushPass()` readable in subsequent passes.
137 * - **FlushCommands:** Executes pending commands from the CommandBuffer.
138 * - **FlushManagers:** Processes manager requests (e.g., spawning from pools).
139 *
140 * The order is: PassEvents → FlushCommands → FlushManagers.
141 * Commands must be flushed before managers to ensure spawn requests are
142 * generated before being processed.
143 *
144 * @param commitPoint The flags specifying which actions to perform.
145 * @param updateContext The current update context.
146 *
147 * @see CommitPoint
148 * @see Pass::addCommitPoint()
149 * @see UpdateContext::pushPass()
150 * @see UpdateContext::readPass()
151 */
152 void passCommit(
153 const CommitPoint commitPoint,
154 helios::engine::runtime::world::UpdateContext& updateContext) {
155
156
157 if ((commitPoint & CommitPoint::PassEvents) == CommitPoint::PassEvents) {
158 passEventBus_.swapBuffers();
159 }
160
161 // commands must be executed before Managers
162 if ((commitPoint & CommitPoint::FlushCommands) == CommitPoint::FlushCommands) {
163 commandBuffer_.flush(updateContext.gameWorld());
164 }
165
166 if ((commitPoint & CommitPoint::FlushManagers) == CommitPoint::FlushManagers) {
167 updateContext.gameWorld().flushManagers(updateContext);
168 }
169
170 }
171
172 /**
173 * @brief Commits phase-level events and flushes commands and managers.
174 *
175 * Called after each phase completes. This method:
176 * 1. Swaps phase event bus buffers, making events readable in the next phase.
177 * 2. Clears pass event buffers for the new phase.
178 * 3. Flushes the command buffer, executing deferred commands.
179 * 4. Flushes managers, allowing to process any request generated by the commands.
180 *
181 * @param gameWorld Reference to the game world.
182 * @param updateContext The current update context.
183 *
184 * @see UpdateContext::pushPhase()
185 * @see UpdateContext::readPhase()
186 */
187 void phaseCommit(
188 helios::engine::runtime::world::GameWorld& gameWorld,
189 helios::engine::runtime::world::UpdateContext& updateContext) {
190 phaseEventBus_.swapBuffers();
191 passEventBus_.clearAll();
192
193 // command buffer generates request for managers, so this comes first
194 commandBuffer_.flush(updateContext.gameWorld());
195
196 // managers process requests
197 gameWorld.flushManagers(updateContext);
198 }

281 /**
282 * @brief Executes one full frame update across all phases.
283 *
284 * @details Iterates through Pre, Main, and Post phases, updating all registered
285 * systems and committing events and commands after each phase. The frame lifecycle:
286 *
287 * 1. **Pre Phase:** Input processing, command generation, preparation
288 * 2. **Main Phase:** Core gameplay logic, physics, collision detection
289 * 3. **Post Phase:** Cleanup, synchronization, rendering preparation
290 *
291 * After each phase, `phaseCommit()` is called to:
292 * - Swap phase event buffers (events become readable in next phase)
293 * - Clear pass event buffers
294 * - Flush command buffer
295 * - Flush managers
296 *
297 * After the Post phase, the frame event bus is swapped, making frame-level
298 * events readable in the next frame.
299 *
300 * @param gameWorld Reference to the game world.
301 * @param deltaTime Time elapsed since the last frame in seconds.
302 * @param inputSnapshot Snapshot of the current input state.
303 *
304 * @pre init() must have been called before the first update.
305 *
306 * @see Phase
307 * @see phaseCommit()
308 * @see UpdateContext
309 */
310 void update(helios::engine::runtime::world::GameWorld& gameWorld, float deltaTime, const helios::input::InputSnapshot& inputSnapshot) noexcept {
311
312 assert(initialized_ && "GameLoop not initialized");
313
314 auto updateContext = helios::engine::runtime::world::UpdateContext(
315 commandBuffer_,
316 gameWorld,
317 deltaTime,
318 phaseEventBus_,
319 passEventBus_,
320 frameEventBus_,
321 inputSnapshot
322 );
323
324
325 // gameloop phases
326 for (auto phase : {helios::engine::runtime::gameloop::PhaseType::Pre,
327 helios::engine::runtime::gameloop::PhaseType::Main,
328 helios::engine::runtime::gameloop::PhaseType::Post}) {
329
330 switch (phase) {
331 case helios::engine::runtime::gameloop::PhaseType::Pre:
332 prePhase_.update(updateContext);
333 phaseCommit(gameWorld, updateContext);
334 break;
335 case helios::engine::runtime::gameloop::PhaseType::Main:
336 mainPhase_.update(updateContext);
337 phaseCommit(gameWorld, updateContext);
338
339 break;
340 case helios::engine::runtime::gameloop::PhaseType::Post:
341 postPhase_.update(updateContext);
342 phaseCommit(gameWorld, updateContext);
343 frameEventBus_.swapBuffers();
344 break;
345 }
346
347
348 }
349 }
