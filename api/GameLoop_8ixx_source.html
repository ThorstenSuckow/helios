29 /**
30 * @brief Central orchestrator for the game update cycle.
31 *
32 * The GameLoop manages the execution of game systems across three distinct phases:
33 * Pre, Main, and Post. Each phase can contain multiple passes, and each pass can
34 * have an optional commit point for event synchronization.
35 *
36 * The GameLoop owns:
37 * - Three phases (Pre, Main, Post), each containing passes with registered systems.
38 * - A CommandBuffer for deferred command execution.
39 * - Two event buses: one for phase-level and one for pass-level event propagation.
40 *
41 * @see Phase
42 * @see Pass
43 * @see CommandBuffer
44 */

107 /**
108 * @brief Commits pass-level events by swapping the pass event bus buffers.
109 *
110 * Called after each pass that has a commit point. Events pushed via
111 * `UpdateContext::pushPass()` become readable in subsequent passes
112 * via `UpdateContext::readPass()`.
113 *
114 * @see Pass::addCommitPoint()
115 * @see UpdateContext::pushPass()
116 * @see UpdateContext::readPass()
117 */
118 void passCommit() {
119 passEventBus_.swapBuffers();
120 }
121
122 /**
123 * @brief Commits phase-level events and flushes commands and managers.
124 *
125 * Called after each phase completes. This method:
126 * 1. Swaps phase event bus buffers, making events readable in the next phase.
127 * 2. Clears pass event buffers for the new phase.
128 * 3. Flushes the command buffer, executing deferred commands.
129 * 4. Flushes managers, allowing to process any request generated by the commands.
130 *
131 * @param gameWorld Reference to the game world.
132 * @param updateContext The current update context.
133 *
134 * @see UpdateContext::pushPhase()
135 * @see UpdateContext::readPhase()
136 */
137 void phaseCommit(helios::engine::runtime::world::GameWorld& gameWorld, helios::engine::runtime::world::UpdateContext& updateContext) {
138 phaseEventBus_.swapBuffers();
139 passEventBus_.clearAll();
140
141 // command buffer generates request for managers, so this comes first
142 commandBuffer_.flush(updateContext.gameWorld());
143
144 // managers process requests
145 gameWorld.flushManagers(updateContext);
146 }

218 /**
219 * @brief Executes one full frame update across all phases.
220 *
221 * Iterates through Pre, Main, and Post phases, updating all registered
222 * systems and committing events and commands after each phase.
223 *
224 * @param gameWorld Reference to the game world.
225 * @param deltaTime Time elapsed since the last frame in seconds.
226 * @param inputSnapshot Snapshot of the current input state.
227 */
228 void update(helios::engine::runtime::world::GameWorld& gameWorld, float deltaTime, const helios::input::InputSnapshot& inputSnapshot) noexcept {
229
230 assert(initialized_ && "GameLoop not initialized");
231
232 auto updateContext = helios::engine::runtime::world::UpdateContext(
233 commandBuffer_,
234 gameWorld,
235 phaseEventBus_,
236 passEventBus_
237 );
238
239 updateContext.setDeltaTime(deltaTime);
240 updateContext.setInputSnapshot(inputSnapshot);
241
242 // gameloop phases
243 for (auto phase : {helios::engine::runtime::gameloop::PhaseType::Pre,
244 helios::engine::runtime::gameloop::PhaseType::Main,
245 helios::engine::runtime::gameloop::PhaseType::Post}) {
246
247 switch (phase) {
248 case helios::engine::runtime::gameloop::PhaseType::Pre:
249 prePhase_.update(updateContext);
250 break;
251 case helios::engine::runtime::gameloop::PhaseType::Main:
252 mainPhase_.update(updateContext);
253 break;
254 case helios::engine::runtime::gameloop::PhaseType::Post:
255 postPhase_.update(updateContext);
256 break;
257 }
258
259 phaseCommit(gameWorld, updateContext);
260 }
261 }
