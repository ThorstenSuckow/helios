20 /**
21 * @brief Facade class that coordinates object pooling with the game world.
22 *
23 * @details `GameObjectPoolFacade` provides a convenient interface for acquiring
24 * and releasing pooled GameObjects while maintaining synchronization between
25 * the GameObjectPool and the GameWorld.
26 *
27 * This facade ensures that released objects are deactivated in the GameWorld and
28 * acquired objects are validated against the GameWorld before use.
29 *
30 * Stale pool entries (objects no longer in the world) are automatically cleaned up,
31 * This process is currently implemented in the acquire()-method.
32 */
33 class GameObjectPoolFacade {
34
35 public:

37 /**
38 * @brief Releases a GameObject back to the pool and deactivates it.
39 *
40 * @details Locates the GameObject in the GameWorld by its entity ID,
41 * releases it back to the pool, and sets its active state to `false`.
42 * If the entity is not found in the world or the release fails, the
43 * object is not modified.
44 *
45 * @param gameWorld The GameWorld containing the entity.
46 * @param gameObjectPool The pool managing the entity lifecycle.
47 * @param entityId The unique identifier of the entity to release.
48 *
49 * @return Pointer to the released GameObject, or `nullptr` if not found.
50 */
51 helios::engine::ecs::GameObject* release(
52 helios::engine::runtime::world::GameWorld& gameWorld,
53 helios::engine::runtime::pooling::GameObjectPool& gameObjectPool,
54 const helios::util::Guid& entityId
55 ) {
56 helios::engine::ecs::GameObject* worldGo = gameWorld.find(entityId);
57
58 if (worldGo) {
59 if (gameObjectPool.release(entityId)) {
60 worldGo->onRelease();
61 worldGo->setActive(false);
62 }
63 }
64
65 return worldGo;
66 }
67
68 /**
69 * @brief Acquires an available GameObject from the pool.
70 *
71 * @details Attempts to acquire an available entity from the pool and
72 * validates that it exists in the GameWorld. If an acquired entity is
73 * considered stale (no longer present in the world), it is automatically
74 * removed from the pool and the next available entity is tried.
75 *
76 * If a GameObject was successfully identified, its onAcquire()-method
77 * is called before its returned.
78 *
79 * @param gameWorld The GameWorld to validate entities against.
80 * @param gameObjectPool The pool to acquire from.
81 *
82 * @return Pointer to the acquired GameObject, or `nullptr` if the pool
83 * is exhausted or no valid entities are available.
84 */
85 [[nodiscard]] helios::engine::ecs::GameObject* acquire(
86 helios::engine::runtime::world::GameWorld& gameWorld,
87 helios::engine::runtime::pooling::GameObjectPool& gameObjectPool
88 ) {
89 helios::util::Guid guid{helios::util::no_init};
90
91 while (gameObjectPool.acquire(guid)) {
92
93 auto* worldGo = gameWorld.find(guid);
94
95 if (worldGo) {
96 worldGo->onAcquire();
97 return worldGo;
98 }
99
100 // we assert the pool is owned by this gameWorld,
101 // so removing this guid does not impact another gameWorld that is
102 // using this pool
103 gameObjectPool.releaseAndRemove(guid);
104 }
105
106 return nullptr;
107
108 }
