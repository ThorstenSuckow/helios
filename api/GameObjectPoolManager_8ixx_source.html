34 /**
35 * @brief High-level manager for GameObject pooling operations.
36 *
37 * @details GameObjectPoolManager provides a unified interface for managing
38 * multiple GameObjectPools. It handles pool configuration, initialization,
39 * and the acquire/release lifecycle of pooled GameObjects.
40 *
41 * The manager integrates with the GameWorld to clone prefabs during pool
42 * initialization and to look up GameObjects by their EntityHandle during
43 * acquire/release operations.
44 *
45 * ## Lifecycle
46 *
47 * 1. **Configuration**: Add pool configurations via `addPoolConfig()`.
48 * 2. **Initialization**: Call `init()` to create pools and populate them
49 * with cloned prefab instances. This **locks** the pools.
50 * 3. **Runtime**: Use `acquire()` and `release()` to manage entity lifecycle.
51 *
52 * ## Pool Locking
53 *
54 * After `init()` completes, each pool is **locked**. A locked pool:
55 * - Cannot accept new EntityHandles via `addInactive()`
56 * - Has its sparse arrays sized based on min/max EntityIds
57 * - Is ready for O(1) acquire/release operations
58 *
59 * This design optimizes memory layout but means the pool size is fixed
60 * at initialization time.
61 *
62 * ## Trade-offs
63 *
64 * - **Fixed Capacity**: Pool size cannot grow after initialization. If more
65 * entities are needed, `acquire()` returns nullptr when exhausted.
66 * - **Memory Overhead**: Sparse arrays are sized for the EntityId range,
67 * which may waste memory if EntityIds are not contiguous.
68 * - **Initialization Cost**: All prefab clones are created upfront during
69 * `init()`, which may cause a startup delay for large pools.
70 *
71 * ## Example
72 *
73 * ```cpp
74 * GameObjectPoolManager poolManager;
75 *
76 * // Create prefab
77 * auto bulletPrefab = gameWorld.addGameObject();
78 * bulletPrefab.add<RenderableComponent>(mesh, material);
79 * bulletPrefab.add<Move2DComponent>();
80 * bulletPrefab.setActive(false);
81 *
82 * auto config = std::make_unique<GameObjectPoolConfig>(
83 * bulletPoolId, bulletPrefab, 100
84 * );
85 * poolManager.addPoolConfig(std::move(config));
86 *
87 * poolManager.init(gameWorld);
88 *
89 * // Acquire returns std::optional<GameObject>
90 * if (auto bullet = poolManager.acquire(bulletPoolId)) {
91 * bullet->get<TranslationStateComponent>()->setTranslation(spawnPos);
92 * bullet->setActive(true);
93 * }
94 *
95 * // Release back to pool
96 * poolManager.release(bulletPoolId, bullet->entityHandle());
97 * ```
98 *
99 * @see GameObjectPool
100 * @see GameObjectPoolRegistry
101 * @see Manager
102 */

131 /**
132 * @brief Populates a pool with cloned prefab instances and locks it.
133 *
134 * @details Clones the prefab GameObject until the pool reaches its
135 * configured capacity. Each clone is:
136 * 1. Added to the GameWorld
137 * 2. Immediately deactivated (`setActive(false)`)
138 * 3. Prepared for pooling (`onRelease()`)
139 * 4. Registered as inactive in the pool
140 *
141 * After all clones are created, the pool is **locked** via `lock()`.
142 * Locking finalizes the sparse array sizing based on the min/max EntityIds
143 * and enables O(1) acquire/release operations. Once locked, no new
144 * EntityHandles can be added to the pool.
145 *
146 * @param gameObjectPoolId The pool to fill.
147 * @param gameObjectPrefab The prefab to clone.
148 *
149 * @post The pool is locked and ready for acquire/release operations.
150 */
151 void fillPool(
152 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId,
153 helios::engine::ecs::GameObject gameObjectPrefab
154 ) {
155 helios::engine::ecs::EntityHandle entityHandle{};
156
157 auto* gameObjectPool = pool(gameObjectPoolId);
158
159 const size_t used = gameObjectPool->activeCount() + gameObjectPool->inactiveCount();
160 const size_t space = used < gameObjectPool->size() ? gameObjectPool->size() - used : 0;
161
162 for (size_t i = 0; i < space; i++) {
163 helios::engine::ecs::GameObject go = gameWorld_->clone(gameObjectPrefab);
164 go.setActive(false);
165 go.onRelease();
166 gameObjectPool->addInactive(go.entityHandle());
167 }
168
169 gameObjectPool->lock();
170 }

217 /**

220 * @details Marks the entity as inactive in both the pool and the
221 * GameWorld. Calls `onRelease()` on the GameObject to trigger

223 *

230 std::optional<helios::engine::ecs::GameObject> release(
231 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId,
232 const helios::engine::ecs::EntityHandle& entityHandle
233 ) {
234 auto* gameObjectPool = pool(gameObjectPoolId);
235
236 auto worldGo = gameWorld_->find(entityHandle);
237
238 if (worldGo) {
239 if (gameObjectPool->release(entityHandle)) {
240 worldGo->onRelease();
241 worldGo->setActive(false);
242 }
243 }
244
245 return worldGo;
246 }

248 /**
249 * @brief Acquires an inactive GameObject from the pool.
250 *
251 * @details Retrieves the next available inactive entity and calls
252 * `onAcquire()` to trigger component initialization hooks. The caller
253 * is responsible for activating the entity via `setActive(true)`.
254 *
255 * If an acquired entity is no longer valid in the GameWorld (stale handle),
256 * it is removed from the pool and the next available entity is tried.
257 *
258 * @param gameObjectPoolId The pool to acquire from.
259 *
260 * @return Optional containing the acquired GameObject if available,
261 * std::nullopt if the pool is exhausted.
262 */
263 [[nodiscard]] std::optional<helios::engine::ecs::GameObject> acquire(
264 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId
265 ) {
266 helios::engine::ecs::EntityHandle entityHandle{};
267
268 auto* gameObjectPool = pool(gameObjectPoolId);
269
270 while (gameObjectPool->acquire(entityHandle)) {
271
272 auto worldGo = gameWorld_->find(entityHandle);
273
274 if (worldGo) {
275 worldGo->onAcquire();
276 return worldGo;
277 }
278
279 // we assume the pool is owned by this gameWorld,
280 // so removing this entityHandle does not impact another gameWorld that is
281 // using this pool
282 gameObjectPool->releaseAndRemove(entityHandle);
283 }
284
285 return std::nullopt;
286
287 }

347 * @details Iterates through all registered pools and releases every active

358 auto activeSpan = poolPtr->activeGameObjects();
359 auto toRelease = std::vector(activeSpan.begin(), activeSpan.end());
360
361 for (auto entityHandle : toRelease) {
