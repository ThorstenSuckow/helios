28 /**
29 * @brief Object pool for efficient GameObject lifecycle management.
30 *
31 * @details GameObjectPool manages a fixed-size collection of GameObject identifiers,
32 * tracking which objects are currently active (in-use) and which are inactive (available).
33 * This pattern eliminates runtime allocation overhead for frequently spawned entities
34 * like projectiles, particles, or enemies.
35 *
36 * The pool uses O(1) operations for both acquire and release:
37 * - **acquire**: Pops from the inactive list and adds to active tracking
38 * - **release**: Swap-and-pop removal from active list, push to inactive
39 *
40 * GameObjects themselves are owned by GameWorld; this pool only tracks their EntityHandles.
41 *
42 * @todo Prevent duplicate EntityHandles from being added to the pool.
43 */
44 class GameObjectPool {

142 /**
143 * @brief Acquires an inactive GameObject from the pool.
144 *
145 * @details Removes a EntityHandle from the inactive list and adds it to the active
146 * tracking structures. The caller is responsible for activating the actual
147 * GameObject in the GameWorld.
148 *
149 * @param[out] entityHandle Receives the EntityHandle of the acquired object on success.
150 *
151 * @return True if an object was acquired, false if the pool is exhausted.
152 */
153 [[nodiscard]] bool acquire(helios::engine::ecs::EntityHandle& entityHandle) {
154
155 if (inactiveGameObjects_.empty()) {
156 return false;
157 }
158
159 entityHandle = inactiveGameObjects_.back();
160 inactiveGameObjects_.pop_back();
161
162
163 auto idx = entityHandle.entityId - delta_;
164
165 if (activeIndex_.size() <= idx) {
166 activeIndex_.resize(idx + 1, helios::engine::core::data::EntityTombstone);
167 versionIndex_.resize(idx + 1, helios::engine::core::data::EntityTombstone);
168 }
169
170 activeIndex_[idx] = activeGameObjects_.size();
171 versionIndex_[idx] = entityHandle.versionId;
172
173 activeGameObjects_.push_back(entityHandle);
174
175 return true;
176 }

187 /**
188 * @brief Locks the pool for acquire/release operations.
189 *
190 * @details After locking, no more EntityHandles can be added via `addInactive()`.
191 * The sparse arrays are sized based on the min/max EntityIds added.
192 */
193 void lock() noexcept {
194 locked_ = true;
195 delta_ = minEntityId_;
196 activeIndex_.resize(maxEntityId_ - delta_ + 1, helios::engine::core::data::EntityTombstone);
197 versionIndex_.resize(maxEntityId_ - delta_ + 1, helios::engine::core::data::EntityTombstone);
198 }

200 /**
201 * @brief Adds a EntityHandle to the inactive list without acquiring it.
202 *
203 * @details Used during pool initialization to register pre-created GameObjects.
204 * Fails if the pool is already at capacity.
205 *
206 * @param entityHandle The EntityHandle of the GameObject to add.
207 *
208 * @return True if added successfully, false if pool is full.
209 */
210 bool addInactive(const helios::engine::ecs::EntityHandle entityHandle) {
211
212 assert(!locked_ && "Pool is locked");
213
214 assert(entityHandle.isValid() && "Unexpected invalid entityHandle");
215
216 const size_t used = (activeCount() + inactiveCount());
217
218 minEntityId_ = std::min(minEntityId_, entityHandle.entityId);
219 maxEntityId_ = std::max(maxEntityId_, entityHandle.entityId);
220
221
222 if (used < size()) {
223 inactiveGameObjects_.push_back(entityHandle);
224 return true;
225 }
226
227 return false;
228 }

230 /**
231 * @brief Releases a GameObject back to the pool by its EntityHandle.
232 *
233 * @details
234 * Validates the EntityHandle against both the GameWorld and the active tracking list.
235 * Uses swap-and-pop for O(1) removal from the active list. The object is
236 * marked inactive and added to the inactive list for future acquisition.
237 *
238 * @param entityHandle The unique identifier of the GameObject to release.
239 *
240 * @return True if the object was successfully released, false if the EntityHandle
241 * was not found in the GameWorld or not tracked as active.
242 */
243 bool release(const helios::engine::ecs::EntityHandle entityHandle) {
244
245 assert(entityHandle.isValid() && "Unexpected invalid entityHandle");
246
247 assert(entityHandle.entityId >= delta_ && "Unexpected entityHandle");
248
249 const auto sparseIdx = entityHandle.entityId - delta_;
250
251 assert(sparseIdx < activeIndex_.size() && "Unexpected sparse index");
252
253 const auto denseIndex = activeIndex_[sparseIdx];
254 if (denseIndex == helios::engine::core::data::EntityTombstone) {
255 return false;
256 }
257
258 assert(versionIndex_[sparseIdx] == entityHandle.versionId && "Version mismatch");
259
260 auto lastEntityHandle = activeGameObjects_.back();
261
262 if (denseIndex != activeGameObjects_.size() - 1) {
263 // swap the last entityHandle in activeGameObjects with the
264 // entityHandle to remove, effectively overwriting entityHandle
265 // to release with a currently active entityHandle
266 activeGameObjects_[denseIndex] = lastEntityHandle;
267 activeIndex_[lastEntityHandle.entityId-delta_] = denseIndex;
268 versionIndex_[lastEntityHandle.entityId-delta_] = lastEntityHandle.versionId;
269 }
270
271
272 // the swap operation has create a duplicate entry,
273 // remove the one at the tail
274 activeGameObjects_.pop_back();
275
276 // clear the queried entityHandle from active index and update
277 // inactiveGameObjects
278 activeIndex_[sparseIdx] = helios::engine::core::data::EntityTombstone;
279 versionIndex_[sparseIdx] = helios::engine::core::data::EntityTombstone;
280
281 inactiveGameObjects_.push_back(entityHandle);
282
283 return true;
284 }

286 /**
287 * @brief Releases and permanently removes a GameObject from the pool.
288 *
289 * @details Unlike `release()`, this method does not add the EntityHandle back to the
290 * inactive list. Use this when a pooled object is being destroyed rather than
291 * recycled.
292 *
293 * @param entityHandle The unique identifier of the GameObject to remove.
294 *
295 * @return True if removed successfully, false if EntityHandle was not active.
296 */
297 bool releaseAndRemove(const helios::engine::ecs::EntityHandle entityHandle) {
298
299 assert(entityHandle.isValid() && "Unexpected invalid entityHandle");
300
301 const auto sparseIdx = entityHandle.entityId - delta_;
302
303 assert(sparseIdx < activeIndex_.size() && "Unexpected sparse index");
304
305 const auto denseIndex = activeIndex_[sparseIdx];
306
307 if (denseIndex == helios::engine::core::data::EntityTombstone) {
308 return false;
309 }
310
311 assert(versionIndex_[sparseIdx] == entityHandle.versionId && "Version mismatch");
312
313 if (denseIndex != activeGameObjects_.size() - 1) {
314 const auto lastEntityHandle = activeGameObjects_.back();
315 activeIndex_[lastEntityHandle.entityId - delta_] = denseIndex;
316 versionIndex_[lastEntityHandle.entityId - delta_] = lastEntityHandle.versionId;
317 activeGameObjects_[denseIndex] = lastEntityHandle;
318 }
319
320 activeGameObjects_.pop_back();
321
322 activeIndex_[sparseIdx] = helios::engine::core::data::EntityTombstone;
323 versionIndex_[sparseIdx] = helios::engine::core::data::EntityTombstone;
324
325 return true;
326
327 }
