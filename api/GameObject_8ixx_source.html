24 /**
25 * @brief Container for components that represents an entity in the game world.
26 *
27 * @details A GameObject is the fundamental entity type in helios. It serves as a
28 * container for Component instances that define its behavior and data. Each GameObject
29 * has a unique Guid for identification and lookup within the GameWorld.
30 *
31 * ## Component Management
32 *
33 * Components are added via `add<T>()` and retrieved via `get<T>()`. The GameObject
34 * uses a type-indexed map (`std::unordered_map<std::type_index, Component*>`) for
35 * **O(1) amortized** component lookup by type.
36 *
37 * ```cpp
38 * auto entity = std::make_unique<GameObject>();
39 * entity->add<SceneNodeComponent>(sceneNode);
40 * entity->add<Move2DComponent>();
41 *
42 * auto* move = entity->get<Move2DComponent>(); // O(1) lookup
43 * ```
44 *
45 * ## Active State
46 *
47 * Each GameObject has an `isActive()` flag that controls participation in the game loop:
48 *
49 * | State | Behavior |
50 * |-------|----------|
51 * | `isActive() == true` | Processed by systems, rendered, collides |
52 * | `isActive() == false` | Skipped by systems, exists but dormant |
53 *
54 * When the active state changes, `onActivate()` or `onDeactivate()` is called on all
55 * attached components.
56 *
57 * ## Lifecycle Callbacks
58 *
59 * Components receive lifecycle notifications:
60 * - `onAttach()` — Called when a component is added to the GameObject
61 * - `onActivate()` / `onDeactivate()` — Called when active state changes
62 * - `onAcquire()` / `onRelease()` — Called when acquired from/released to a pool
63 *
64 * ## Ownership
65 *
66 * GameObjects are typically owned by a GameWorld via `std::unique_ptr`. The GameWorld
67 * provides lookup by Guid with amortized O(1) complexity.
68 *
69 * @see Component
70 * @see GameWorld
71 * @see GameObjectFilter
72 */

139 /**
140 * @brief Adds a new component of type T to the GameObject.
141 *
142 * The component is created, attached to this GameObject (onAttach is called),
143 * and if it implements Updatable, it is added to the update list.
144 * Normalizes T and uses the type-indexed componentIndex_ for O(1) lookup complexity.
145 *
146 * @tparam T The type of component to add. Must derive from Component.
147 * @tparam Args Argument types for the component's constructor.
148 * @param args Arguments forwarded to the component's constructor.
149 *
150 * @return Reference to the newly created component.
151 */
152 template<typename T, typename... Args>
153 T& add(Args&&... args) {
154 using U = std::remove_cvref_t<T>;
155
156 assert(!has<U>() && "Cannot add, Component is already existing.");
157
158 auto component_ptr = std::make_unique<U>(std::forward<Args>(args)...);
159 U* raw_component_ptr = component_ptr.get();
160
161 if constexpr (std::derived_from<U, helios::engine::ecs::Updatable>) {
162 updatables_.push_back(raw_component_ptr);
163 }
164 components_.push_back(std::move(component_ptr));
165 componentIndex_[typeid(U)] = raw_component_ptr;
166
167 // make sure component is registered before onAttach is called,
168 // in case onAttach implementations query the gameObject for **this** component
169 // Use raw_component_ptr instead of component_ptr since ownership was moved above
170 finalizeAttach(raw_component_ptr);
171
172 return *raw_component_ptr;
173 }
174
175 /**
176 * @brief Adds a pre-constructed component if no component of the same type exists.
177 *
178 * @details If a component of the same runtime type already exists, returns the
179 * existing component and does not take ownership of the passed pointer.
180 * Otherwise, takes ownership and attaches the component.
181 *
182 * Uses the type-indexed componentIndex_ for O(1) amortized lookup complexity.
183 *
184 * @param component Unique pointer to the component to add. Ownership is transferred
185 * only if no component of the same type exists.
186 *
187 * @return Pointer to the existing or newly added component, or nullptr if
188 * the input was nullptr.
189 */
190 helios::engine::ecs::Component* getOrAdd(
191 std::unique_ptr<helios::engine::ecs::Component> component) {
192
193 assert(component != nullptr && "Unexpected nullptr for component.");
194
195 if (!component) {
196 return nullptr;
197 }
198
199 auto* raw_component_ptr = component.get();
200 const std::type_index idx = typeid(*raw_component_ptr);
201
202 if (has(typeid(*raw_component_ptr))) {
203 return componentIndex_[idx];
204 }
205
206 components_.push_back(std::move(component));
207 componentIndex_[idx] = raw_component_ptr;
208
209 // make sure component is registered before onAttach is called,
210 // in case onAttach implementations query the gameObject for **this** component
211 // Use raw_component_ptr instead of component_ptr since ownership was moved above
212 finalizeAttach(raw_component_ptr);
213
214 return raw_component_ptr;
215 }

218 /**
219 * @brief Retrieves an existing component of type T, or creates and adds it if not present.
220 *
221 * Normalizes T and uses the type-indexed componentIndex_ for O(1) lookup complexity. This is a convenience
222 * method that combines get() and add(). If the component already exists, the arguments are ignored.
223 *
224 * @tparam T The type of component to retrieve or add. Must derive from Component.
225 * @tparam Args Argument types for the component's constructor.
226 * @param args Arguments forwarded to the component's constructor if it needs to be created.
227 * @return Reference to the existing or newly created component.
228 */
229 template<typename T, typename... Args>
230 std::remove_cvref_t<T>& getOrAdd(Args&&... args) {
231
232 using U = std::remove_cvref_t<T>;
233 if (auto* cmp = get<U>()) {
234 return *cmp;
235 }
236
237 return add<U>(std::forward<Args>(args)...);
238 }

240 /**
241 * @brief Retrieves a component of type T.
242 *
243 * Normalizes T and uses the type-indexed componentIndex_ for O(1) lookup complexity.
244 *
245 * @tparam T The type of component to retrieve.
246 *
247 * @return Pointer to the component if found, nullptr otherwise.
248 */
249 template<typename T>
250 std::remove_cvref_t<T>* get() const {
251 // remove ref (if provided) and return type refered to by T
252 using U = std::remove_cvref_t<T>;
253 auto it = componentIndex_.find(typeid(U));
254
255 if (it == componentIndex_.end()) {
256 return nullptr;
257 }
258
259 return static_cast<U*>(it->second);
260 }

378 /**
379 * @brief Sets the active state of this GameObject.
380 *
381 * @details When the active state changes, this method notifies all attached
382 * components by calling onActivate() or onDeactivate() accordingly.
383 * If the new state equals the current state, no action is taken. (@see add())
384 *
385 * @param active The new active state.
386 */
387 void setActive(bool active) noexcept {
388 if (isActive_ == active) {
389 return;
390 }
391
392 isActive_ = active;
393
394 if (isActive_) {
395 for (auto& it: components_) {
396 it->onActivate();
397 }
398 } else {
399 for (auto& it: components_) {
400 it->onDeactivate();
401 }
402 }

437 /**
438 * @brief Finalizes attachment of components by calling lifecycle callbacks.
439 *
440 * @details If a specific component is passed, only that component is finalized.
441 * Otherwise, all unattached components are finalized. Calls `onAttach()` followed
442 * by `onActivate()` or `onDeactivate()` based on the GameObject's active state.
443 *
444 * @param component Optional pointer to a specific component to finalize.
445 * If nullptr, all components are processed.
446 */
447 void finalizeAttach(helios::engine::ecs::Component* component = nullptr) {
448
449 // call attach on all components
450 if (component != nullptr) {
451 if (component->isAttached()) {
452 return;
453 }
454 component->onAttach(this);
455
456 if (isActive_) {
457 component->onActivate();
458 } else {
459 component->onDeactivate();
460 }
461
462 return;
463 }
464
465 // call attach on all components
466 for (auto [idx, component] : componentIndex_) {
467 if (component->isAttached()) {
468 return;
469 }
470 component->onAttach(this);
471
472 if (isActive_) {
473 component->onActivate();
474 } else {
475 component->onDeactivate();
476 }
477 }
