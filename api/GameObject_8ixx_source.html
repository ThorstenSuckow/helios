25 /**
26 * @brief Base class for interactive game entities using a component-based architecture.
27 *
28 * @details A GameObject is a container for Component instances that define its behavior.
29 * Each GameObject has a unique Guid for identification in the GameWorld and supports
30 * per-frame updates for components implementing the Updatable interface.
31 *
32 * GameObjects are typically owned by a GameWorld and should be created using
33 * std::make_unique before being added to the world.
34 *
35 * Example usage:
36 * ```cpp
37 * auto entity = std::make_unique<helios::engine::game::GameObject>();
38 * entity->add<SceneNodeComponent>(sceneNode);
39 * entity->add<Move2DComponent>();
40 *
41 * auto* ptr = world.addGameObject(std::move(entity));
42 * ```
43 */
44 class GameObject {

98 /**
99 * @brief Adds a new component of type T to the GameObject.
100 *
101 * The component is created, attached to this GameObject (onAttach is called),
102 * and if it implements Updatable, it is added to the update list.
103 * Normalizes T and uses the type-indexed componentIndex_ for O(1) lookup complexity.
104 *
105 * @tparam T The type of component to add. Must derive from Component.
106 * @tparam Args Argument types for the component's constructor.
107 * @param args Arguments forwarded to the component's constructor.
108 * @return Reference to the newly created component.
109 */
110 template<typename T, typename... Args>
111 T& add(Args&&... args) {
112 using U = std::remove_cvref_t<T>;
113
114 assert(!has<U>() && "Cannot add, Component is already existing.");
115
116 auto component_ptr = std::make_unique<U>(std::forward<Args>(args)...);
117 U* raw_component_ptr = component_ptr.get();
118
119 if constexpr (std::derived_from<U, helios::engine::game::Updatable>) {
120 updatables_.push_back(raw_component_ptr);
121 }
122 components_.push_back(std::move(component_ptr));
123 componentIndex_[typeid(U)] = raw_component_ptr;
124
125 // make sure component is registered before onAttach is called,
126 // in case onAttach implementations query the gameObject for **this** component
127 // Use raw_component_ptr instead of component_ptr since ownership was moved above
128 raw_component_ptr->onAttach(this);
129
130 return *raw_component_ptr;
131 }
132
133 /**
134 * @brief Retrieves an existing component of type T, or creates and adds it if not present.
135 *
136 * Normalizes T and uses the type-indexed componentIndex_ for O(1) lookup complexity. This is a convenience
137 * method that combines get() and add(). If the component already exists, the arguments are ignored.
138 *
139 * @tparam T The type of component to retrieve or add. Must derive from Component.
140 * @tparam Args Argument types for the component's constructor.
141 * @param args Arguments forwarded to the component's constructor if it needs to be created.
142 * @return Reference to the existing or newly created component.
143 */
144 template<typename T, typename... Args>
145 std::remove_cvref_t<T>& getOrAdd(Args&&... args) {
146
147 using U = std::remove_cvref_t<T>;
148 if (auto* cmp = get<U>()) {
149 return *cmp;
150 }
151
152 return add<U>(std::forward<Args>(args)...);
153 }
154
155 /**
156 * @brief Retrieves a component of type T.
157 *
158 * Normalizes T and uses the type-indexed componentIndex_ for O(1) lookup complexity.
159 *
160 * @tparam T The type of component to retrieve.
161 *
162 * @return Pointer to the component if found, nullptr otherwise.
163 */
164 template<typename T>
165 std::remove_cvref_t<T>* get() const {
166 // remove ref (if provided) and return type refered to by T
167 using U = std::remove_cvref_t<T>;
168 auto it = componentIndex_.find(typeid(U));
169
170 if (it == componentIndex_.end()) {
171 return nullptr;
172 }
173
174 return static_cast<U*>(it->second);
175 }
176
177 /**
178 * @brief Checks if the GameObject has a component of type T.
179 *
180 * Normalizes T and uses the type-indexed componentIndex_ for O(1) lookup complexity.
181 *
182 * @tparam T The type of component to check for.
183 *
184 * @return true if the component exists, false otherwise.
185 */
186 template<typename T>
187 [[nodiscard]] bool has() const {
188 using U = std::remove_cvref_t<T>;
189 return componentIndex_.contains(typeid(U));
190 }

223 /**
224 * @brief Sets the active state of this GameObject.
225 *
226 * @details When the active state changes, this method notifies all attached
227 * components by calling onActivate() or onDeactivate() accordingly.
228 * If the new state equals the current state, no action is taken.
229 *
230 * @param active The new active state.
231 */
232 void setActive(bool active) noexcept {
233 if (isActive_ == active) {
234 return;
235 }
236
237 isActive_ = active;
238
239 if (isActive_) {
240 for (auto& it: components_) {
241 it->onActivate();
242 }
243 } else {
244 for (auto& it: components_) {
245 it->onDeactivate();
246 }
247 }
