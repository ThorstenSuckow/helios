51 /**
52 * @brief Central registry for game entities, managers, pools, and the active level.
53 *
54 * @details
55 * The GameWorld is the root container for the game state. It manages the lifecycle
56 * of entities via EntityRegistry/EntityManager, coordinates Managers, and holds
57 * the current Level.
58 *
59 * ## Key Responsibilities
60 *
61 * - **Entity Management:** Creates entities via `addGameObject()` which returns
62 * lightweight (~16 bytes) `GameObject` wrappers. Entities are identified by
63 * versioned `EntityHandle` for stale reference detection.
64 * - **Component Queries:** Efficient iteration over entities with specific components
65 * via `view<Components...>()` with optional filtering via `.whereEnabled()` and
66 * `.exclude<T>()`.
67 * - **Entity Cloning:** Deep-copy entities with all components via `clone()`.
68 * - **Pool Management:** Registers and provides access to SpawnCommandHandlers for
69 * entity recycling.
70 * - **Manager Coordination:** Holds Managers that handle cross-cutting concerns
71 * (spawning, projectile pooling) and flushes them each frame.
72 * - **Level Management:** Holds the active Level instance with arena bounds.
73 *
74 * ## Usage with GameLoop
75 *
76 * The GameWorld is passed to Systems via UpdateContext. Systems query entities
77 * using views, while mutations are typically performed through Commands
78 * that are flushed by the CommandBuffer.
79 *
80 * ```cpp
81 * // In a System
82 * void update(UpdateContext& ctx) noexcept override {
83 * for (auto [entity, transform, velocity, active] : gameWorld_->view<
84 * TransformComponent,
85 * VelocityComponent,
86 * Active
87 * >().whereEnabled()) {
88 * // Process entities with both components
89 * }
90 * }
91 * ```
92 *
93 * ## Entity Lifecycle
94 *
95 * ```cpp
96 * // Create entity
97 * auto player = gameWorld.addGameObject();
98 *
99 * // Add components
100 * player.add<TransformComponent>(position);
101 * player.add<HealthComponent>(100.0f);
102 *
103 * // Activate
104 * player.setActive(true);
105 *
106 * // Find by handle
107 * if (auto entity = gameWorld.find(handle)) {
108 * entity->get<HealthComponent>()->takeDamage(10.0f);
109 * }
110 *
111 * // Clone
112 * auto clone = gameWorld.clone(player);
113 * ```
114 *
115 * @see GameObject
116 * @see EntityHandle
117 * @see EntityManager
118 * @see View
119 * @see UpdateContext
120 * @see Manager
121 * @see Level
122 */

304 /**
305 * @brief Adds and registers a Manager of the specified type.
306 *
307 * @details Creates a new Manager instance and adds it to the world.
308 * The manager's onAdd() callback is invoked after registration.
309 *
310 * @tparam T The Manager type to add. Must derive from Manager.
311 * @tparam Args Constructor argument types.
312 *
313 * @param args Arguments forwarded to the Manager constructor.
314 *
315 * @return Reference to the newly added Manager.
316 *
317 * @pre No Manager of type T is already registered.
318 */
319 template<typename T, typename... Args>
320 requires std::is_base_of_v<helios::engine::runtime::world::Manager, T>
321 T& addManager(Args&&... args) {
322
323 assert(!hasManager<T>() && "Manager already registered.");
324
325 auto manager = std::make_unique<T>(std::forward<Args>(args)...);
326 auto* manager_ptr = manager.get();
327
328 managers_.push_back(std::move(manager));
329
330 manager_ptr->onAdd(*this);
331 return *manager_ptr;
332 }

432 * @details Associates a TypedStateCommandHandler with the given state type.

446 bool registerStateCommandHandler(
447 helios::engine::state::TypedStateCommandHandler<T>& stateCommandHandler
448 ) {

450 const auto idx = typeId.value();

452 if (idx >= stateCommandHandlers_.size()) {
453 stateCommandHandlers_.resize(idx + 1);
