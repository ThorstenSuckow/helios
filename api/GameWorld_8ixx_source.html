45 /**
46 * @brief Central registry for game entities, managers, pools, and the active level.
47 *
48 * @details
49 * The GameWorld is the root container for the game state. It manages the lifecycle
50 * of GameObjects, coordinates Managers, and holds the current Level.
51 *
52 * ## Key Responsibilities
53 *
54 * - **Entity Management:** Owns all GameObjects and provides amortized O(1) lookup by Guid
55 * via `std::unordered_map`.
56 * - **Component Queries:** Efficient iteration over GameObjects with specific components
57 * via `find<Components...>()` with optional filtering by active/enabled state.
58 * - **Pool Management:** Registers and provides access to GameObjectPools for entity recycling.
59 * - **Manager Coordination:** Holds Managers that handle cross-cutting concerns
60 * (spawning, projectile pooling) and flushes them each frame.
61 * - **Level Management:** Holds the active Level instance with arena bounds.
62 *
63 * ## Usage with GameLoop
64 *
65 * The GameWorld is passed to Systems via UpdateContext. Systems query GameObjects
66 * and their components, while mutations are typically performed through Commands
67 * that are flushed by the CommandBuffer.
68 *
69 * ```cpp
70 * // In a System
71 * void update(UpdateContext& ctx) noexcept override {
72 * for (auto [obj, move] : ctx.gameWorld().find<Move2DComponent>().each()) {
73 * // Process entities with Move2DComponent
74 * }
75 * }
76 * ```
77 *
78 * ## Filtering
79 *
80 * The `find()` methods accept a `GameObjectFilter` bitmask to control which
81 * entities are returned:
82 *
83 * - `GameObjectFilter::Active` — Only active GameObjects
84 * - `GameObjectFilter::ComponentEnabled` — Only objects with enabled components
85 *
86 * Default filter is `Active | ComponentEnabled`.
87 *
88 * @see GameObject
89 * @see GameObjectFilter
90 * @see UpdateContext
91 * @see Manager
92 * @see Level
93 */

233 * @brief Adds and registers a Manager of the specified type.
234 *
235 * @details Creates a new Manager instance and adds it to the world.
236 * The manager's onAdd() callback is invoked after registration.
237 *
238 * @tparam T The Manager type to add. Must derive from Manager.
239 * @tparam Args Constructor argument types.
240 *
241 * @param args Arguments forwarded to the Manager constructor.
242 *
243 * @return Reference to the newly added Manager.
244 *
245 * @pre No Manager of type T is already registered.
246 */
247 template<typename T, typename... Args>
248 requires std::is_base_of_v<helios::engine::runtime::world::Manager, T>

250
251 assert(!hasManager<T>() && "Manager already registered.");
252
253 auto manager = std::make_unique<T>(std::forward<Args>(args)...);
254 auto* manager_ptr = manager.get();
255
256 managers_.push_back(std::move(manager));
257
258 manager_ptr->onAdd(*this);
259 return *manager_ptr;

367 /**
368 * @brief Adds a GameObject to the world and transfers ownership.
369 *
370 * @param gameObject The GameObject to add. Ownership is transferred to the GameWorld.
371 *
372 * @return Pointer to the added GameObject, or nullptr if addition failed.
373 *
374 * @retval GameObject* Non-owning pointer to the added object (valid until removed)
375 * @retval nullptr If gameObject was null or a GameObject with the same Guid already exists
376 *
377 * @note The GameWorld takes ownership of the GameObject. The returned pointer remains
378 * valid until the GameObject is removed from the world.
379 * @note Attempting to add a nullptr or a GameObject with a duplicate Guid will fail,
380 * return nullptr, and log a warning.
381 */
382 [[nodiscard]] helios::engine::ecs::GameObject* addGameObject(std::unique_ptr<helios::engine::ecs::GameObject> gameObject) {
383 if (!gameObject) {
384 logger_.warn("Attempted to add null GameObject to GameWorld");
385 return nullptr;
386 }
387 if (gameObjects_.contains(gameObject->guid())) {
388 logger_.warn(std::format("GameObject with Guid {} already exists in GameWorld",
389 gameObject->guid().value()));
390 return nullptr;
391 }
392
393 auto* ptr = gameObject.get();
394 gameObjects_.emplace(gameObject->guid(), std::move(gameObject));
395
396 return ptr;
397 }

418 /**
419 * @brief Finds all GameObjects that have the specified component types.
420 *
421 * @details
422 * Returns a lazy range that filters GameObjects on-the-fly during iteration.
423 * Only GameObjects possessing all specified component types are yielded.
424 *
425 * The query is controlled by a `GameObjectFilter` bitmask:
426 *
427 * | Filter | Effect |
428 * |--------|--------|
429 * | `Active` | Only `obj->isActive() == true` |
430 * | `Inactive` | Only `obj->isActive() == false` |
431 * | `ComponentEnabled` | Only objects where queried components are enabled |
432 * | `ComponentDisabled` | Only objects where queried components are disabled |
433 *
434 * Filters can be combined: `Active | ComponentEnabled` (default).
435 *
436 * Example usage:
437 * ```cpp
438 * // Range-based for loop with default filter (Active + ComponentEnabled)
439 * for (auto* obj : gameWorld.find<Move2DComponent, SceneNodeComponent>()) {
440 * auto* move = obj->get<Move2DComponent>();
441 * move->setVelocity({1.0f, 0.0f, 0.0f});
442 * }
443 *
444 * // Using each() for direct component access (structured bindings)
445 * for (auto [obj, move] : gameWorld.find<Move2DComponent>().each()) {
446 * move.get().setVelocity({1.0f, 0.0f, 0.0f});
447 * }
448 *
449 * // Custom filter: find inactive objects
450 * for (auto* obj : gameWorld.find<HealthComponent>(GameObjectFilter::Inactive)) {
451 * // Process inactive entities (e.g., respawn logic)
452 * }
453 * ```
454 *
455 * @tparam Cs The component types to filter by. GameObjects must have all types.
456 *
457 * @param query Filter bitmask controlling which entities are included.
458 * Default: `Active | ComponentEnabled`.
459 *
460 * @return A GameObjectView object that can be iterated over.
461 *
462 * @see GameObjectFilter
463 * @see GameObjectView
464 */
465 template<class... Cs>
466 [[nodiscard]] auto find(GameObjectFilter query = GameObjectFilter::Active | GameObjectFilter::ComponentEnabled) {
467 return GameObjectRange<Map, Cs...>(gameObjects_, query);
468 }

510 /**
511 * @brief Creates a clone of an existing GameObject.
512 *
513 * @details Clones all Cloneable components from the source GameObject.
514 * The cloned GameObject is initially inactive and receives a new Guid.
515 * Non-cloneable components are skipped with a warning.
516 *
517 * @param gameObject The source GameObject to clone.
518 *
519 * @return Pointer to the newly created clone, or nullptr on failure.
520 *
521 * @note The clone is added to the world in an inactive state.
522 */
523 [[nodiscard]] helios::engine::ecs::GameObject* clone(const helios::engine::ecs::GameObject& gameObject) {
524
525
526 auto newGo = std::make_unique<helios::engine::ecs::GameObject>();
527 /**
528 * @todo Optional ClonePolicy where rules are specified?
529 */
530 newGo->setActive(false);
531
532 for (const auto* component : gameObject.components()) {
533 if (const auto* cc = dynamic_cast<const helios::engine::ecs::Cloneable*>(component)) {
534 auto cComponent = cc->clone();
535 // use getOrAdd since cloned components may have already added
536 // other components in onAttach(), which we will not defer for now.
537 // we assume that the cloned component that allows for adding
538 // components in onAttach is the source of truth for said components
539 newGo->getOrAdd(std::move(cComponent));
540 } else {
541 logger_.warn("Component not cloneable.");
542 }
543
544 }
545
546 return addGameObject(std::move(newGo));
547 }

550 /**
551 * @brief Removes a GameObject from the world and transfers ownership to the caller.
552 *
553 * @param gameObject Reference to the GameObject to remove.
554 *
555 * @return Unique pointer to the removed GameObject, or nullptr if not found.
556 *
557 * @retval std::unique_ptr<GameObject> Ownership of the removed object
558 * @retval nullptr If the GameObject was not found in the world
559 *
560 * @note After successful removal, all non-owning pointers to the GameObject
561 * become invalid unless the caller maintains the returned unique_ptr.
562 * @note Attempting to remove a GameObject that doesn't exist returns nullptr
563 * and logs a warning.
564 */
565 [[nodiscard]] std::unique_ptr<helios::engine::ecs::GameObject> removeGameObject(const helios::engine::ecs::GameObject& gameObject) {
566 auto node = gameObjects_.extract(gameObject.guid());
567
568 if (node.empty()) {
569 logger_.warn(std::format("Attempted to remove non-existent GameObject with Guid {} from GameWorld",
570 gameObject.guid().value()));
571 return nullptr;
572 }
573
574 return std::move(node.mapped());
575 }
