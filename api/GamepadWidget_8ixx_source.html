26 /**
27 * @brief A debug widget to visualize the state of a specific gamepad.
28 *
29 * @details This widget provides a comprehensive visual overview of a connected gamepad's state.
30 * It includes:
31 * - **Connection Status:** Live indicator if the gamepad is connected.
32 * - **Stick Visualization:** 2D plot showing the position of left/right analog sticks within a circular boundary.
33 * - **Triggers:** Progress bars visualizing the pressure level of left/right triggers.
34 * - **Buttons:** Read-only checkboxes showing the current pressed state of all standard gamepad buttons.
35 *
36 * The widget allows switching between different Gamepad IDs (1-4) at runtime via a dropdown menu.
37 *
38 * @note This widget is not thread-safe. Use from the main/render thread only.
39 */

71 /**
72 * @brief Helper method to draw a visual representation of an analog stick.
73 *
74 * Draws a circle representing the boundary and a dot representing the stick's position.
75 *
76 * @param label The label to display below the visualizer.
77 * @param value The normalized (x, y) vector of the stick axis.
78 * @param radius The radius of the visualization circle in pixels.
79 */
80 void drawStickVisualizer(const char* label, const helios::math::vec2f& value, float radius) {
81 ImVec2 p = ImGui::GetCursorScreenPos();
82 ImDrawList* draw_list = ImGui::GetWindowDrawList();
83
84 float center_x = p.x + radius;
85 float center_y = p.y + radius;
86
87 // Draw Background Circle (Dark Grey)
88 draw_list->AddCircleFilled(ImVec2(center_x, center_y), radius, IM_COL32(50, 50, 50, 255));
89 // Draw Boundary (White)
90 draw_list->AddCircle(ImVec2(center_x, center_y), radius, IM_COL32(255, 255, 255, 255));
91
92 // Calculate Dot Position
93 // ImGui coordinates: Y increases downwards.
94 // Standard Gamepad Y: Up is usually positive (or negative depending on normalization,
95 // but we visually want "Up" stick to be "Up" on screen).
96 // Assuming GamepadState provides standard Cartesian (Up=+Y), we subtract Y for screen coords.
97 float dot_x = center_x + (value[0] * radius);
98 float dot_y = center_y - (value[1] * radius);
99
100 // Draw Stick Position Dot (Green)
101 draw_list->AddCircleFilled(ImVec2(dot_x, dot_y), 4.0f, IM_COL32(0, 255, 0, 255));
102
103 // Advance cursor to reserve space for the drawing
104 ImGui::Dummy(ImVec2(radius * 2, radius * 2));
105 ImGui::Text("%s: (%.2f, %.2f)", label, value[0], value[1]);
106 }
