39 /**
40 * @brief Collision detection system using uniform spatial partitioning for Broadphase and
41 * AABB overlaps in the Narrowphase.
42 *
43 * This system implements a grid-based spatial partitioning approach for efficient collision
44 * detection, following the principles outlined in Ericson's "Real-Time Collision Detection"
45 * (Chapter 7). The algorithm divides the world into a uniform 3D grid of cells and assigns
46 * each collidable entity to the cells it overlaps.
47 *
48 * The detection process consists of two phases:
49 * 1. **Broadphase:** Entities are inserted into grid cells based on their AABB. Only entities
50 * sharing the same cell are considered potential collision pairs.
51 * 2. **Narrowphase:** For each cell with multiple candidates, AABB intersection tests
52 * determine actual collisions.
53 *
54 * Collision events are published to the `UpdateContext`'s event sink, distinguishing between:
55 * - **Solid collisions:** Symmetric collisions where both entities can physically interact.
56 * - **Trigger collisions:** Asymmetric collisions for gameplay logic (e.g., pickups, zones).
57 *
58 * During the broadphase, the system uses additional layer masks to filter which entity types
59 * can collide with each other, enabling fine-grained control over collision pairs.
60 *
61 * @see CollisionComponent
62 * @see AabbColliderComponent
63 * @see TriggerCollisionEvent
64 * @see SolidCollisionEvent
65 *
66 * @see [Eri05, Chapter 7]
67 */
68 class GridCollisionDetectionSystem : public System {

85 /**
86 * @brief Hash functor for pairs of GUIDs.
87 *
88 * Enables the use of GUID pairs as keys in unordered containers. The hash combines
89 * both GUIDs using XOR with a bit-shift to reduce collision probability for symmetric pairs.
90 *
91 * @todo switch to uint32_t once helios/#174 is implemented
92 */
93 struct GuidPairHash {
94
95 /**
96 * @brief Computes a hash value for a GUID pair.
97 *
98 * @param pair The pair of GUIDs to hash.
99 *
100 * @return A combined hash value for both GUIDs.
101 */
102 std::size_t operator()(const std::pair<helios::util::Guid, helios::util::Guid>& pair) const {
103
104 auto g1 = std::hash<helios::util::Guid>{}(pair.first);
105 auto g2 = std::hash<helios::util::Guid>{}(pair.second);
106
107 // compute the hash for the pair - shift g2 one position left, then xor with g1.
108 return g1 ^ (g2 << 1);
109 };
110
111 };

206 /**
207 * @brief Initializes the grid based on world bounds and cell size.
208 *
209 * Computes the number of cells needed in each dimension and allocates
210 * the cell storage.
211 */
212 void initGrid() {
213
214 helios::math::vec3f size = gridBounds_.size();
215
216 cellsX_ = std::max(1u, static_cast<unsigned int>(std::ceil(size[0] / cellSize_)));
217 cellsY_ = std::max(1u, static_cast<unsigned int>(std::ceil(size[1] / cellSize_)));
218 cellsZ_ = std::max(1u, static_cast<unsigned int>(std::ceil(size[2] / cellSize_)));
219
220 const size_t cellCount = static_cast<size_t>(cellsX_) * cellsY_ * cellsZ_;
221
222 // this would make 100'000'000 * sizeof(GridCell) Bytes
223 // if we have 24 Bytes per GridCell, we end up with 2,4 GB alone for this spatial grid.
224 if (cellCount > 100'000'000) {
225 logger_.warn(std::format("Spatial Grid requested {0} cells", cellCount));
226 throw std::runtime_error("Cell count too high.");
227 }
228
229 trackedCells_.reserve(cellCount);
230 cells_.resize(cellCount);
231 }

306 /**
307 * @brief Determines the collision type between two entities based on their layer masks.
308 *
309 * Evaluates both solid and trigger collision masks to determine if and how two entities
310 * can collide. Solid collisions are symmetric (both entities must accept collision with
311 * each other), while trigger collisions are asymmetric (either entity can trigger).
312 *
313 * @param cc Collision component of the first entity.
314 * @param matchCC Collision component of the second entity.
315 *
316 * @return CollisionStruct a struct with the requested collision information.
317 */
318 [[nodiscard]] inline CollisionStruct findCollisionType(
319 const helios::engine::game::components::physics::CollisionComponent* cc,
320 const helios::engine::game::components::physics::CollisionComponent* matchCC
321 ) const noexcept {
322
323 auto isSolidCollision = false;
324 auto isTriggerCollision = false;
325
326 auto aIsCollisionReporter = cc->isCollisionReporter();
327 auto bIsCollisionReporter = matchCC->isCollisionReporter();
328
329 // none of the reports a collision as a pair? skip!
330 if (!aIsCollisionReporter && !bIsCollisionReporter) {
331 return CollisionStruct{};
332 }
333
334 bool aCanSolidCollideWithB = (cc->solidCollisionMask() & matchCC->layerId()) != 0;
335 bool bCanSolidCollideWithA = (matchCC->solidCollisionMask() & cc->layerId()) != 0;
336
337 bool aCanTriggerCollideWithB = (cc->triggerCollisionMask() & matchCC->layerId()) != 0;
338 bool bCanTriggerCollideWithA = (matchCC->triggerCollisionMask() & cc->layerId()) != 0;
339
340 // solid collision is treated symmetric
341 isSolidCollision = aCanSolidCollideWithB && bCanSolidCollideWithA;
342
343 #if HELIOS_DEBUG
344 // detect asymmetric solid collision
345 if (aCanSolidCollideWithB && !bCanSolidCollideWithA) {
346 logger_.warn("Collision Asymmetry detected!");
347 }
348 #endif
349
350 // trigger collision is treated asymmetric
351 isTriggerCollision = aCanTriggerCollideWithB || bCanTriggerCollideWithA;
352
353 return CollisionStruct {
354 isSolidCollision,
355 isTriggerCollision,
356 aIsCollisionReporter,
357 bIsCollisionReporter
358 };
359 }

363 /**
364 * @brief Constructs a GridCollisionDetectionSystem with specified bounds and cell size.
365 *
366 * The grid is initialized to cover the given world bounds. Cell size determines
367 * the granularity of spatial partitioning; smaller cells reduce false positives
368 * but increase memory and insertion overhead.
369 *
370 * @param bounds World-space AABB defining the region where collision detection is active.
371 * @param cellSize Size of each grid cell in world units. Must be greater than zero.
372 *
373 * @note Given n objects, the 1/3 rule should be applied for determining the number of cells for partitioning,
374 * i.e. with n objects, the cube should consist of k x k x k cells, with k = n^{1/3} - that is, on average,
375 * one cell fits one object that should be tested for collision. Dimensions of said cells
376 * are derived by taking the reference space into account, e.g. WorldBounds.size/k (see [HS99]).
377 */
378 explicit GridCollisionDetectionSystem(
379 const helios::math::aabbf& bounds,
380 const float cellSize
381 ) : gridBounds_(bounds),
382 cellSize_(cellSize) {
383 assert(cellSize_ > 0.0f && "cellSize must not be <= 0.0f");
384 initGrid();
385 }

387 /**
388 * @brief Performs collision detection for all active entities.
389 *
390 * Executes the complete collision detection pipeline each frame:
391 * 1. Clears all grid cells and resets solved collision pairs.
392 * 2. Iterates all active GameObjects with CollisionComponent and AabbColliderComponent.
393 * 3. Inserts each entity into all grid cells its AABB overlaps.
394 * 4. For each cell with multiple candidates, runs narrow-phase AABB intersection tests.
395 * 5. Publishes collision events (TriggerCollisionEvent, SolidCollisionEvent) to the event queue.
396 *
397 * @param updateContext The update context providing access to GameWorld and event queue.
398 */
399 void update(helios::engine::game::UpdateContext& updateContext) noexcept override {
400
401 prepareCollisionDetection();
402
403 // only consider enabled CollisionComponents
404 constexpr auto filter = helios::engine::game::GameObjectFilter::Active | helios::engine::game::GameObjectFilter::ComponentEnabled;
405
406 for (auto [entity, cc, acc] : gameWorld_->find<
407 helios::engine::game::components::physics::CollisionComponent,
408 helios::engine::game::components::physics::AabbColliderComponent
409
410 >(filter).each()) {
411
412 if (!acc->boundsInitialized()) {
413 continue;
414 }
415
416 if (!cc->isEnabled() || (cc->triggerCollisionMask() == 0 && cc->solidCollisionMask() == 0)) {
417 continue;
418 }
419
420 const auto& worldBounds = acc->bounds();
421
422 updateCollisionCandidate(
423 entity,
424 worldBoundsToGridBounds(worldBounds),
425 acc,
426 cc
427 );
428 }
429
430 for (const auto idx : trackedCells_) {
431
432 if (cells_[idx].collisionCandidates.size() < 2) {
433 continue;
434 }
435
436 solveCell(cells_[idx], updateContext);
437 }
438 }

441 /**
442 * @brief Converts world-space AABB bounds to grid cell indices.
443 *
444 * Transforms a floating-point AABB in world space to integer cell coordinates,
445 * clamped to the valid grid range. Used to determine which cells an entity occupies.
446 *
447 * @param aabbf The world-space AABB to convert.
448 *
449 * @return An integer AABB representing the range of grid cell indices the entity spans.
450 */
451 [[nodiscard]] helios::math::aabbi worldBoundsToGridBounds(const helios::math::aabbf& aabbf) const noexcept {
452
453 helios::math::vec3f min = aabbf.min() - gridBounds_.min();
454 helios::math::vec3f max = aabbf.max() - gridBounds_.min();
455
456 int xMin = static_cast<int>(std::floor(min[0] / cellSize_));
457 int yMin = static_cast<int>(std::floor(min[1] / cellSize_));
458 int zMin = static_cast<int>(std::floor(min[2] / cellSize_));
459
460 int xMax = static_cast<int>(std::floor(max[0] / cellSize_));
461 int yMax = static_cast<int>(std::floor(max[1] / cellSize_));
462 int zMax = static_cast<int>(std::floor(max[2] / cellSize_));
463
464 return helios::math::aabbi{
465 std::clamp(xMin, 0, static_cast<int>(cellsX_ - 1)),
466 std::clamp(yMin, 0, static_cast<int>(cellsY_ - 1)),
467 std::clamp(zMin, 0, static_cast<int>(cellsZ_ - 1)),
468 std::clamp(xMax, 0, static_cast<int>(cellsX_ - 1)),
469 std::clamp(yMax, 0, static_cast<int>(cellsY_ - 1)),
470 std::clamp(zMax, 0, static_cast<int>(cellsZ_ - 1)),
471 };
472 }

486 inline void updateCollisionCandidate(
487 helios::engine::game::GameObject* go,
488 const helios::math::aabbi& bounds,
489 helios::engine::game::components::physics::AabbColliderComponent* aabbColliderComponent,
490 helios::engine::game::components::physics::CollisionComponent* collisionComponent
491 ) {

493 const auto xMax = bounds.max()[0];
494 const auto yMin = bounds.min()[1];

496 const auto zMin = bounds.min()[2];
497 const auto zMax = bounds.max()[2];
498
499 for (int x = xMin; x <= xMax; x++) {

524 /**
525 * @brief Performs narrow-phase collision detection for all candidates in a cell.
526 *
527 * Tests all unique pairs of collision candidates within the cell using AABB intersection.
528 * For each collision detected, determines collision type (solid/trigger) based on layer
529 * masks and publishes appropriate events. Uses a solved-set to avoid duplicate events
530 * when entities span multiple cells.
531 *
532 * @param cell The grid cell containing collision candidates to test.
533 * @param updateContext Context for pushing collision events to the event queue.
534 */
535 inline void solveCell(GridCell& cell, helios::engine::game::UpdateContext& updateContext) {
536
537 auto& candidates = cell.collisionCandidates;
538
539 // we will skip this cell if none of the candidates reports a collision
540 bool isCollisionReporter = false;
541 for (const auto& candidate : candidates) {
542 if (candidate.collisionComponent->isCollisionReporter()) {
543 isCollisionReporter = true;
544 break;
545 }
546 }
547
548 if (!isCollisionReporter) {
549 return;
550 }
551
552 for (size_t i = 0; i < candidates.size(); i++) {
553
554 CollisionCandidate& candidate = candidates[i];
555 helios::engine::game::components::physics::CollisionComponent* cc = candidate.collisionComponent;
556
557 const helios::math::aabbf& aabbCandidate = candidate.aabbColliderComponent->bounds();
558
559 for (size_t j = i+1; j < candidates.size(); j++) {
560
561 auto& [gameObject, aabbColliderComponent, collisionComponent] = candidates[j];
562
563 helios::engine::game::components::physics::CollisionComponent* matchCC = collisionComponent;
564
565 const auto collisionStruct = findCollisionType(cc, matchCC);
566
567 if (!collisionStruct.hasAnyInteraction()) {
568 continue;
569 }
570
571 const helios::math::aabbf& aabbMatch = aabbColliderComponent->bounds();
572 if (!aabbCandidate.intersects(aabbMatch)) {
573 continue;
574 }
575
576 auto lGuid = candidate.gameObject->guid();
577 auto rGuid = gameObject->guid();
578
579 if (lGuid > rGuid) {
580 std::swap(lGuid, rGuid);
581 }
582
583 if (solvedCollisions_.contains({lGuid, rGuid})) {
584 continue;
585 }
586
587 solvedCollisions_.insert({lGuid, rGuid});
588
589 postEvent(
590 candidate.gameObject, gameObject,
591 helios::math::overlapCenter(aabbCandidate, aabbMatch),
592 collisionStruct.aIsCollisionReporter, collisionStruct.bIsCollisionReporter,
593 collisionStruct.isSolidCollision, collisionStruct.isTriggerCollision,
594 updateContext
595 );
596 }
597 }
598 }
