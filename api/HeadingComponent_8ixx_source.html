113 * @details Updates the heading input vector and turn intensity. If the

116 *
117 * @param direction Normalized direction vector representing the desired heading.

121 */
122 void setHeading(helios::math::vec3f direction, float turnIntensity) {
123
124 headingInput_ = direction;
125 turnIntensity_ = turnIntensity;
126 isInputActive_ = true;
127
128 if (turnIntensity_ <= helios::math::EPSILON_LENGTH) {
129 headingInput_ = {0.0f, 0.0f, 0.0f};
130 isInputActive_ = false;
131 turnIntensity_ = 0.0f;
132 return;
133 }
134
135 assert(std::abs(direction.length() - 1.0f) <= 0.001f && "Unexpected direction vector - not normalized");
