40 /**
41 * @brief Performs the rotation physics update for a HeadingComponent.
42 *
43 * @details Applies dampening when input is inactive, calculates rotation steps,
44 * and snaps to target when within threshold.
45 *
46 * @param cmp The HeadingComponent to update. Must not be nullptr.
47 * @param deltaTime Time elapsed since last frame in seconds.
48 */
49 void updateHeading(
50 helios::engine::game::components::physics::HeadingComponent* cmp,
51 float deltaTime
52 ) noexcept {
53
54 assert(cmp != nullptr && "Unexpected invalid HeadingComponent passed");
55 assert(deltaTime >= 0 && "Unexpected negative value for deltaTime");
56
57 if (deltaTime == 0) {
58 return;
59 }
60
61 float currentRotationSpeed = cmp->currentRotationSpeed();
62 float currentRotationAngle = cmp->currentRotationAngle();
63 float rotationAngleDelta = cmp->rotationAngleDelta();
64 bool isInputActive = cmp->isInputActive();
65
66 // Apply rotation dampening when input is inactive
67 if (!isInputActive) {
68 currentRotationSpeed *= std::pow(cmp->rotationDampening(), deltaTime);
69
70 if (currentRotationSpeed < cmp->rotationSpeedThreshold()) {
71 currentRotationSpeed = 0.0f;
72 }
73
74 cmp->setCurrentRotationSpeed(currentRotationSpeed);
75 }
76
77 // Apply rotation step towards target angle
78 if (currentRotationSpeed > 0.0f) {
79 float rotationStep = std::copysign(
80 std::min(
81 std::abs(rotationAngleDelta),
82 currentRotationSpeed * deltaTime
83 ),
84 rotationAngleDelta
85 );
86 currentRotationAngle = currentRotationAngle + rotationStep;
87 rotationAngleDelta = std::fmod((cmp->targetRotationAngle() - currentRotationAngle) + 540.0f, 360.0f) - 180.0f;
88
89 // Snap to target when close enough
90 if (std::abs(rotationAngleDelta) <= 0.5f) {
91 currentRotationAngle = cmp->targetRotationAngle();
92 rotationAngleDelta = 0.0f;
93 currentRotationSpeed = 0.0f;
94 }
95
96 cmp->setCurrentRotationAngle(currentRotationAngle);
97 cmp->setRotationAngleDelta(rotationAngleDelta);
98 cmp->setCurrentRotationSpeed(currentRotationSpeed);
99 }

113
114 for (auto [entity, hc, rsc, dc] : gameWorld_->find<
115 helios::engine::game::components::physics::HeadingComponent,
116 helios::engine::game::components::physics::RotationStateComponent,

118 >().each()) {
119
120 if (hc->isInputActive()) {
121 auto direction = hc->headingInput();
122 auto targetRotationAngle = helios::math::degrees(std::atan2(direction[1], direction[0]));
123 auto rotationAngleDelta = std::fmod((targetRotationAngle - hc->currentRotationAngle()) + 540.0f, 360.0f) - 180.0f;
124
125 float turnBoost = 1.0f + 0.5f*std::clamp((abs(rotationAngleDelta))/180.f, 0.0f, 1.0f);
126 float currentRotationSpeed = turnBoost * hc->rotationSpeed() * hc->turnIntensity();
127
128 hc->setTargetRotationAngle(targetRotationAngle);
129 hc->setRotationAngleDelta(rotationAngleDelta);
130 hc->setCurrentRotationSpeed(currentRotationSpeed);
131 }
132
133
134 updateHeading(hc, updateContext.deltaTime());
135
136 float rotationAngle = hc->currentRotationAngle();
137 rsc->setHeadingRotationAngle(rotationAngle);
138 rsc->setHeadingRotationAxis(hc->rotationAxis());

140 float rad = helios::math::radians(rotationAngle);
141 /**
