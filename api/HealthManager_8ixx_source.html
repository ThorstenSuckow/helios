41 /**
42 * @brief Manager that processes damage commands and updates entity health.
43 *
44 * HealthManager implements the Manager interface for integration into the
45 * game loop flush cycle, and the TypedCommandHandler interface for
46 * ApplyDamageCommand so that incoming damage is queued and applied in batch.
47 *
48 * During flush(), each queued DamageContext is resolved: the target's
49 * HealthComponent is damaged, a HealthChangedEvent is emitted, and if
50 * health reaches zero a HealthDepletedEvent is pushed together with an
51 * optional DeadTagComponent.
52 *
53 * Event phases:
54 * - HealthChangedEvent is pushed to the **phase** bus
55 * (available to systems in the next phase).
56 * - HealthDepletedEvent is pushed to the **pass** bus
57 * (available to systems after commit within the same pass).
58 *
59 * @see ApplyDamageCommand
60 * @see HealthComponent
61 * @see HealthDepletedEvent
62 * @see Manager
63 */
64 class HealthManager : public Manager, public TypedCommandHandler<ApplyDamageCommand> {
65

67 * @brief Logger scoped to HealthManager.

70 HELIOS_LOG_SCOPE);

72 /**
73 * @brief Pending damage contexts collected via submit().
74 */
75 std::vector<DamageContext> damageContexts_;

77 public:

79 /**
80 * @brief Applies all queued damage and emits health events.
81 *
82 * For each pending DamageContext the target's HealthComponent is
83 * looked up and damaged. A HealthChangedEvent is pushed per hit.
84 * When health transitions from alive to dead, the configured
85 * HealthDepletedBehavior flags are evaluated and a
86 * HealthDepletedEvent is emitted.
87 *
88 * HealthChangedEvent is pushed to the phase bus (available next phase);
89 * HealthDepletedEvent is pushed to the pass bus (available after
90 * commit within the same pass).
91 *
92 * @param updateContext The current frame's update context.
93 */
94 void flush(UpdateContext& updateContext) noexcept override {
95
96 for (const auto& damageContext : damageContexts_) {
97
98 auto interactionContext = damageContext.interactionContext;
99
100 auto target = updateContext.find(interactionContext.target);
101
102 auto* hc = target->get<HealthComponent>();
103 if (!hc) {
104 continue;
105 }
106
107 bool wasAlive = hc->isAlive();
108 auto damageApplied = damageContext.damage;
109 hc->takeDamage(damageApplied);
110
111 auto* lac = target->get<LastDamageComponent>();
112 if (lac && damageApplied > 0) {
113 lac->setDamageContext(damageContext);
114 }
115
116 auto huc = HealthChangeContext{
117 interactionContext,
118 -damageApplied
119 };
120
121 updateContext.pushPhase<HealthChangedEvent>(huc);
122
123 if (wasAlive && !hc->isAlive()) {
124 auto healthDepletedBehavior = hc->healthDepletedBehavior();
125
126 if (hasHealthDepletedFlag(healthDepletedBehavior, HealthDepletedBehavior::DeadTag)) {
127 target->add<DeadTagComponent>();
128 }
129
130 updateContext.pushPass<HealthDepletedEvent>(target->entityHandle(), damageContext);
131 }
132 }
133
134 damageContexts_.clear();
135 }
136
137 /**
138 * @brief Enqueues a damage command for deferred processing.

141 *
142 * @return True if the command was accepted.
143 */
144 bool submit(

146 ) noexcept override {
147 damageContexts_.push_back(applyDamageCommand.damageContext());
148 return true;
149 };
