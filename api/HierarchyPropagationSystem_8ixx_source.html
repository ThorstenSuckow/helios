33 /**
34 * @brief Propagates activation state from parent to child entities.
35 *
36 * @details This system processes entities with HierarchyComponent and
37 * propagates their active/inactive state to all descendants. Only root
38 * entities (those without a parent) that are marked dirty trigger
39 * propagation.
40 *
41 * The system runs in two passes:
42 * - Active roots: propagate active state to children
43 * - Inactive roots: propagate inactive state to children
44 */
45 class HierarchyPropagationSystem : public helios::engine::ecs::System {

47 /**
48 * @brief Recursively updates activation state for an entity and its children.
49 *
50 * @param go The entity to update.
51 * @param active The activation state to apply.
52 */
53 void updateEntityHierarchy(GameObject go, bool active) {
54
55 auto* hc = go.get<helios::engine::ecs::components::HierarchyComponent>();
56 if (!hc) {
57 return;
58 }
59 hc->clearDirty();
60 for (auto child : hc->children()) {
61 if (auto ent = gameWorld_->find(child)) {
62 ent->setActive(active);
63 updateEntityHierarchy(ent.value(), active);
64 }
65 }
66 }
