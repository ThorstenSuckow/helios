43 /**
44 * @brief Controls menu visibility based on game and match state.
45 *
46 * @details Uses a CombinedStateToIdMapPair to determine which menus
47 * should be active for the current state combination. Automatically
48 * shows/hides menus and manages focus when states change.
49 *
50 * ## Behavior
51 *
52 * - Menus not associated with the current state are hidden
53 * - Menus associated with the current state are shown
54 * - When no item has focus, the first menu's default item is focused
55 * - Inactive focused items have their UiFocusComponent removed
56 *
57 * @see CombinedStateToIdMapPair
58 * @see MenuComponent
59 * @see MenuNavigationSystem
60 */

62 class MenuDisplaySystem : public helios::engine::ecs::System {

100 void focusMenu(const MenuId menuId, View<MenuComponent>& components) {
101 for (auto [entity, mc] : components) {
102 if (mc->menuId() == menuId) {
103 mc->selectDefaultIndex();
104
105 if (mc->menuItems().size() > mc->selectedIndex()) {
106 auto menuItem = gameWorld_->find(mc->menuItems()[mc->selectedIndex()]);
107 menuItem->getOrAdd<UiFocusComponent>();
108 }
109 break;
110 }
111 }

128 /**
129 * @brief Updates menu visibility based on current state.
130 *
131 * @details Queries the current game and match state, then:
132 * 1. Hides menus that are no longer active
133 * 2. Shows menus for the current state
134 * 3. Ensures at least one menu item has focus
135 *
136 * @param updateContext The current frame's update context.
137 */
138 void update(helios::engine::runtime::world::UpdateContext& updateContext) noexcept override {
139
140
141 auto& session = updateContext.session();
142
143 auto gameState = session.state<StateLft>();
144 auto matchState = session.state<StateRgt>();
145
146 auto menuIds = stateToMenuMap_.ids(gameState, matchState);
147
148 if (menuIds.empty() && prevMenuIds_.empty()) {
149 return;
150 }
151
152 View<MenuComponent> components = gameWorld_->view<MenuComponent>().whereEnabled();
153
154
155 for (auto& prevMenuId : prevMenuIds_) {
156 if (std::ranges::find(menuIds, prevMenuId) == menuIds.end()) {
157 showMenu(prevMenuId, false, components);
158 }
159 }
160
161 prevMenuIds_.assign(menuIds.begin(), menuIds.end());
162 for (auto& menuId : menuIds) {
163 showMenu(menuId, true,components);
164 }
165
166 if (menuIds.empty()) {
167 return;
168 }
169
170
171 bool hasFocus = false;
172 inactiveItems_.clear();
173
174 for (auto [entity, fc] : gameWorld_->view<UiFocusComponent>().whereEnabled()) {
175
176 /**
177 * @todo Check if item is child element of menu?
178 */
179 if (entity.isActive()) {
180 hasFocus = true;
181 break;
182 }
183
184 /**
185 * @todo guaranteee that the UiFocusComponent is exclusive,
186 * break here if first comp with focus was found.
187 */
188 inactiveItems_.push_back(entity);
189 }
190
191 for (auto& item : inactiveItems_) {
192 item.remove<UiFocusComponent>();
193 }
194
195 // focus an item from the first menu
196 if (!hasFocus) {
197 focusMenu(menuIds[0], components);
198 }
199
200 }
201
202 };
