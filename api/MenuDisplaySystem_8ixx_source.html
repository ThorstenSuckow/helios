39 /**
40 * @brief Controls menu visibility based on session UI focus state.
41 *
42 * @details This system monitors the session's focused entity and shows/hides
43 * menus accordingly. When the focused entity changes:
44 * - The previously focused menu (if any) is hidden
45 * - The newly focused menu is shown along with its menu items
46 *
47 * The system tracks the previous focused entity to detect focus changes
48 * and avoid redundant visibility updates.
49 *
50 * @see MenuComponent
51 * @see UiFocusComponent
52 * @see Session::focusedEntity()
53 */

61 /**

68 void showMenu(

90 /**
91 * @brief Updates menu visibility based on focus changes.
92 *
93 * @details Compares the current focused entity with the previous one.
94 * If focus changed, hides the previous menu and shows the new one.
95 * Skips processing if focus remains unchanged.
96 *
97 * @param updateContext The current frame's update context.
98 */
99 void update(helios::engine::runtime::world::UpdateContext& updateContext) noexcept override {
100
101
102 auto focusedEntity = updateContext.gameWorld().session().focusedEntity();
103
104 if (focusedEntity && focusedEntity->entityHandle() == prevEntityHandle_) {
105 return;
106 }
107
108 if (prevEntityHandle_.isValid()) {
109 auto go = updateContext.gameWorld().find(prevEntityHandle_);
110 if (go) {
111 showMenu(go.value(), false,updateContext.gameWorld());
112 }
113 prevEntityHandle_ = helios::engine::ecs::EntityHandle();
114 }
115
116 if (!focusedEntity) {
117 return;
118 }
119
120 prevEntityHandle_ = focusedEntity->entityHandle();
121
122
123 auto* mc = focusedEntity->get<helios::engine::modules::ui::widgets::components::MenuComponent>();
124
125 if (!mc) {
126 return;
127 }
128
129 showMenu(focusedEntity.value(), true, updateContext.gameWorld());
130
131 }
