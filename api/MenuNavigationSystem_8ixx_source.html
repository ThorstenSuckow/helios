78 void updateMenu(helios::engine::runtime::world::UpdateContext& updateContext, MenuComponent* mc, const size_t index) {
79
80 auto menuItems = mc->menuItems();
81
82 const auto prevIndex = mc->previousSelectedIndex();
83
84 if (prevIndex == index && !mc->isDirty()) {
85 return;

88 UiStateComponent* usc = nullptr;

92 if (usc) {
93 usc->setSelected(false);

96 mc->setSelectedIndex(index);

98 // update ui state
99 usc = gameWorld_->find(menuItems[index])->get<UiStateComponent>();

110 /**
111 * @brief Processes menu navigation input.
112 *
113 * @details Finds the currently focused menu item, handles Up/Down
114 * navigation, and triggers UiActionCommand when the A button is
115 * pressed on an item with a UiActionComponent.
116 *
117 * @param updateContext The current frame's update context.
118 */
119 void update(helios::engine::runtime::world::UpdateContext& updateContext) noexcept override {
120
121 MenuComponent* focusedMenu = nullptr;
122
123 for (auto [entity, fc, hc, active] : gameWorld_->view<
124 UiFocusComponent, HierarchyComponent, Active
125 >().whereEnabled()) {
126 assert(hc->parent() && gameWorld_->find(*hc->parent()) && gameWorld_->find(*hc->parent())->get<MenuComponent>() && "Item expected to have parent menu component.");
127 focusedMenu = gameWorld_->find(*hc->parent())->get<MenuComponent>();
128 break;
129 }
130
131 if (!focusedMenu) {
132 return;
133 }
134
135 const auto size = focusedMenu->menuItems().size();
136
137 const auto& gamepadState = updateContext.inputSnapshot().gamepadState();
138
139 if (gamepadState.isButtonPressed(GamepadInput::Up)) {
140
141 if (focusedMenu->selectedIndex() >= 1) {
142 updateMenu(updateContext, focusedMenu, focusedMenu->selectedIndex() - 1);
143 } else {
144 updateMenu(updateContext, focusedMenu, 0);
145 }
146
147 } else if (gamepadState.isButtonPressed(GamepadInput::Down)) {
148
149 if (focusedMenu->selectedIndex() + 1 < size) {
150 updateMenu(updateContext, focusedMenu, focusedMenu->selectedIndex() + 1);
151 } else {
152 updateMenu(updateContext, focusedMenu, size - 1);
153 }
154
155 } else {
156 updateMenu(updateContext, focusedMenu, focusedMenu->selectedIndex());
157 }
158
159 if (gamepadState.isButtonPressed(GamepadInput::A)) {
160 auto* uac = gameWorld_->find(focusedMenu->menuItems()[focusedMenu->selectedIndex()])->get<
161 helios::engine::modules::ui::widgets::components::UiActionComponent
162 >();
163
164 if (uac) {
165 updateContext.commandBuffer().add<helios::engine::modules::ui::commands::UiActionCommand>(
166 focusedMenu->menuItems()[focusedMenu->selectedIndex()], uac->actionId()
167 );
168 }
169 }
170 }
