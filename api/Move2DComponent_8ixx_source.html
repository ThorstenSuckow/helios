28 /**
29 * @brief Component for 2D physics-based movement with rotation.
30 *
31 * @details Handles directional movement and rotation for GameObjects in a 2D plane.
32 * Implements smooth acceleration, deceleration with exponential dampening, and
33 * rotation towards the input direction. Movement is applied to the associated
34 * SceneNode through a SceneNodeComponent.
35 *
36 * The component receives input via move() which sets target direction and throttle.
37 * Each frame, update() applies physics calculations to smoothly rotate towards
38 * the target direction and move in the current facing direction.
39 *
40 * @note Requires a SceneNodeComponent to be attached to the same GameObject.

206 /**
207 * @brief Sets the movement direction and throttle.
208 *
209 * @details Updates the target rotation angle based on the input direction
210 * and calculates the shortest angular distance to the target.
211 *
212 * @param direction Normalized 2D direction vector from analog stick.
213 * @param throttle Magnitude of the stick input (0.0 to 1.0).
214 */
215 void move(helios::math::vec2f direction, float throttle) {
216 if (throttle <= helios::math::EPSILON_LENGTH) {
217 steeringInput_ = helios::math::vec2f{0.0f, 0.0f};
218 throttle_ = 0.0f;
219 isInputActive_ = false;
220 return;
221 }
222
223 steeringInput_ = direction;
224 throttle_ = throttle;
225
226 assert(std::abs(direction.length() - 1.0f) <= 0.001f && "Unexpected direction vector - not normalized");
227
228
229 targetRotationAngle_ = helios::math::degrees(std::atan2(direction[1], direction[0]));
230 rotationAngleDelta_ = std::fmod((targetRotationAngle_ - actualRotationAngle_) + 540.0f, 360.0f) - 180.0f;
231
232
233 isInputActive_ = true;
234
235 float turnBoost = 1.0f + 0.5f*std::clamp((abs(rotationAngleDelta_))/180.f, 0.0f, 1.0f);
236 actualRotationSpeed_ = turnBoost * rotationSpeed_ * throttle_;
237 actualMovementSpeed_ = movementSpeed_ * throttle_;
238
239 }
240
241 /**
242 * @brief Updates movement and rotation each frame.
243 *
244 * @details Applies rotation towards the target angle with dampening, then
245 * integrates velocity to update position. Handles both active input and
246 * coasting (when input stops) with exponential decay.
247 *
248 * @param updateContext Context containing frame delta time.
249 */
250 void update(helios::engine::game::UpdateContext& updateContext) noexcept override {
251
252 float deltaTime = updateContext.deltaTime();
253
254 assert(deltaTime >= 0 && "Unexpected negative value for deltaTime");
255
256
257
258 if (deltaTime == 0) {
259 return;
260 }
261
262 // ROTATION
263 if (!isInputActive_) {
264 actualRotationSpeed_ *= std::pow(rotationDampening_, deltaTime);;
265 if (actualRotationSpeed_ < rotationSpeedThreshold_) {
266 actualRotationSpeed_ = 0.0f;
267 }
268 }
269
270 if (actualRotationSpeed_ > 0.0f) {
271 float rotationStep = std::copysign(
272 std::min(std::abs(rotationAngleDelta_), actualRotationSpeed_ * deltaTime),
273 rotationAngleDelta_
274 );
275 actualRotationAngle_ += rotationStep;
276
277 rotationAngleDelta_ = std::fmod((targetRotationAngle_ - actualRotationAngle_) + 540.0f, 360.0f) - 180.0f;
278
279 if (std::abs(rotationAngleDelta_) <= 0.5f) {
280 actualRotationAngle_ = targetRotationAngle_;
281 rotationAngleDelta_ = 0.0f;
282 actualRotationSpeed_ = 0.0f;
283 }
284
285 if (sceneNodeComponent_) {
286 sceneNodeComponent_->setRotation(helios::math::rotate(
287 helios::math::mat4f::identity(),
288 helios::math::radians(actualRotationAngle_),
289 helios::math::vec3f(0.0f, 0.0f, 1.0f)
290 ));
291 }
292
293
294 }
295
296 // MOVEMENT
297 // Compute the current facing direction from the actual rotation angle.
298 // This converts the angle to a unit vector in the XY plane.
299 const auto actualDirection_ = helios::math::vec3f(
300 cos(helios::math::radians(actualRotationAngle_)), sin(helios::math::radians(actualRotationAngle_)), 0.0f
301 );
302
303 if (!isInputActive_) {
304 // Apply exponential drag when no input is active.
305 // This creates a smooth deceleration effect (velocity approaches zero over time).
306 const float drag = std::pow(movementDampening_, deltaTime);
307 velocity_ = velocity_ * drag;
308
309 if (drag <= helios::math::EPSILON_LENGTH) {
310 velocity_ = helios::math::vec3f{0.0f, 0.0f, 0.0f};
311 }
312
313 } else {
314 // Accelerate in the current facing direction.
315 // Uses throttle (input intensity) to scale acceleration.
316 velocity_ = velocity_ + actualDirection_ * (movementAcceleration_ * throttle_ * deltaTime);
317 }
318
319 // Clamp velocity to maximum speed to prevent unlimited acceleration.
320 if (velocity_.length() > movementSpeed_) {
321 velocity_ = velocity_.normalize() * movementSpeed_;
322 }
323
324
325 if (sceneNodeComponent_) {
326 // Integrate velocity to update position.
327 sceneNodeComponent_->setTranslation(sceneNodeComponent_->translation() + velocity_ * deltaTime);
328 }
