47 /**
48 * @brief Computes translation delta for an entity based on its Move2DComponent state.
49 *
50 * @details Calculates velocity changes based on input state. When input is active,
51 * accelerates in the facing direction. When inactive, applies exponential
52 * drag to slow down. Velocity is clamped to maximum speed.
53 *
54 * @param cmp Pointer to the Move2DComponent.
55 * @param currentDirection The current direction vector from DirectionComponent.
56 * @param deltaTime Frame delta time in seconds.
57 *
58 * @return Translation delta to apply to the entity this frame.
59 */
60 [[nodiscard]] static helios::math::vec3f moveGameObject(
61 helios::engine::game::components::physics::Move2DComponent* cmp,
62 helios::math::vec3f currentDirection,
63 float deltaTime
64 ) noexcept {
65
66 assert(cmp != nullptr && "Unexpected invalid Move2DComponent passed");
67 assert(deltaTime >= 0 && "Unexpected negative value for deltaTime");
68
69 if (deltaTime == 0) {
70 return helios::math::vec3f{0.0f};
71 }
72
73 bool isInputActive = cmp->isInputActive();
74
75
76 auto velocity = cmp->velocity();
77
78 if (!isInputActive) {
79 float movementDampening = cmp->movementDampening();
80
81 // Apply exponential drag when no input is active.
82 // This creates a smooth deceleration effect (velocity approaches zero over time).
83 const float drag = std::pow(movementDampening, deltaTime);
84 velocity = velocity * drag;
85
86 if (velocity.length() <= helios::math::EPSILON_LENGTH) {
87 velocity = {0.0f, 0.0f, 0.0f};
88 }
89
90 } else {
91
92 // Accelerate in the current facing direction.
93 // Uses throttle (input intensity) to scale acceleration.
94 velocity = velocity + currentDirection * (cmp->movementAcceleration() * cmp->throttle() * deltaTime);
95 }
96
97 // Clamp velocity to maximum speed to prevent unlimited acceleration.
98 if (velocity.length() > cmp->movementSpeed()) {
99 velocity = velocity.normalize() * cmp->movementSpeed();
100 }
101
102 cmp->setVelocity(velocity);
103
104 return velocity * deltaTime;
105 }
