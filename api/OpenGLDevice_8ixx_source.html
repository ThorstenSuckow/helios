34 /**
35 * @brief OpenGL implementation of `RenderingDevice`.
36 *
37 * `OpenGLDevice` is the concrete OpenGL backend for the helios rendering system.
38 * It handles all OpenGL-specific operations including mesh rendering, shader management,
39 * and text rendering via `OpenGLGlyphTextRenderer`.
40 *
41 * ## Responsibilities
42 *
43 * - **Initialization:** Load OpenGL function pointers via GLAD.
44 * - **Render Pass Execution:** Configure viewport, clear buffers, and process render commands.
45 * - **Mesh Rendering:** Bind VAOs and issue draw calls for geometry.
46 * - **Text Rendering:** Delegate text commands to `OpenGLGlyphTextRenderer`.
47 *
48 * ## Usage
49 *
50 * ```cpp
51 * auto textRenderer = std::make_unique<OpenGLGlyphTextRenderer>();
52 * auto device = std::make_unique<OpenGLDevice>(std::move(textRenderer));
53 *
54 * // After creating OpenGL context
55 * device->init();
56 *
57 * // Load fonts via the text renderer
58 * device->textRenderer().addFontFamily(FontId{1}, "fonts/arial.ttf");
59 * ```
60 *
61 * @see RenderingDevice
62 * @see OpenGLGlyphTextRenderer
63 */
64 class OpenGLDevice : public helios::rendering::RenderingDevice {
65
66 /**
67 * @brief Translates helios abstract PrimitiveType enum to its corresponding OpenGL GLenum value.
68 *
69 * This is a utility function used exclusively by the OpenGLDevice to interpret the
70 * API-agnostic primitive type requested by the helios configuration.
71 *
72 * @param primitiveType The API-agnostic PrimitiveType value.
73 *
74 * @return The corresponding OpenGL primitive type as a GLenum. If mapping does not
75 * succeed, the function falls back to GL_TRIANGLES.
76 */
77 [[nodiscard]] GLenum toOpenGL(helios::rendering::model::config::PrimitiveType primitiveType) const noexcept {
78 switch (primitiveType) {
79 case helios::rendering::model::config::PrimitiveType::Points:
80 return GL_POINTS;
81 case helios::rendering::model::config::PrimitiveType::Lines:
82 return GL_LINES;
83 case helios::rendering::model::config::PrimitiveType::LineLoop:
84 return GL_LINE_LOOP;
85 case helios::rendering::model::config::PrimitiveType::LineStrip:
86 return GL_LINE_STRIP;
87 case helios::rendering::model::config::PrimitiveType::Triangles:
88 return GL_TRIANGLES;
89 case helios::rendering::model::config::PrimitiveType::TriangleStrip:
90 return GL_TRIANGLE_STRIP;
91 case helios::rendering::model::config::PrimitiveType::TriangleFan:
92 return GL_TRIANGLE_FAN;
93 default:
94 logger_.warn("Failed to resolve primitive type, falling back to GL_TRIANGLES.");
95 return GL_TRIANGLES;
96 }
97 }

119 /**
120 * @brief Initializes the OpenGL device to access modern OpenGL.
121 *
122 * This method must be called **after** the helios application's `setCurrent()` was called
123 * for creating a current context. This OpenGL device will then load from this context.
124 *
125 * This method uses the `gladLoadGL` method in conjunction with `glfwGetProcAddress`,
126 * which returns the address of the specific OpenGL core.
127 *
128 * @see https://www.glfw.org/docs/latest/group__context.html#ga35f1837e6f666781842483937612f163
129 *
130 * @todo Provide abstraction for glfwGetProcAddress.
131 */
132 void init() override {
133 if (initialized_) {
134 return;
135 }
136 const GLADloadfunc procAddressLoader = glfwGetProcAddress;
137 const int gl_ver = gladLoadGL(procAddressLoader);
138
139 if (gl_ver == 0) {
140 logger_.error("Failed to load OpenGL");
141 throw std::runtime_error("Failed to load OpenGL");
142 }
143
144 textRenderer_->init();
145
146 logger_.info(std::format("OpenGL {0}.{1} loaded", GLAD_VERSION_MAJOR(gl_ver), GLAD_VERSION_MINOR(gl_ver)));
147
148 initialized_ = true;
149 }

151 /**
152 * @brief Begins a new render pass.
153 *
154 * This implementation makes sure that the rendering surface is cleared
155 * with the current clear color, which can be configured via `clearColor()`.
156 *
157 * @param renderPass The render pass to begin.
158 *
159 * @see clear()
160 * @see clearColor()
161 */
162 void beginRenderPass(helios::rendering::RenderPass& renderPass) const noexcept override {
163 const auto& viewport = renderPass.viewport();
164
165 assert(viewport.renderTarget() && "Unexpected missing render target for viewport");
166
167 const helios::rendering::RenderTarget& renderTarget = *(viewport.renderTarget());
168 const auto viewportBounds = viewport.bounds();
169 const auto col = viewport.clearColor();
170 glViewport(static_cast<int>(renderTarget.width() * viewportBounds[0]),
171 static_cast<int>(renderTarget.height() * viewportBounds[1]),
172 static_cast<int>(renderTarget.width() * viewportBounds[2]),
173 static_cast<int>(renderTarget.height() * viewportBounds[3]));
174
175
176 // this is equally important for the GlpyhTextRenderer
177 // enable blending since the font's fragment shader uses the alpha channel
178 glEnable(GL_BLEND);
179 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
180
181
182 glClearColor(col[0], col[1], col[2], col[3]);
183
184 const int clearFlags = viewport.clearFlags();
185 glClear(((clearFlags & std::to_underlying(helios::rendering::ClearFlags::Color)) ? GL_COLOR_BUFFER_BIT : 0) |
186 ((clearFlags & std::to_underlying(helios::rendering::ClearFlags::Depth)) ? GL_DEPTH_BUFFER_BIT : 0) |
187 ((clearFlags & std::to_underlying(helios::rendering::ClearFlags::Stencil)) ? GL_STENCIL_BUFFER_BIT
188 : 0));
189 }

191 /**
192 * @brief Processes render commands and draws geometry and text.
193 *
194 * Iterates through all render commands in the render queue, binds the appropriate
195 * shaders and VAOs, applies uniform values, and issues draw calls. After processing
196 * geometry, delegates any text render commands to `renderTextCommands()`.
197 *
198 * @param renderPass The render pass containing the render queue to be processed.
199 *
200 * @see renderTextCommands()
201 */
202 void doRender(helios::rendering::RenderPass& renderPass) const noexcept override {
203 const auto& renderQueue = renderPass.renderQueue();
204
205 const helios::ext::opengl::rendering::shader::OpenGLShader* lastShader = nullptr;
206 unsigned int lastVao = 0;
207
208 for (auto& rc: renderQueue.renderCommands()) {
209
210 if (const auto renderPrototype_ptr = rc->renderPrototype().lock()) {
211 const auto& baseShader = renderPrototype_ptr->material().shader();
212 const auto& baseMesh = renderPrototype_ptr->mesh();
213
214 const auto* shader = static_cast<const helios::ext::opengl::rendering::shader::OpenGLShader*>(&baseShader);
215 assert(shader && "Unexpected failure when casting to OpenGLShader.");
216
217 if (shader != lastShader) {
218 shader->use();
219 lastShader = shader;
220 }
221
222 shader->applyUniformValues(renderPass.frameUniformValues());
223 shader->applyUniformValues(rc->objectUniformValues());
224 shader->applyUniformValues(rc->materialUniformValues());
225
226 const auto* mesh = static_cast<const helios::ext::opengl::rendering::model::OpenGLMesh*>(&baseMesh);
227 assert(mesh && "Unexpected failure when casting to OpenGLMesh.");
228
229 const auto [primitiveType] = mesh->meshConfig();
230
231 if (mesh->vao() != lastVao) {
232 glBindVertexArray(mesh->vao());
233 lastVao = mesh->vao();
234 }
235
236 glDrawElements(toOpenGL(primitiveType), mesh->indexCount(), GL_UNSIGNED_INT, nullptr);
237 glBindVertexArray(0);
238 }
239 }
240
241 if (renderQueue.textRenderCommandsSize() > 0) {
242 renderTextCommands(renderPass);
243 }
244 }

246 /**
247 * @brief Renders all text commands in the render pass.
248 *
249 * Iterates through the text render commands in the render queue and delegates
250 * each command to the `OpenGLGlyphTextRenderer` for rendering.
251 *
252 * @param renderPass The render pass containing text render commands.
253 */
254 void renderTextCommands(const helios::rendering::RenderPass& renderPass) const noexcept {
255 const auto& renderQueue = renderPass.renderQueue();
256
257 for (auto& rc: renderQueue.textRenderCommands()) {
258
259 textRenderer_->render(rc, renderPass.frameUniformValues());//shader, text, screenPosition, scale);
260 }
261 }
