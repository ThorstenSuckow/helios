35 /**
36 * @brief Translates helios abstract PrimitiveType enum to its corresponding OpenGL GLenum value.
37 *
38 * This is a utility function used exclusively by the OpenGLDevice to interpret the
39 * API-agnostic primitive type requested by the helios configuration.
40 *
41 * @param primitiveType The API-agnostic PrimitiveType value.
42 *
43 * @return The corresponding OpenGL primitive type as a GLenum. If mapping does not
44 * succeed, the function falls back to GL_TRIANGLES.
45 */
46 [[nodiscard]] GLenum toOpenGL(helios::rendering::model::config::PrimitiveType primitiveType) const noexcept {
47 switch (primitiveType) {
48 case helios::rendering::model::config::PrimitiveType::Points:
49 return GL_POINTS;
50 case helios::rendering::model::config::PrimitiveType::Lines:
51 return GL_LINES;
52 case helios::rendering::model::config::PrimitiveType::LineLoop:
53 return GL_LINE_LOOP;
54 case helios::rendering::model::config::PrimitiveType::LineStrip:
55 return GL_LINE_STRIP;
56 case helios::rendering::model::config::PrimitiveType::Triangles:
57 return GL_TRIANGLES;
58 case helios::rendering::model::config::PrimitiveType::TriangleStrip:
59 return GL_TRIANGLE_STRIP;
60 case helios::rendering::model::config::PrimitiveType::TriangleFan:
61 return GL_TRIANGLE_FAN;
62 default:
63 logger_.warn("Failed to resolve primitive type, falling back to GL_TRIANGLES.");
64 return GL_TRIANGLES;
65 }
66 }

72 /**
73 * @brief Initializes the OpenGL device to access modern OpenGL.
74 *
75 * This method must be called **after** the helios application's `setCurrent()` was called
76 * for creating a current context. This OpenGL device will then load from this context.
77 *
78 * This method uses the `gladLoadGL` method in conjunction with `glfwGetProcAddress`,
79 * which returns the address of the specific OpenGL core.
80 *
81 * @see https://www.glfw.org/docs/latest/group__context.html#ga35f1837e6f666781842483937612f163
82 *
83 * @todo Provide abstraction for glfwGetProcAddress.
84 */
85 void init() override {
86 if (initialized_) {
87 return;
88 }
89 const GLADloadfunc procAddressLoader = glfwGetProcAddress;
90 const int gl_ver = gladLoadGL(procAddressLoader);
91
92 if (gl_ver == 0) {
93 logger_.error("Failed to load OpenGL");
94 throw std::runtime_error("Failed to load OpenGL");
95 }
96
97 logger_.info(std::format("OpenGL {0}.{1} loaded", GLAD_VERSION_MAJOR(gl_ver), GLAD_VERSION_MINOR(gl_ver)));
98
99 initialized_ = true;
100 }

102 /**
103 * @brief Begins a new render pass.
104 *
105 * This implementation makes sure that the rendering surface is cleared
106 * with the current clear color, which can be configured via `clearColor()`.
107 *
108 * @param renderPass The render pass to begin.
109 *
110 * @see clear()
111 * @see clearColor()
112 */
113 void beginRenderPass(helios::rendering::RenderPass& renderPass) const noexcept override {
114 const auto& viewport = renderPass.viewport();
115
116 if (!viewport.renderTarget()) {
117 logger_.warn("Viewport has no render target, skipping renderPass");
118 return;
119 }
120
121 logger_.info("Begin RenderPass.");
122
123 const helios::rendering::RenderTarget& renderTarget = *(viewport.renderTarget());
124 const auto viewportBounds = viewport.bounds();
125 const auto col = viewport.clearColor();
126 glViewport(static_cast<int>(renderTarget.width() * viewportBounds[0]),
127 static_cast<int>(renderTarget.height() * viewportBounds[1]),
128 static_cast<int>(renderTarget.width() * viewportBounds[2]),
129 static_cast<int>(renderTarget.height() * viewportBounds[3]));
130
131
132 glClearColor(col[0], col[1], col[2], col[3]);
133
134 const int clearFlags = viewport.clearFlags();
135 glClear(((clearFlags & std::to_underlying(helios::rendering::ClearFlags::Color)) ? GL_COLOR_BUFFER_BIT : 0) |
136 ((clearFlags & std::to_underlying(helios::rendering::ClearFlags::Depth)) ? GL_DEPTH_BUFFER_BIT : 0) |
137 ((clearFlags & std::to_underlying(helios::rendering::ClearFlags::Stencil)) ? GL_STENCIL_BUFFER_BIT
138 : 0));
139 }

141 /**
142 * @brief Binds the VAOs and draws the elements based on the RenderCommands available with the RenderPass.
143 *
144 * @param renderPass The render pass containing the render queue to be processed.
145 */
146 void doRender(helios::rendering::RenderPass& renderPass) const noexcept override {
147 const auto& renderQueue = renderPass.renderQueue();
148
149 logger_.info(std::format("Rendering {0} item(s)...", renderQueue.count()));
150
151 for (auto& rc: renderQueue.renderCommands()) {
152
153 if (const auto renderPrototype_ptr = rc->renderPrototype().lock()) {
154 const auto& baseShader = renderPrototype_ptr->material().shader();
155 const auto& baseMesh = renderPrototype_ptr->mesh();
156
157 const auto* shader = dynamic_cast<const helios::ext::opengl::rendering::shader::OpenGLShader*>(&baseShader);
158 if (!shader) {
159 logger_.error("Failed to cast shader to OpenGLShader.");
160 continue;
161 }
162 logger_.info("activating shader...");
163 shader->use();
164 shader->applyUniformValues(renderPass.frameUniformValues());
165 shader->applyUniformValues(rc->objectUniformValues());
166 shader->applyUniformValues(rc->materialUniformValues());
167
168 const auto* mesh = dynamic_cast<const helios::ext::opengl::rendering::model::OpenGLMesh*>(&baseMesh);
169 if (!mesh) {
170 logger_.error("Failed to cast mesh to OpenGLMesh.");
171 continue;
172 }
173 const auto [primitiveType] = mesh->meshConfig();
174 logger_.info(std::format("Binding vao {0}", mesh->vao()));
175 glBindVertexArray(mesh->vao());
176 glEnable(GL_BLEND);
177 glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
178 glDrawElements(toOpenGL(primitiveType), mesh->indexCount(), GL_UNSIGNED_INT, nullptr);
179 glBindVertexArray(0);
180 }
181 }
182 }
