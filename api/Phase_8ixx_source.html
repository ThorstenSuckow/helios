57 /**
58 * @brief Represents a phase in the game loop containing multiple passes.
59 *
60 * A Phase groups related systems into passes that are executed sequentially.
61 * Each pass can contain multiple systems and may have a commit point for
62 * event synchronization.
63 *
64 * Commit points allow fine-grained control over when commands are flushed,
65 * managers are processed, and pass-level events are synchronized. See
66 * CommitPoint for available synchronization options.
67 *
68 * Phases are owned by the GameLoop and should not be created directly.

72 * @see GameLoop

105 /**
106 * @brief Updates all passes within this phase.
107 *
108 * @details Iterates through all passes, updating their systems and then
109 * invoking the commit action based on the pass's configured CommitPoint.
110 * The commit point determines whether pass-level events are synchronized,
111 * the command buffer is flushed, or managers are processed.
112 *
113 * Passes are conditionally executed based on their configured game state.
114 * A pass is only updated if its `runsIn()` state matches the current game state.
115 *
116 * @param updateContext The current update context.
117 * @param gameState The current game state used to filter pass execution.
118 *
119 * @see CommitPoint
120 * @see Pass::addCommitPoint()
121 * @see Pass::runsIn()
122 */
123 void update(helios::engine::runtime::world::UpdateContext& updateContext){
124
125 for (auto& pass : passEntries_) {
126
127 if (pass->shouldRun(updateContext)) {
128 pass->update(updateContext);
129 notifyPassCommitListeners(pass->commitPoint(), updateContext);
130 }
131
132 }
133 };

135 /**
136 * @brief Notifies all registered listeners about a pass commit.
137 *
138 * @details Called after each pass completes its update cycle. Each registered
139 * PassCommitListener receives the commit point flags, allowing it to perform
140 * the appropriate synchronization actions (event swapping, command flushing,
141 * manager processing).
142 *
143 * @param commitPoint The CommitPoint flags from the completed pass.
144 * @param updateContext The current update context.
145 *
146 * @return Always returns true.
147 *
148 * @see PassCommitListener::onPassCommit()
149 * @see addPassCommitListener()
150 */
151 bool notifyPassCommitListeners(CommitPoint commitPoint, helios::engine::runtime::world::UpdateContext& updateContext) {
152
153 for (const auto & passCommitListener : passCommitListeners_) {
154 passCommitListener->onPassCommit(commitPoint, updateContext);
155 }
156 return true;
157 }

181 /**
182 * @brief Registers a listener to be notified when passes commit.
183 *
184 * @details The listener will receive notifications for all passes within this phase.
185 * The GameLoop typically registers itself to handle event synchronization,
186 * command buffer flushing, and manager processing based on commit point flags.
187 *
188 * Duplicate registrations are prevented; attempting to add the same listener
189 * twice will return false and leave the listener list unchanged.
190 *
191 * @param passCommitListener Pointer to the listener to register. Must remain
192 * valid for the lifetime of this Phase or until removed.
193 *
194 * @return True if the listener was added, false if it was already registered.
195 *
196 * @see PassCommitListener
197 * @see notifyPassCommitListeners()
198 */
199 bool addPassCommitListener(PassCommitListener* passCommitListener) {
200
201 for (int i = 0; i < passCommitListeners_.size(); i++) {
202 if (passCommitListeners_[i] == passCommitListener) {
203 return false;
204 }
205 }
206
207 passCommitListeners_.emplace_back(passCommitListener);
208
209 return true;
210 }
