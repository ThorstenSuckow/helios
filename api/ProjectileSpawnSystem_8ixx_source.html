27 /**
28 * @brief System that spawns projectiles for entities with active ShootComponents.
29 *
30 * @details ProjectileSpawnSystem iterates all entities that have a ShootComponent,
31 * Aim2DComponent, and TranslationStateComponent. When a ShootComponent has non-zero
32 * intensity and the aim component has a valid frequency, the system calculates how
33 * many projectiles to spawn based on accumulated time and cooldown.
34 *
35 * ## Fire Rate Mechanics
36 *
37 * The system implements an intensity-based fire rate:
38 * 1. Accumulates `deltaTime * intensity` into the cooldown timer
39 * 2. When timer >= cooldownDelta, spawns `floor(timer / cooldownDelta)` projectiles
40 * 3. Reduces timer by spawned amount, preserving fractional remainder
41 *
42 * This allows continuous fire with variable intensity affecting the effective rate.
43 *
44 * ## Spawn Context
45 *
46 * Each SpawnCommand includes an EmitterContext containing:
47 * - **position:** Current translation from TranslationStateComponent
48 * - **velocity:** sourceVelocity + (aimDirection * projectileSpeed)
49 *
50 * Example setup:
51 * ```cpp
52 * auto projectileSpawnSystem = std::make_unique<ProjectileSpawnSystem>(
53 * SpawnProfileId{1} // ID of the projectile spawn profile
54 * );
55 * mainPhase.addPass().add(std::move(projectileSpawnSystem));
56 * ```
57 *
58 * @note Entities must have ShootComponent, Aim2DComponent, and
59 * TranslationStateComponent to be processed.
60 *
61 * @see ShootComponent
62 * @see Aim2DComponent
63 * @see SpawnCommand
64 * @see EmitterContext
65 */

90 /**
91 * @brief Processes all shooting entities and spawns projectiles.
92 *
93 * @details For each entity with ShootComponent, Aim2DComponent, and
94 * TranslationStateComponent:
95 *
96 * 1. Skips if aim frequency or intensity are near zero
97 * 2. Accumulates `deltaTime * intensity` into the cooldown timer
98 * 3. If timer < cooldownDelta, updates timer and continues
99 * 4. Calculates projectile count as `floor(timer / cooldownDelta)`
100 * 5. Reduces timer by `count * cooldownDelta`
101 * 6. Enqueues SpawnCommands with EmitterContext for each projectile
102 *
103 * @param updateContext The current frame's update context.
104 */
105 void update(helios::engine::runtime::world::UpdateContext& updateContext) noexcept override {
106
107 for (auto [entity, tsc, ac, sc] : gameWorld_->find<
108 helios::engine::modules::spatial::transform::components::TranslationStateComponent,
109 helios::engine::mechanics::combat::components::Aim2DComponent,
110 helios::engine::mechanics::combat::components::ShootComponent
111 >().each()) {
112
113
114 const float intensity = sc->intensity();
115 const float cooldown = sc->cooldownDelta();
116
117 float timer = sc->cooldownTimer();
118
119 // only process for meaningful values
120 assert(cooldown > helios::math::EPSILON_LENGTH && "Unexpected cooldownDelta");
121 if (ac->frequency() <= helios::math::EPSILON_LENGTH ||
122 intensity <= helios::math::EPSILON_LENGTH) {
123 continue;
124 };
125
126 const float delta = updateContext.deltaTime();
127
128 // update the timer to accumulate frame times * intensity.
129 // will be updated in this frame if projectiles spawned,
130 timer += delta * intensity;
131
132 if (timer < cooldown) {
133 sc->setCooldownTimer(timer);
134 continue;
135 }
136
137 const unsigned int amount = static_cast<unsigned int>(timer / cooldown);
138
139 // reduce timer by the product of projectiles and cooldown,
140 // which leaves a fractional part that did not add a whole projectile
141 // to this frame
142 sc->setCooldownTimer(timer - (amount * cooldown));
143
144 const auto aimDirection = ac->direction().toVec3();
145 assert(aimDirection.isNormalized() && "Unexpected aimDirection.length()");
146
147 for (unsigned int i = 0; i < amount; i++) {
148 updateContext.commandBuffer().add<
149 helios::engine::runtime::spawn::commands::SpawnCommand
150 >(
151 spawnProfileId_,
152 helios::engine::runtime::spawn::SpawnContext{
153 helios::engine::runtime::spawn::EmitterContext{
154 tsc->translation(),
155 sc->sourceVelocity() + (aimDirection * sc->projectileSpeed())
156 }
157 }
158 );
159 }
160 }
161 }
