58 /**
59 * @brief Builds a `RenderPass` from a given `helios::scene::Snapshot`.
60 *
61 * This method orchestrates the creation of a `RenderQueue` and populates it
62 * with `RenderCommand`s derived from the `SnapshotItem`s within the snapshot.
63 * It also sets up frame-specific uniform values for the `RenderPass`.
64 *
65 * @param snapshot A const ref to the snapshot containing the scene data for which
66 * the `RenderPass` is to be built.
67 *
68 * @return A fully constructed `RenderPass` object. The returned object owns the
69 * `RenderQueue` and the frame-specific `UniformValueMap`.
70 *
71 * @see ::populateRenderQueue()
72 */
73 [[nodiscard]] RenderPass buildRenderPass(const helios::scene::Snapshot& snapshot) const {
74
75 auto renderQueue = std::make_unique<RenderQueue>();
76
77 populateRenderQueue(snapshot, *renderQueue);
78
79 const auto& projectionMatrix = snapshot.projectionMatrix();
80 const auto& viewMatrix = snapshot.viewMatrix();
81
82 auto frameUniformValues = std::make_unique<helios::rendering::shader::UniformValueMap>();
83 frameUniformValues->set(helios::rendering::shader::UniformSemantics::ProjectionMatrix, projectionMatrix);
84 frameUniformValues->set(helios::rendering::shader::UniformSemantics::ViewMatrix, viewMatrix);
85
86 return RenderPass(
87 snapshot.viewport(),
88 std::move(renderQueue),
89 std::move(frameUniformValues)
90 );
91 }

93 /**

95 *
96 * This method clears the specified `RenderQueue` before adding new `RenderCommand`s to it.

98 * @param snapshot A const ref to the `Snapshot` providing scene data.

101 *
102 * @see ::makeRenderCommand()
103 */
104 void populateRenderQueue(
105 const helios::scene::Snapshot& snapshot, helios::rendering::RenderQueue& renderQueue) const {
106
107 // clear the queue
108 renderQueue.clear();
109
110 const auto& snapshotItems = snapshot.snapshotItems();
111
112 for (const auto& item : snapshotItems) {
113 makeRenderCommand(item, renderQueue);
114 }

117 /**
118 * @brief Emits render commands from a `helios::scene::SnapshotItem` to the render queue.
119 *
120 * This method extracts necessary data from the `SnapshotItem`, sets up the object
121 * uniform values (e.g., model matrix), and delegates to the `Renderable` to emit
122 * its render commands to the queue.
123 *
124 * @param snapshotItem A const ref to the `SnapshotItem` from which to create render commands.
125 * @param renderQueue A ref to the `RenderQueue` to emit commands to.
126 *
127 * @note If the `Renderable` pointer in the `SnapshotItem` is `nullptr`, a warning is
128 * logged and no commands are emitted.
129 */
130 void makeRenderCommand(
131 const helios::scene::SnapshotItem& snapshotItem,
132 helios::rendering::RenderQueue& renderQueue) const noexcept {
133
134 const auto* renderable = snapshotItem.renderable();
135
136 if (renderable == nullptr) {
137 logger_.warn("Renderable no longer available");
138 return;
139 }
140
141 auto objectUniformValues = helios::rendering::shader::UniformValueMap();
142 auto materialUniformValues = helios::rendering::shader::UniformValueMap();
143 objectUniformValues.set(
144 helios::rendering::shader::UniformSemantics::ModelMatrix,
145 snapshotItem.worldMatrix()
146 );
147
148 renderable->emit(
149 renderQueue,
150 std::move(objectUniformValues),
151 std::move(materialUniformValues)
152 );
153
154 }
