59 /**
60 * @brief Type-indexed registry for engine resources with O(1) lookup.
61 *
62 * @details ResourceRegistry provides a central store for Managers,
63 * CommandBuffers, and CommandHandlers. Each resource type receives a
64 * unique compile-time index via ResourceTypeId, enabling O(1) access
65 * through a flat `void*` array (`fastAccess_`).
66 *
67 * ## Ownership Model
68 *
69 * - **Managers and CommandBuffers** are heap-allocated and owned via
70 * `ErasedUnique` (type-erased unique ownership, 16 bytes per entry).
71 * - **CommandHandlers** are registered by reference (non-owning). They
72 * typically live as part of a Manager that is already owned.
73 *
74 * ## Dual-Access Pattern
75 *
76 * Resources are stored in two parallel structures:
77 * - `fastAccess_[idx]` — O(1) type-based lookup (hot path)
78 * - `managers_` / `commandBuffers_` — linear iteration for batch operations
79 * (e.g., `GameWorld::flushManagers()`, `GameWorld::init()`)
80 *
81 * @see ResourceTypeId
82 * @see ErasedUnique
83 * @see Manager
84 * @see CommandBuffer
85 * @see TypedCommandHandler
86 */
87 class ResourceRegistry {

112 * @details Indexed by `ResourceTypeId::id<T>().value()`. Each slot
113 * holds a raw `void*` to the resource instance.
114 */

120 /**
121 * @brief Creates and registers an owning resource.
122 *
123 * @details Heap-allocates a resource of type T, stores it in `owned_`
124 * via ErasedUnique, and indexes it in `fastAccess_`. If T derives from
125 * Manager or CommandBuffer, it is additionally tracked in the
126 * corresponding iteration list.
127 *
128 * @tparam T The resource type. Must derive from Manager or CommandBuffer.
129 * @tparam Args Constructor argument types.
130 *
131 * @param args Arguments forwarded to the T constructor.
132 *
133 * @return Reference to the newly created resource.
134 *
135 * @pre No resource of type T is already registered.
136 */
137 template<class T, class... Args>
138 T& registerResource(Args&&... args) {
139
140 auto uniquePtr = std::make_unique<T>(std::forward<Args>(args)...);
141 T* raw = uniquePtr.get();
142
143 const size_t idx = ResourceTypeId::id<T>().value();
144
145 if (idx >= fastAccess_.size()) {
146 fastAccess_.resize(idx + 1, nullptr);
147 }
148
149 assert(fastAccess_[idx] == nullptr && "Resource already registered");
150
151 fastAccess_[idx] = raw;
152
153 if constexpr (std::derived_from<T, Manager>) {
154 managers_.push_back(static_cast<Manager*>(raw));
155 }
156
157 if constexpr (std::derived_from<T, CommandBuffer>) {
158 commandBuffers_.push_back(static_cast<CommandBuffer*>(raw));
159 }
160
161 owned_.emplace_back(ErasedUnique{std::move(uniquePtr)});
162
163 return *raw;
164 }

166 /**
167 * @brief Registers a non-owning CommandHandler reference.
168 *
169 * @details Stores a pointer to an externally owned CommandHandler in
170 * `fastAccess_` and `commandHandlers_`. The handler must outlive this
171 * registry (typically guaranteed because the handler is a Manager that
172 * is already owned by this registry).
173 *
174 * @tparam T The CommandHandler type. Must derive from CommandHandler.
175 *
176 * @param ref Reference to the handler to register.
177 *
178 * @return Reference to the registered handler.
179 *
180 * @pre No resource of type T is already registered.
181 */
182 template<class T>
183 requires std::derived_from<T, CommandHandler>
184 T& registerResource(T& ref) {
185
186
187 const size_t idx = ResourceTypeId::id<T>().value();
188
189 if (idx >= fastAccess_.size()) {
190 fastAccess_.resize(idx + 1, nullptr);
191 }
192
193 assert(fastAccess_[idx] == nullptr && "Resource already registered");
194
195 fastAccess_[idx] = &ref;
196
197
198 if constexpr (std::derived_from<T, CommandHandler>) {
199
200 if (idx >= commandHandlers_.size()) {
201 commandHandlers_.resize(idx + 1, nullptr);
202 }
203
204 commandHandlers_[idx] = static_cast<CommandHandler*>(&ref);
205 }
206
207 return ref;
208 }
