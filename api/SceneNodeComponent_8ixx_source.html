16 /**
17 * @brief Component that links a GameObject to a SceneNode in the scene graph.
18 *
19 * @details Provides transform operations (translation, rotation, scale) on the
20 * underlying SceneNode, as well as unit-aware sizing via setSize(). This component
21 * bridges the game logic layer (GameObject) with the rendering layer (SceneNode).
22 *
23 * The canonical size is captured from the SceneNode's mesh AABB at construction
24 * time. This enables setSize() to calculate the correct scale factors for
25 * achieving a desired world-space size in Helios units.
26 *
27 * @note The SceneNode must remain valid for the lifetime of this component.
28 */

63 /**

65 *
66 * This operation sets the scale of the underlying SceneNode according to the canonical size
67 * of the Renderable represented by the SceneNode.
68 *
69 * @param width Desired width in the specified unit.
70 * @param height Desired height in the specified unit.
71 * @param depth Desired depth in the specified unit.
72 * @param unit Unit of measurement (default: Meter).
73 *
74 * @return Reference to this component for method chaining.
75 *
76 * @note Calculates scale factors based on canonical mesh size and applies them.
77 */
78 SceneNodeComponent& setSize(
79 float width, float height, float depth, helios::core::units::Unit unit = helios::core::units::Unit::Meter
80 ) noexcept {
81 const auto& csize = canonicalSize_;
82
83 setScale(
84 helios::math::vec3f{
85 width != 0 && csize[0] != 0 ? helios::core::units::from(width, unit)/csize[0] : 0,
86 height != 0 && csize[1] != 0 ? helios::core::units::from(height, unit)/csize[1] : 0,
87 depth != 0 && csize[2] != 0 ? helios::core::units::from(depth, unit)/csize[2] : 0,
88 }
89 );
90
91 return *this;
92 };
