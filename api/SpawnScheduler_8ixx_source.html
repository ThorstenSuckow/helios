29 /**
30 * @brief Scheduler that evaluates spawn rules and produces spawn plans.
31 *
32 * @details SpawnScheduler is the central coordinator for rule-based spawning.
33 * It maintains a collection of spawn rules, evaluates them each frame, and
34 * produces ScheduledSpawnPlan instances for rules whose conditions are met.
35 *
36 * ## Workflow
37 *
38 * 1. **Registration:** Rules are added via `addRule()` with their profile ID
39 * 2. **Evaluation:** `evaluate()` is called each frame by GameObjectSpawnSystem
40 * 3. **Scheduling:** Rules that pass conditions produce ScheduledSpawnPlans
41 * 4. **Draining:** `drainScheduledPlans()` returns and clears pending plans
42 * 5. **Commit:** `commit()` is called when spawns complete to update state
43 *
44 * Example:
45 * ```cpp
46 * SpawnScheduler scheduler;
47 * scheduler.addRule(enemyProfileId, std::make_unique<TimerSpawnRule>(
48 * ruleId, 2.0f, 3 // Every 2s, spawn 3
49 * ));
50 *
51 * // In game loop
52 * scheduler.evaluate(updateContext);
53 * auto plans = scheduler.drainScheduledPlans();
54 * ```
55 *
56 * @see SpawnRule
57 * @see ScheduledSpawnPlan
58 * @see GameObjectSpawnSystem
59 */

94 /**
95 * @brief Evaluates all spawn rules against current conditions.
96 *
97 * @details Iterates through all registered rules, updates their state,
98 * and creates ScheduledSpawnPlan entries for rules that should spawn.
99 *
100 * @param updateContext The current frame's update context.
101 */
102 void evaluate(const helios::engine::runtime::world::UpdateContext& updateContext) {
103
104 const auto* poolManager = updateContext.gameWorld().getManager<helios::engine::runtime::pooling::GameObjectPoolManager>();
105 const auto* spawnManager = updateContext.gameWorld().getManager<helios::engine::runtime::spawn::SpawnManager>();
106
107 scheduledSpawnPlans_.clear();
108
109 for (auto& [spawnProfileId, rule] : spawnRules_) {
110
111 const auto* spawnProfile = spawnManager->spawnProfile(spawnProfileId);
112 if (!spawnProfile) {
113 /**
114 * @todo remove stale entry
115 */
116 continue;
117 }
118
119 const auto& [gameObjectPoolId, _, __] = *spawnProfile;
120
121 auto& spawnRuleState = spawnRuleStates_[rule->spawnRuleId()];
122
123 const auto poolSnapshot = poolManager->poolSnapshot(gameObjectPoolId);
124
125 // tick the rule state
126 spawnRuleState.update(updateContext.deltaTime());
127
128 auto spawnPlan = rule->evaluate(
129 gameObjectPoolId, poolSnapshot,
130 spawnRuleState,
131 updateContext
132 );
133
134 if (spawnPlan.amount == 0) {
135 continue;
136 }
137
138 scheduledSpawnPlans_.push_back({
139 spawnProfileId,
140 std::move(spawnPlan)}
141 );
142
143 }
144
145 }

164 [[nodiscard]] std::vector<ScheduledSpawnPlan> drainScheduledPlans() {

170 /**
171 * @brief Adds a spawn rule for a profile.
172 *
173 * @param spawnProfileId The profile ID to associate with the rule.
174 * @param spawnRule The spawn rule to add. Ownership transferred.
175 *
176 * @return Reference to this scheduler for chaining.
177 *
178 * @pre No rule is already registered for this profile ID.
179 */
180 SpawnScheduler& addRule(
181 const helios::engine::core::data::SpawnProfileId spawnProfileId,
182 std::unique_ptr<helios::engine::runtime::spawn::policy::SpawnRule> spawnRule
183 ) {
184 assert(!spawnRules_.contains(spawnProfileId) && "Duplicate SpawnProfile entry");
185 assert(!spawnRuleStates_.contains(spawnRule->spawnRuleId()) && "Duplicate SpawnRuleId entry");
186
187 spawnRuleStates_.try_emplace(spawnRule->spawnRuleId());
188 spawnRules_.try_emplace(spawnProfileId, std::move(spawnRule));
189
190
191 return *this;
192 }

194 /**
195 * @brief Commits a completed spawn operation to update rule state.
196 *
197 * @details Called when a spawn request has been executed to update
198 * the associated rule's state (e.g., reset timers, update counts).
199 *
200 * @param spawnRuleId The rule that triggered the spawn.
201 * @param spawnCount The number of entities actually spawned.
202 */
203 void commit(const helios::engine::core::data::SpawnRuleId spawnRuleId, const size_t spawnCount) {
204
205 for (auto& spawnRule : spawnRules_ | std::views::values) {
206
207 if (spawnRule->spawnRuleId() == spawnRuleId) {
208
209 auto it = spawnRuleStates_.find(spawnRuleId);
210 assert(it != spawnRuleStates_.end() && "Unexpected missing spawnRuleState");
211
212 spawnRule->commit(it->second, spawnCount);
213 }
214 }
215 }
