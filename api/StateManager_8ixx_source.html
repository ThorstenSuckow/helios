41 /**
42 * @brief Manages state transitions using a rule-based system.
43 *
44 * @details Implements both Manager and TypedStateCommandHandler interfaces.
45 * Processes state transition commands by matching against registered rules,
46 * executing guards, and notifying listeners.
47 *
48 * ## Transition Flow
49 *
50 * 1. Commands are submitted via submit() and queued
51 * 2. During flush(), the last pending command is processed
52 * 3. Rules are checked to find a matching transition
53 * 4. If a guard is present, it must return true
54 * 5. Listeners are notified: onStateExit -> onStateTransition -> onStateEnter
55 * 6. Session state is updated
56 *
57 * @tparam StateType The state enum type.
58 *
59 * @see StateTransitionRule
60 * @see StateTransitionListener
61 * @see StateCommand
62 */
63 template <typename StateType>
64 class StateManager : public helios::engine::runtime::world::Manager,
65 public TypedStateCommandHandler<StateType> {

166 /**
167 * @brief Processes pending state commands.
168 *
169 * @details Processes the last pending command, finds matching rules,
170 * executes guards, and triggers the transition if valid.
171 *
172 * @param gameWorld The game world.
173 * @param updateContext The current frame's update context.
174 */
175 void flush(
176 helios::engine::runtime::world::GameWorld& gameWorld,
177 helios::engine::runtime::world::UpdateContext& updateContext
178 ) noexcept override {
179
180 if (pending_.empty()) {
181 return;
182 }
183
184 auto command = pending_.back();
185
186 auto transitionRequest = command.transitionRequest();
187
188 auto& session = gameWorld.session();
189 auto currentFrom = session.state<StateType>();
190 auto from = transitionRequest.from();
191 auto transitionId = transitionRequest.transitionId();
192
193
194 if (currentFrom != from) {
195 pending_.clear();
196 return;
197 }
198
199
200 for (auto& rule : rules_) {
201 if (rule.from() == from && rule.transitionId() == transitionId) {
202
203 if (rule.guard()) {
204 if (!rule.guard()(updateContext, transitionRequest)) {
205 break;
206 }
207 }
208
209 signalExit(from, rule.to(), transitionId, updateContext);
210 signalTransition(from, rule.to(), transitionId, updateContext);
211 session.setStateFrom<StateType>(StateTransitionContext<StateType>{rule.from(), rule.to(), transitionId});
212 signalEnter(from, rule.to(), transitionId, updateContext);
213 }
214 }
215
216 pending_.clear();
217 }
