165 /**
166 * @brief Sets the heading direction and turn intensity from input.
167 *
168 * @details Updates the heading input vector and turn intensity. If the
169 * turn intensity is below the epsilon threshold, input is deactivated
170 * and the heading is reset to zero.
171 *
172 * @param direction Normalized direction vector representing the desired heading.
173 * @param turnIntensity Magnitude of the input (0.0 to 1.0).
174 *
175 * @pre direction must be a normalized vector (length ~= 1.0).
176 */
177 void setSteeringIntent(helios::math::vec3f direction, float turnIntensity) {
178
179 steeringInput_ = direction;
180 turnIntensity_ = turnIntensity;
181 stateChanged_ = true;
182
183 if (turnIntensity_ <= helios::math::EPSILON_LENGTH) {
184 steeringInput_ = {0.0f, 0.0f, 0.0f};
185 stateChanged_ = false;
186 turnIntensity_ = 0.0f;
187 return;
188 }
189
190 assert(std::abs(direction.length() - 1.0f) <= 0.001f && "Unexpected direction vector - not normalized");
191 }
