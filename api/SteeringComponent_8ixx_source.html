203 /**
204 * @brief Sets the heading direction and turn intensity from input.
205 *
206 * @details Updates the heading input vector and turn intensity. If the
207 * turn intensity is below the epsilon threshold, input is deactivated
208 * and the heading is reset to zero.
209 *
210 * @param direction Normalized direction vector representing the desired heading.
211 * @param turnIntensity Magnitude of the input (0.0 to 1.0).
212 *
213 * @pre direction must be a normalized vector (length ~= 1.0).
214 */
215 void setSteeringIntent(helios::math::vec3f direction, float turnIntensity) {
216
217 steeringInput_ = direction;
218 turnIntensity_ = turnIntensity;
219 stateChanged_ = true;
220
221 if (turnIntensity_ <= helios::math::EPSILON_LENGTH) {
222 steeringInput_ = {0.0f, 0.0f, 0.0f};
223 stateChanged_ = false;
224 turnIntensity_ = 0.0f;
225 return;
226 }
227
228 assert(direction.isNormalized() && "Unexpected direction vector - not normalized");
229 }
