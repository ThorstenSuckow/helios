222 *
223 * @return Reference to this config for method chaining.

225 TextRenderableConfig& build() {
226
227 assert(shader_ && "Unexpected missing shape and/or shader");
228
229 auto shaderProperties = std::make_shared<helios::rendering::text::TextShaderProperties>(color_);
230
231 const auto textPrototype = std::make_shared<helios::rendering::text::TextRenderPrototype>(
232 shader_, shaderProperties, fontResourceProvider_
233 );
234
235 auto renderable = std::make_shared<helios::rendering::text::TextRenderable>(
236 std::make_unique<helios::rendering::text::TextMesh>(std::move(text_), fontScale_, fontId_),
237 textPrototype
238 );
239
240 gameObject_.add<helios::engine::modules::rendering::renderable::components::RenderableComponent>(renderable);
241
242 auto& msc = gameObject_.getOrAdd<helios::engine::modules::rendering::model::components::ModelAabbComponent>();
243 msc.setAabb(renderable->localAABB());
244
245 if (isUiText_) {
246 gameObject_.add<helios::engine::modules::ui::widgets::components::UiTextComponent>(renderable.get())
247 .setTemplate(template_);
248
249 isUiText_ = false;
250 }
251
252 return *this;
253 }
