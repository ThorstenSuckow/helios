21 /**
22 * @brief A SpawnCondition that triggers spawning at fixed time intervals.
23 *
24 * @details
25 * TimerSpawnCondition uses a countdown timer to determine when GameObjects
26 * should be spawned. Each time the countdown reaches zero, it resets and
27 * increments the spawn budget. If the elapsed time exceeds multiple intervals
28 * (e.g., during a frame spike), multiple objects will be budgeted for spawning.
29 *
30 * Example usage:
31 * ```cpp
32 * // Spawn one object every 2 seconds
33 * auto condition = std::make_unique<TimerSpawnCondition>(2.0f);
34 *
35 * auto spawnSystem = std::make_unique<GameObjectSpawnSystem>(
36 * std::move(pool), level, std::move(condition), std::move(strategy)
37 * );
38 * ```
39 *
40 * @see SpawnCondition
41 * @see GameObjectSpawnSystem
42 */

83 * @param gameObjectPool Pointer to the pool. Unused.

92 ) noexcept override {

95
96 countdown_ -= updateContext.deltaTime();
97
98 if (countdown_ <= 0) {

100 const auto intervalsMissed = static_cast<size_t>(spill/interval_);
101 const auto spawnCount = intervalsMissed + 1.0f; // 1.0f accounts for the current call
