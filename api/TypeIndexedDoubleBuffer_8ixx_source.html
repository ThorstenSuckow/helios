58 * @class ReadWriteDoubleBuffer
59 * @brief Type-safe double-buffered container for messages of type T.
60 *
61 * @details Implements a producer-consumer pattern where messages are written
62 * to one buffer while consumers read from another. This decouples message
63 * production from consumption and allows lock-free operation in single-threaded
64 * game loops.
65 *
66 * @tparam T The message type stored in this buffer. Must be movable.
67 */
68 template<typename T>
69 class ReadWriteDoubleBuffer : public BufferBase {
70
71 /**
72 * @brief Buffer containing messages available for reading.
73 */
74 ReadBuffer<T> readBuffer_;
75
76 /**
77 * @brief Buffer where new messages are pushed.
78 */
79 WriteBuffer<T> writeBuffer_;
80
81
82 public:
83
84 /**
85 * @brief Pre-allocates memory for both buffers.
86 *
87 * @param size The number of messages to reserve capacity for.
88 */
89 void reserve(size_t size) {
90 readBuffer_.reserve(size);
91 writeBuffer_.reserve(size);
92 }
93
94 /**
95 * @brief Constructs and pushes a message to the write buffer.
96 *
97 * @tparam Args Constructor argument types for T.
98 *
99 * @param args Arguments forwarded to T's constructor.
100 */
101 template<typename... Args>
102 void push(Args&&... args) {
103 writeBuffer_.push(std::forward<Args>(args)...);
104 }
105
106 /**
107 * @brief Returns a read-only view of messages in the read buffer.
108 *
109 * @return A span over all messages written in the previous frame.
110 */
111 std::span<const T> read() const noexcept {
112 return readBuffer_.read();
113 }
114
115 /**
116 * @brief Swaps read with write buffer.
117 *
118 * @details Swaps the read- with the write-buffer. The write buffer is empty after
119 * this operation, while the read buffer contains the contents of the write buffer.
120 */
121 void swap() override {

123 }

128 void clearReadBuffer() override {
129 readBuffer_.clear();
130 }

140 template<typename Indexer>
141 /**
142 * @class TypeIndexedDoubleBuffer
143 * @brief Central hub for publishing and consuming typed messages.
144 *
145 * @details TypeIndexedDoubleBuffer provides a double-buffered, type-indexed message system
146 * for decoupled inter-system communication. Systems push messages during their update phase,
147 * then at frame boundaries `swapBuffers()` is called to make those messages available for reading.
148 *
149 * Each message type T gets its own dedicated ReadWriteDoubleBuffer, indexed via TypeIndexer
150 * for O(1) access. This design enables efficient, allocation-friendly inter-system communication.
151 *
152 * Example usage:
153 * ```cpp
154 * // Define a message
155 * struct CollisionMessage {
156 * EntityId a;
157 * EntityId b;
158 * vec3f contact;
159 * };
160 *
161 * // In collision system
162 * messageBuffer.push<CollisionMessage>(entityA, entityB, contactPoint);
163 *
164 * // At frame boundary
165 * messageBuffer.swapBuffers();
166 *
167 * // In damage system (next phase or frame)
168 * for (const auto& msg : messageBuffer.read<CollisionMessage>()) {
169 * applyDamage(msg.a, msg.b);
170 * }
171 * ```
172 *
173 * @tparam Indexer The TypeIndexer used for mapping message types to buffer indices.
174 */
175 class TypeIndexedDoubleBuffer {
176
177 std::vector<std::unique_ptr<BufferBase>> buffers_;
178
179 template<typename T>
180 static size_t index() {
181 return Indexer::template typeIndex<T>();
182 }
183
184 public:
185
186 ~TypeIndexedDoubleBuffer() = default;
187 TypeIndexedDoubleBuffer() = default;
188
189 /**
190 * @brief Pushes a message of type T to the write buffer.
191 *
192 * @tparam T The message type.
193 * @tparam Args Constructor argument types for T.
194 *
195 * @param args Arguments forwarded to T's constructor.
196 */
197 template<typename T, typename... Args>
198 void push(Args&&... args) {
199 getOrCreateBuffer<T>().push(std::forward<Args>(args)...);
200 }
201
202 /**
203 * @brief Returns a read-only view of all messages of type T.
204 *
205 * @details Returns messages that were written before the last `swapBuffers()` call.
206 * If no messages of type T exist, returns an empty span.
207 *
208 * @tparam T The message type to read.
209 *
210 * @return A span over all messages of type T in the read buffer.
211 */
212 template<typename T>
213 std::span<const T> read() const noexcept {
214
215 const size_t idx = index<T>();
216
217 if (buffers_.size() <= idx || !buffers_[idx]) {
218 return {};
219 }
220
221 return static_cast<ReadWriteDoubleBuffer<T>*>(buffers_[idx].get())->read();
222 }
223
224 /**
225 * @brief Pre-allocates capacity for messages of type T.
226 *
227 * @details Call during initialization to avoid allocations during gameplay.
228 *
229 * @tparam T The message type.
230 *
231 * @param size The number of messages to reserve capacity for.
232 */
233 template<typename T>
234 void reserve(size_t size) {
235 getOrCreateBuffer<T>().reserve(size);
236 }
237
238 /**
239 * @brief Gets or creates the buffer for message type T.
240 *
241 * @tparam T The message type.
242 *
243 * @return Reference to the ReadWriteDoubleBuffer for type T.
244 */
245 template<typename T>
246 ReadWriteDoubleBuffer<T>& getOrCreateBuffer() {
247 const size_t idx = index<T>();
248
249 if (buffers_.size() <= idx) {
250 buffers_.resize(idx + 1);
251 }
252
253 if (!buffers_[idx]) {
254 buffers_[idx] = std::make_unique<ReadWriteDoubleBuffer<T>>();
255 }
256
257 return *static_cast<ReadWriteDoubleBuffer<T>*>(buffers_[idx].get());
258 }
259
260 /**
261 * @brief Swaps read and write buffers for all message types.
262 *
263 * @details Call once per frame at a consistent point (e.g., end of update phase).
264 * After this call, messages pushed in the current frame become readable.
265 */
266 void swapBuffers() {
267 for (auto& buffer : buffers_) {
268 if (buffer) {
269 buffer->swap();
270 }
271 }
272 }
273
274 /**
275 * @brief Clears all read buffers.
276 *
277 * @details Typically not needed as `swapBuffers()` clears read buffers automatically.
278 */
279 void clearReadBuffers() {
280 for (auto& buffer : buffers_) {

284 }
285 }

292 void clearWriteBuffers() {
