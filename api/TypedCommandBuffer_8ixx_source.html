26 /**
27 * @brief Concept constraining commands that can self-execute.
28 *
29 * @details A command satisfies ExecutableCommand if it provides a
30 * noexcept `execute(UpdateContext&)` method. Commands that do not
31 * satisfy this concept must have a registered TypedCommandHandler.
32 *
33 * @tparam Cmd The command type to check.
34 */
35 template<typename Cmd>
36 concept ExecutableCommand = requires(Cmd const& c, UpdateContext& updateContext) {
37 {c.execute(updateContext) } noexcept;

41 /**
42 * @brief Compile-time typed command buffer with per-type queues and handler routing.
43 *
44 * @details TypedCommandBuffer stores commands in separate `std::vector` queues,
45 * one per command type, packed into a `std::tuple`. This provides:
46 *
47 * - **Zero-overhead dispatch:** Command types are known at compile time,
48 * eliminating virtual dispatch for queue access.
49 * - **Deterministic ordering:** Commands are flushed in the order of the
50 * template parameter list, ensuring reproducible execution.
51 * - **Handler-or-execute routing:** During flush, each command is either
52 * routed to a registered TypedCommandHandler or executed directly via
53 * its `execute()` method (if it satisfies ExecutableCommand).
54 *
55 * ## Flush Routing
56 *
57 * For each command type in the parameter pack:
58 * 1. If a `TypedCommandHandler<Cmd>` is registered → `handler.submit(cmd)`
59 * 2. Else if `Cmd` satisfies `ExecutableCommand` → `cmd.execute(ctx)`
60 * 3. Else → assertion failure (misconfiguration)
61 *
62 * @tparam CommandTypes The command types this buffer manages.
63 *
64 * @see CommandBuffer
65 * @see TypedCommandHandler
66 * @see EngineCommandBuffer
67 * @see ExecutableCommand
68 */
69 template <typename... CommandTypes>
70 class TypedCommandBuffer : public CommandBuffer {

77 /**

84 template<typename CommandType>

89 /**
90 * @brief Flushes a single command queue.
91 *
92 * @details Routes each command to its registered handler, or
93 * executes it directly if no handler is registered and the
94 * command satisfies ExecutableCommand.
95 *
96 * @tparam CommandType The command type to flush.
97 *
98 * @param updateContext The current frame's update context.
99 */
100 template<typename CommandType>
101 void flushCommandQueue(helios::engine::runtime::world::UpdateContext& updateContext) noexcept {
102
103 auto& queue = commandQueue<CommandType>();
104 if (queue.empty()) {
105 return;
106 }
107
108 if (updateContext.resourceRegistry().has<TypedCommandHandler<CommandType>>()) {
109 auto& handler = updateContext.resourceRegistry().resource<TypedCommandHandler<CommandType>>();
110 for (auto& cmd : queue) {
111 handler.submit(cmd);
112 }
113 } else {
114 if constexpr (ExecutableCommand<CommandType>) {
115 for (auto& cmd : queue) {
116 cmd.execute(updateContext);
117 }
118 } else {
119 assert(false && "Command type is not executable");
120 }
121
122
123
124 }
125
126 queue.clear();
127 }
