14 /**
15 * @brief Type-safe handler for a specific command type.
16 *
17 * @details TypedCommandHandler provides the interface for receiving commands
18 * of a specific type during the TypedCommandBuffer flush. Managers that
19 * process commands implement this for each command type they handle.
20 *
21 * A single Manager may implement multiple TypedCommandHandler specializations:
22 *
23 * ```cpp
24 * class SpawnManager : public Manager,
25 * public TypedCommandHandler<SpawnCommand>,
26 * public TypedCommandHandler<DespawnCommand> {
27 * bool submit(SpawnCommand cmd) noexcept override { ... }
28 * bool submit(DespawnCommand cmd) noexcept override { ... }
29 * };
30 * ```
31 *
32 * Handlers are registered with the ResourceRegistry during Manager::init()
33 * and looked up by the TypedCommandBuffer at flush time.
34 *
35 * @tparam CommandType The command type this handler accepts.
36 *
37 * @see CommandHandler
38 * @see TypedCommandBuffer
39 * @see Manager
40 */
41 template<typename CommandType>
42 class TypedCommandHandler : public CommandHandler{
