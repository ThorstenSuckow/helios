128 void update() {
129
130 if (!isDirty_) {

134 if (template_ == "{0}" || template_.empty()) {
135 renderable_->setText(std::to_string(doubleValue_));
136 needsResize_ = true;
137 isDirty_ = false;
138 return;
139 }

141 renderable_->setText(std::vformat(std::string_view(template_), std::make_format_args(doubleValue_)));
142 needsResize_ = true;
143 isDirty_ = false;
