35 /**
36 * @brief Computes screen positions for UI elements based on layout configuration.
37 *
38 * @details This system processes entities with UiTransformComponent and computes
39 * their final screen positions. The positioning algorithm considers:
40 *
41 * - **Anchor:** Where the element attaches to its parent (Center, TopLeft, etc.)
42 * - **Pivot:** The element's own reference point for positioning
43 * - **Offsets:** Margins from the anchor point (top, right, bottom, left)
44 * - **Viewport:** Current viewport dimensions for root-level elements
45 * - **Parent bounds:** For nested UI elements via HierarchyComponent
46 *
47 * ## Hierarchy Support
48 *
49 * For elements with a HierarchyComponent, the system uses the parent entity's
50 * AABB and transform to compute relative positioning. Root elements (those
51 * attached to the scene root) use viewport dimensions as their parent bounds.
52 *
53 * ## Required Components
54 *
55 * | Component | Purpose |
56 * |-----------|---------|
57 * | `UiTransformComponent` | Layout configuration (anchor, pivot, offsets) |
58 * | `TranslationStateComponent` | Receives computed position |
59 * | `ComposeTransformComponent` | Transform composition |
60 * | `ModelAabbComponent` | Element bounds for size calculations |
61 * | `SceneNodeComponent` | Scene graph integration |
62 * | `Active` | Lifecycle tag |
63 *
64 * @see UiTransformComponent
65 * @see Anchor
66 * @see HierarchyComponent
67 */
68 class UiTransformSystem : public helios::engine::ecs::System {

70 /**
71 * @brief Adjusts position based on pivot point and element size.
72 *
73 * @details Transforms a position from anchor-relative to pivot-relative
74 * coordinates. The pivot determines which point of the element is placed
75 * at the anchor position.
76 *
77 * @param unanchored The position before pivot adjustment.
78 * @param size The size of the element.
79 * @param pivot The pivot point to apply.
80 *
81 * @return The adjusted position accounting for pivot offset.
82 */
83 helios::math::vec3f anchor(
84 const helios::math::vec3f unanchored,
85 helios::math::vec3f size,
86 const helios::engine::modules::ui::layout::Anchor pivot) {
87
88 switch (pivot) {
89
90 case helios::engine::modules::ui::layout::Anchor::Center:
91 return unanchored - size * 0.5f;
92
93 case helios::engine::modules::ui::layout::Anchor::TopRight:
94 return unanchored - size;
95
96 case helios::engine::modules::ui::layout::Anchor::TopLeft:
97 return unanchored.withY(unanchored[1] - size[1]);
98
99 case helios::engine::modules::ui::layout::Anchor::BottomLeft:
100 return unanchored;
101 }
102
103 assert(false && "Unreachable!");
104 std::unreachable();
105 }

112 * @details Iterates over all entities with UI layout components and computes

114 *

116 * 2. Determines parent bounds (viewport for root, parent AABB otherwise)
117 * 3. Computes anchor position based on anchor type and offsets
