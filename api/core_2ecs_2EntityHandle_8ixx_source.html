18 /**
19 * @brief A versioned handle for referencing entities in a registry.
20 *
21 * `EntityHandle` combines an entity ID with a version number and a
22 * domain-specific strong ID to provide safe, typed references to
23 * registered entities. The version ensures that handles to removed
24 * entities are detected as stale.
25 *
26 * @tparam TStrongId A strong ID type carrying domain semantics.
27 *
28 * @see EntityRegistry
29 */
30 template<typename TStrongId>
31 struct EntityHandle {
32
33 /**
34 * @brief The unique identifier for the entity within the registry.
35 */
36 EntityId entityId;
37
38 /**
39 * @brief The version number for stale handle detection.
40 *
41 * Incremented when an entity is removed from the registry.
42 */
43 VersionId versionId;
44
45 /**
46 * @brief The domain-specific strong ID associated with this handle.
47 */
48 TStrongId strongId;
49
50 /**
51 * @brief Compares two handles for equality.
52 */
53 bool operator==(const EntityHandle&) const = default;
54
55 /**
56 * @brief Compares two handles by entity ID (less-than).
57 *
58 * @param entityHandle The handle to compare against.
59 *
60 * @return True if this handle's entity ID is less than the other.
61 */
62 constexpr bool operator<(const EntityHandle& entityHandle) const noexcept {
63 return entityId < entityHandle.entityId;
64 }
65
66 /**
67 * @brief Compares two handles by entity ID (greater-than).
68 *
69 * @param entityHandle The handle to compare against.
70 *
71 * @return True if this handle's entity ID is greater than the other.
72 */
73 constexpr bool operator>(const EntityHandle& entityHandle) const noexcept {
74 return entityId > entityHandle.entityId;
