24 /**
25 * @brief Generates and registers ComponentOps for a component type.
26 *
27 * @details `ComponentReflector` uses compile-time trait detection to
28 * generate appropriate function pointers for each lifecycle hook that
29 * a component type implements. This enables zero-overhead conditional
30 * invocation at runtime.
31 *
32 * ## Template Parameters
33 *
34 * The reflector is templated on `THandle` and `TEntityManager` so that
35 * the generated clone function can cast the type-erased `void*` pointers
36 * back to the concrete types of the target domain. This also ensures
37 * that `ComponentOpsRegistry<THandle>` stores the ops in the correct
38 * per-domain registry.
39 *
40 * ## Usage
41 *
42 * ```cpp
43 * // Register a component type for the game domain
44 * ComponentReflector<GameHandle, GameEM>::registerType<HealthComponent>();
45 *
46 * // Or use the module registry pattern
47 * inline void registerComponents() {
48 * using R = ComponentReflector<GameHandle, GameEM>;
49 * R::registerType<CollisionComponent>();
50 * R::registerType<AabbColliderComponent>();
51 * }
52 * ```
53 *
54 * ## Trait Detection
55 *
56 * The reflector checks for the following traits at compile time:
57 * - `HasOnAcquire` - `onAcquire()` method
58 * - `HasOnRelease` - `onRelease()` method
59 * - `HasOnRemove` - `onRemove()` method returning bool
60 * - `HasToggleable` - `enable()` and `disable()` methods
61 * - `HasClone` - `onClone(const T&)` method
62 * - `HasActivatable` - `onActivate()` and `onDeactivate()` methods
63 *
64 * @tparam THandle The concrete `EntityHandle<TStrongId>` specialisation.
65 * Used to scope `ComponentTypeId` and `ComponentOpsRegistry`.
66 * @tparam TEntityManager The concrete `EntityManager` specialisation.
67 * Used by the clone function to emplace components
68 * on the target entity.
69 *
70 * @see ComponentOps
71 * @see ComponentOpsRegistry
72 * @see Traits
73 */

82 /**
83 * @brief Registers a component type with the reflection system.
84 *
85 * @details Generates `ComponentOps` function pointers based on which
86 * lifecycle hooks the type implements. Uses `if constexpr` for
87 * zero-overhead when hooks are not present.
88 *
89 * @tparam T The component type to register. Must be copy-constructible
90 * for cloning support.
91 */
92 template<typename T>
93 static void registerType() {
94
95 ComponentOps ops{
96
97 .onAcquire = [](void* ptr) {
98 if constexpr (HasOnAcquire<T>) {
99 static_cast<T*>(ptr)->onAcquire();
100 }
101 },
102
103 .onRelease = [](void* ptr) {
104 if constexpr (HasOnRelease<T>) {
105 static_cast<T*>(ptr)->onRelease();
106 }
107 },
108
109 .onRemove = [](void* ptr) -> bool {
110 if constexpr (HasOnRemove<T>) {
111 return static_cast<T*>(ptr)->onRemove();
112 }
113 return true;
114 },
115
116 .enable = [](void* ptr) {
117 if constexpr (HasToggleable<T>) {
118 static_cast<T*>(ptr)->enable();
119 }
120 },
121
122 .disable = [](void* ptr) {
123 if constexpr (HasToggleable<T>) {
124 static_cast<T*>(ptr)->disable();
125 }
126 },
127
128 .clone = [](void* managerRaw, const void* sourceRaw, const void* targetRaw) -> void* {
129
130 auto* manager = static_cast<TEntityManager*>(managerRaw);
131 const auto* source = static_cast<const T*>(sourceRaw);
132 const auto* target = static_cast<const Handle_type*>(targetRaw);
133
134 T* cmp = nullptr;
135 if constexpr (std::is_copy_constructible_v<T>) {
136 cmp = manager->template emplace<T>(*target, *source);
137 } else {
138 return nullptr;
139 }
140
141 if (!cmp) {
142 return nullptr;
143 }
144
145 if constexpr (HasClone<T>) {
146 cmp->onClone(*source);
147 }
148
149 return cmp;
150 },
151
152 .onActivate = [](void* ptr) {
153 if constexpr (HasActivatable<T>) {
154 static_cast<T*>(ptr)->onActivate();
155 }
156 },
157
158 .onDeactivate = [](void* ptr) {
159 if constexpr (HasActivatable<T>) {
160 static_cast<T*>(ptr)->onDeactivate();
161 }
162 },
163
164 };
165
166 const auto typeId = ComponentTypeId<Handle_type>::template id<T>();
167
168 ComponentOpsRegistry<Handle_type>::setOps(typeId, ops);
169
170 }
171
172 };
