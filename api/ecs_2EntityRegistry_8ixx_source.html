27 /**
28 * @brief Generic registry for creating and managing strongly-typed entity handles.
29 *
30 * `EntityRegistry` serves as the authoritative source for entity lifecycle
31 * management. It is responsible for:
32 *
33 * - **Handle Creation:** Generates unique `EntityHandle` instances with versioned IDs
34 * and domain-specific strong IDs.
35 * - **Validation:** Determines whether a given handle refers to a currently alive entity.
36 * - **Destruction:** Marks entities as destroyed by incrementing their version and
37 * recycling the index for future use (when `TAllowRemoval` is true).
38 *
39 * ## Versioning
40 *
41 * Each entity slot maintains a version number. When an entity is destroyed, its
42 * version is incremented. This allows the registry to detect stale handles —
43 * handles that reference an entity that has been destroyed and potentially replaced
44 * by a new entity at the same index.
45 *
46 * ## Index Recycling
47 *
48 * Destroyed entity indices are added to a free list and reused when creating new
49 * entities. This keeps the dense version array compact and minimizes memory growth.
50 *
51 * ## Strong ID Collision Detection
52 *
53 * The registry delegates collision checks to a configurable `TLookupStrategy`,
54 * which tracks which strong IDs are currently in use. The default strategy
55 * (`HashedLookupStrategy`) provides O(1) amortized lookups.
56 *
57 * @tparam TStrongId A strong ID type carrying domain semantics.
58 * @tparam TLookupStrategy The strategy used for strong ID collision detection.
59 * @tparam TAllowRemoval If false, `destroy()` triggers an assertion instead of removing.
60 * @tparam TCapacity Default initial capacity for pre-allocation.
61 *
62 * @see EntityHandle
63 * @see HashedLookupStrategy
64 * @see LinearLookupStrategy
65 */
66 template<
67 typename TStrongId,
68 typename TLookupStrategy = HashedLookupStrategy,
69 bool TAllowRemoval = true,
70 size_t TCapacity = DEFAULT_ENTITY_MANAGER_CAPACITY
71 >
72 requires helios::ecs::concepts::IsStrongIdLike<TStrongId> &&
73 helios::ecs::concepts::IsStrongIdCollisionResolverLike<TLookupStrategy>
74 class EntityRegistry {

109 /**
110 * @brief Constructs a registry with pre-allocated capacity.
111 *
112 * Reserving capacity upfront can improve performance when the approximate
113 * number of entities is known in advance.
114 *
115 * @param capacity The initial capacity to reserve.
116 */
117 explicit EntityRegistry(const size_t capacity = TCapacity)
118 : lookupStrategy_(capacity) {
119 versions_.reserve(capacity);
120 strongIds_.reserve(capacity);
121 freeList_.reserve(capacity);
122 }
123
124
125 /**
126 * @brief Creates a new entity and returns its handle.
127 *
128 * If recycled indices are available in the free list, one is reused.
129 * Otherwise, a new index is allocated at the end of the version vector.
130 * If the provided `strongId` is invalid, one is auto-assigned from the
131 * entity index.
132 *
133 * @param strongId Optional domain-specific strong ID. Defaults to auto-assignment.
134 *
135 * @return A valid `EntityHandle` for the newly created entity.
136 */
137 EntityHandle<TStrongId> create(TStrongId strongId = TStrongId{}) {
138
139 EntityId idx;
140 VersionId version;
141
142 if (freeList_.empty()) {
143 idx = static_cast<EntityId>(versions_.size());
144 version = InitialVersion;
145 versions_.push_back(version);
146 strongIds_.push_back(StrongId_t{});
147 } else {
148 idx = freeList_.back();
149 freeList_.pop_back();
150
151 // version was already incremented in destroy
152 version = versions_[idx];
153 }
154
155 if (!strongId.isValid()) {
156 strongId = TStrongId{static_cast<StrongId_t>(++strongIdCounter_)};
157 }
158 assert(strongId.isValid() && "EntityRegistry: invalid strongId");
159
160 assert(!lookupStrategy_.has(strongId.value()) && "EntityRegistry: strongId collision");
161
162 strongIds_[idx] = strongId.value();
163
164 const bool added = lookupStrategy_.add(strongId.value());
165
166 assert(added && "EntityRegistry: failed to add strongId to lookupStrategy");
167
168 return {idx, version, strongId};
169
170 }

172 /**
173 * @brief Looks up the version for an EntityId.
174 *
175 * @param entityId The entity to retrieve the version for.
176 *
177 * @return The version for the EntityId, or `InvalidVersion` if not found.
178 */
179 [[nodiscard]] VersionId version(const EntityId entityId) const {
180 if (entityId >= static_cast<EntityId>(versions_.size())) {
181 return InvalidVersion;
182 }
183 return versions_[entityId];
184 }

186 /**
187 * @brief Looks up the strong ID for an entity index.
188 *
189 * @param entityId The entity index to retrieve the strong ID for.
190 *
191 * @return The strong ID for the entity, or a default-constructed
192 * (invalid) `TStrongId` if out of bounds.
193 */
194 [[nodiscard]] TStrongId strongId(const EntityId entityId) const {
195 if (entityId >= static_cast<EntityId>(strongIds_.size())) {
196 return TStrongId{};
197 }
198 return static_cast<TStrongId>(strongIds_[entityId]);
199 }
200
201
202 /**
203 * @brief Checks whether the given handle refers to a currently alive entity.
204 *
205 * A handle is valid if its index is within bounds, its version matches
206 * the current version stored in the registry, and its strong ID matches
207 * the registered value.
208 *
209 * @param handle The handle to validate.
210 *
211 * @return True if the handle is valid and the entity is alive.
212 */
213 [[nodiscard]] bool isValid(const EntityHandle<TStrongId> handle) const noexcept {
214 const auto index = handle.entityId;
215
216 if (index >= static_cast<EntityId>(versions_.size())) {
217 return false;
218 }
219
220 return handle.strongId.value() == strongIds_[index] &&
221 handle.versionId == versions_[index] &&
222 handle.strongId.isValid();
223 }
224
225
226 /**
227 * @brief Destroys the entity referenced by the given handle.
228 *
229 * If the handle is valid, the entity's version is incremented (invalidating
230 * all existing handles to it), the strong ID is unregistered, and the index
231 * is added to the free list for recycling.
232 *
233 * If `TAllowRemoval` is false, triggers an assertion failure.
234 *
235 * @param handle The handle of the entity to destroy.
236 *
237 * @return True if the entity was successfully destroyed, false if the handle
238 * was already invalid.
239 */
240 bool destroy(const EntityHandle<TStrongId> handle) {
241
242 if constexpr (!TAllowRemoval) {
243 assert(false && "EntityRegistry: Entity removal is not allowed");
244 return false;
245 }
246
247 if (!isValid(handle)) {
248 return false;
249 }
250
251 const auto index = handle.entityId;
252
253 versions_[index] += 1;
254 strongIds_[index] = 0;
255
256 const bool removed = lookupStrategy_.remove(handle.strongId.value());
257 assert(removed && "EntityRegistry: failed to remove strongId from lookupStrategy");
258
259 freeList_.push_back(index);
260
261 return true;
262 }
263
264 };
