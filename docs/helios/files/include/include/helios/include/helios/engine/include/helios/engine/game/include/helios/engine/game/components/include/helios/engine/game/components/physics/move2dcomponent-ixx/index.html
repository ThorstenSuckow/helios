26 /**
27 * @brief Component for 2D physics-based movement.
28 *
29 * @details
30 * Stores configuration and runtime state for 2D physics-based movement.
31 * This component is used by the Move2DSystem to apply smooth acceleration
32 * and deceleration to the entity.
33 *
34 * The component receives input via move() which sets target direction and throttle.
35 * The actual physics simulation (integration of velocity, application of dampening)
36 * is performed by the Move2DSystem.
37 *
38 * @note Rotation/heading is handled separately by HeadingComponent and HeadingSystem.
39 *
40 * @see helios::engine::game::components::physics::HeadingComponent
41 * @see helios::engine::game::systems::physics::Move2DSystem
42 */

146 /**
147 * @brief Sets the movement direction and throttle.
148 *
149 * @param direction Normalized 3D direction vector.
150 * @param throttle Magnitude of the stick input (0.0 to 1.0).
151 */
152 void move(helios::math::vec3f direction, float throttle) {
153 if (throttle <= helios::math::EPSILON_LENGTH) {
154 steeringInput_ = helios::math::vec2f{0.0f, 0.0f};
155 throttle_ = 0.0f;
156 isInputActive_ = false;
157 return;
158 }
159
160 steeringInput_ = direction.toVec2();
161 throttle_ = throttle;
162
163 assert(std::abs(direction.length() - 1.0f) <= 0.001f && "Unexpected direction vector - not normalized");
164
165
166 isInputActive_ = true;
167
168 currentMovementSpeed_ = movementSpeed_ * throttle_;
169
170 }
