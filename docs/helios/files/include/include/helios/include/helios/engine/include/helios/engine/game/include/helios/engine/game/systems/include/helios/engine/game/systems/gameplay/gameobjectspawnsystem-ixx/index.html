34 /**
35 * @brief System responsible for spawning GameObjects into the level.
36 *
37 * @details
38 * This system manages the activation of GameObjects from a pool and their
39 * placement within the level. It handles the initialization of SceneNodeComponents
40 * for renderable objects during the warmup phase.
41 *
42 * During update, it acquires inactive GameObjects from the pool, and delegates
43 * Spawn condition and logic to SpawnCondition and SpawnStrategy, respectively.
44 */
45 class GameObjectSpawnSystem : public System {
46
47 /**
48 * @brief The pool of GameObjects to manage.
49 */
50 std::unique_ptr<helios::engine::game::GameObjectPool> gameObjectPool_;
51
52 /**
53 * @brief Reference to the active level.
54 */
55 const helios::engine::game::Level& level_;
56
57 /**
58 * @brief Strategy used to determine how GameObjects are spawned in the level.
59 */
60 std::unique_ptr<helios::engine::game::SpawnStrategy> spawnStrategy_;
61
62 /**
63 * @brief Strategy for determining the conditions under which GameObjects are spawned.
64 */
65 std::unique_ptr<helios::engine::game::SpawnCondition> spawnCondition_;
66
67 /**
68 * @brief Prepares the object pool and scene graph.
69 *
70 * @details
71 * Iterates through all inactive objects in the pool. If an object has a
72 * RenderableComponent, a corresponding SceneNode is created and attached
73 * to the level's root node. A SceneNodeComponent is then added to the
74 * GameObject to link it with the scene graph.
75 */
76 void warmup() {
77
78 std::vector<helios::engine::game::GameObject*> gameObjects = gameObjectPool_->inactiveGameObjects();
79
80 auto* spawnNode = level_.rootNode();
81
82 for (auto* go : gameObjects) {
83 auto* rc = go->get<helios::engine::game::components::rendering::RenderableComponent>();
84
85 assert(rc != nullptr && "RenderableComponent must not be nullptr");
86
87 if (!rc) {
88 continue;
89 }
90
91 auto sceneNode = std::make_unique<helios::scene::SceneNode>(rc->renderable());
92 auto sceneNode_ptr = sceneNode.get();
93 std::ignore = spawnNode->addNode(std::move(sceneNode));
94 go->add<helios::engine::game::components::scene::SceneNodeComponent>(sceneNode_ptr);
95 go->setActive(false);
96 }
97
98 }

102 /**
103 * @brief Constructs a new GameObjectSpawnSystem.
104 *
105 * @param gameObjectPool Unique pointer to the GameObjectPool to manage.
106 * @param level Reference to the Level where objects will be spawned.
107 */
108 explicit GameObjectSpawnSystem(
109 std::unique_ptr<helios::engine::game::GameObjectPool> gameObjectPool,
110 const helios::engine::game::Level& level,
111 std::unique_ptr<helios::engine::game::SpawnCondition> spawnCondition,

113 ) :

115 level_(level),
116 spawnCondition_(std::move(spawnCondition)),
117 spawnStrategy_(std::move(spawnStrategy))
118 {
119 assert(spawnStrategy_ != nullptr && "Unexpected nullptr for SpawnStrategy");
120 assert(spawnCondition_ != nullptr && "Unexpected nullptr for SpawnCondition");
121 assert(gameObjectPool_ != nullptr && "Unexpected nullptr for GameObjectPool");
122 warmup();
123 }

126 /**
127 * @brief Updates the system by spawning GameObjects based on SpawnCondition and SpawnStrategy.
128 *
129 * @details Checks whether there are remaining inactive objects that could be spawned.
130 * Delegates to the SpawnCondition this instance was configured with to determine how many
131 * objects should be considered for spawning, then calls the SpawnStrategy for spawning
132 * the GameObjects into the GameWorld.
133 *
134 * @param updateContext The update context for the current frame.
135 */
136 void update(helios::engine::game::UpdateContext& updateContext) noexcept override {
137
138 auto available = gameObjectPool_->inactiveCount();
139 if (available == 0) {
140 return;
141 }
142
143 size_t spawnCount = spawnCondition_->spawnBudget(gameWorld_, &level_, gameObjectPool_.get(), updateContext);
144
145 spawnCount = std::min(spawnCount, available);
146
147 for (size_t i = 0; i < spawnCount; i++) {
148 helios::engine::game::GameObject* go = gameObjectPool_->acquire();
149
150 if (go == nullptr) {
151 return;
152 }
153
154 if (!spawnStrategy_->spawn(gameWorld_, &level_, go, updateContext)) {
155 gameObjectPool_->release(go);
156 }
157 }
158
159 }

161 };
