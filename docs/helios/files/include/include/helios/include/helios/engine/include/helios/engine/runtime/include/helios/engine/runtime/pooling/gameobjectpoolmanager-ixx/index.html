32 /**
33 * @brief High-level manager for GameObject pooling operations.
34 *
35 * @details GameObjectPoolManager provides a unified interface for managing
36 * multiple GameObjectPools. It handles pool configuration, initialization,
37 * and the acquire/release lifecycle of pooled GameObjects.
38 *
39 * The manager integrates with the GameWorld to clone prefabs during pool
40 * initialization and to look up GameObjects by their GUID during
41 * acquire/release operations.
42 *
43 * ## Lifecycle
44 *
45 * 1 **Configuration**: Add pool configurations via `addPoolConfig()`.
46 * 2 **Initialization**: Call `init()` to create pools and populate them
47 * with cloned prefab instances.
48 * 3 **Runtime**: Use `acquire()` and `release()` to manage entity lifecycle.
49 *
50 * ## Example
51 *
52 * ```cpp
53 * GameObjectPoolManager poolManager;
54 *
55 * auto config = std::make_unique<GameObjectPoolConfig>(
56 * bulletPoolId, bulletPrefab, 100
57 * );
58 * poolManager.addPoolConfig(std::move(config));
59 *
60 * poolManager.init(gameWorld);
61 *
62 * if (auto* bullet = poolManager.acquire(bulletPoolId)) {
63 * bullet->get<TransformComponent>()->setTranslation(spawnPos);
64 * }
65 *
66 * poolManager.release(bulletPoolId, bullet->guid());
67 * ```
68 *
69 * @see GameObjectPool
70 * @see GameObjectPoolRegistry
71 * @see Manager
72 */
73 class GameObjectPoolManager : public helios::engine::runtime::world::Manager {

93 *

101 /**
102 * @brief Populates a pool with cloned prefab instances.
103 *
104 * @details Clones the prefab GameObject until the pool reaches its
105 * configured capacity. Each clone is immediately deactivated and
106 * marked as inactive in the pool.
107 *
108 * @param gameObjectPoolId The pool to fill.
109 * @param gameObjectPrefab The prefab to clone.
110 */
111 void fillPool(
112 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId,
113 const helios::engine::ecs::GameObject& gameObjectPrefab
114 ) {
115 helios::util::Guid guid{helios::util::no_init};
116
117 auto* gameObjectPool = pool(gameObjectPoolId);
118
119 const size_t used = gameObjectPool->activeCount() + gameObjectPool->inactiveCount();
120 const size_t space = used < gameObjectPool->size() ? gameObjectPool->size() - used : 0;
121
122 for (size_t i = 0; i < space; i++) {
123 helios::engine::ecs::GameObject* go = gameWorld_->clone(gameObjectPrefab);
124 if (go) {
125 go->setActive(false);
126 go->onRelease();
127 gameObjectPool->addInactive(go->guid());
128 }
129 }
130 }

135 * @brief Registers a pool configuration for later initialization.
136 *
137 * @details The configuration specifies the pool ID, capacity, and prefab
138 * to use for cloning. Configurations are processed during `init()`.
139 *
140 * @param gameObjectPoolConfig The pool configuration to register.

146 GameObjectPoolManager& addPoolConfig(std::unique_ptr<GameObjectPoolConfig> gameObjectPoolConfig) {

177 /**
178 * @brief Releases a GameObject back to its pool.
179 *
180 * @details Marks the entity as inactive in both the pool and the
181 * GameWorld. Calls `onRelease()` on the GameObject to allow cleanup.
182 *
183 * @param gameObjectPoolId The pool that owns this entity.
184 * @param entityId The GUID of the entity to release.
185 *
186 * @return Pointer to the released GameObject, or nullptr if not found.
187 */
188 helios::engine::ecs::GameObject* release(
189 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId,
190 const helios::util::Guid& entityId
191 ) {
192 auto* gameObjectPool = pool(gameObjectPoolId);
193
194 helios::engine::ecs::GameObject* worldGo = gameWorld_->find(entityId);
195
196 if (worldGo) {
197 if (gameObjectPool->release(entityId)) {
198 worldGo->onRelease();
199 worldGo->setActive(false);
200 }
201 }
202
203 return worldGo;
204 }

206 /**
207 * @brief Acquires an inactive GameObject from the pool.
208 *
209 * @details Retrieves the next available inactive entity, activates it,
210 * and calls `onAcquire()` to prepare it for use.
211 *
212 * If an entity's GUID is no longer valid in the GameWorld, it is removed
213 * from the pool and the next entity is tried.
214 *
215 * @param gameObjectPoolId The pool to acquire from.
216 *
217 * @return Pointer to the acquired GameObject, or nullptr if pool exhausted.
218 */
219 [[nodiscard]] helios::engine::ecs::GameObject* acquire(
220 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId
221 ) {
222 helios::util::Guid guid{helios::util::no_init};
223
224 auto* gameObjectPool = pool(gameObjectPoolId);
225
226 while (gameObjectPool->acquire(guid)) {
227
228 auto* worldGo = gameWorld_->find(guid);
229
230 if (worldGo) {
231 worldGo->onAcquire();
232 return worldGo;
233 }
234
235 // we assume the pool is owned by this gameWorld,
236 // so removing this guid does not impact another gameWorld that is
237 // using this pool
238 gameObjectPool->releaseAndRemove(guid);
239 }
240
241 return nullptr;
242
243 }

245 /**
246 * @brief Initializes all registered pools.
247 *
248 * @details Creates GameObjectPool instances from the pending configurations
249 * and populates them by cloning the specified prefabs into the GameWorld.
250 *
251 * @param gameWorld The GameWorld to associate with this manager.
252 */
253 void init(helios::engine::runtime::world::GameWorld& gameWorld) override {
254
255 gameWorld_ = &gameWorld;
256
257 for (const auto& [gameObjectPoolId, poolConfig] : poolConfigs_) {
258
259 auto pool = std::make_unique<helios::engine::runtime::pooling::GameObjectPool>(
260 poolConfig->amount);
261
262 pools_.addPool(gameObjectPoolId, std::move(pool));
263 fillPool(gameObjectPoolId, *poolConfig->prefab);
264 }
265
266 };
