47 /**
48 * @brief Central registry for game entities, managers, pools, and the active level.
49 *
50 * @details
51 * The GameWorld is the root container for the game state. It manages the lifecycle
52 * of GameObjects, coordinates Managers, and holds the current Level.
53 *
54 * ## Key Responsibilities
55 *
56 * - **Entity Management:** Owns all GameObjects and provides amortized O(1) lookup by Guid
57 * via `std::unordered_map`.
58 * - **Component Queries:** Efficient iteration over GameObjects with specific components
59 * via `find<Components...>()` with optional filtering by active/enabled state.
60 * - **Pool Management:** Registers and provides access to GameObjectPools for entity recycling.
61 * - **Manager Coordination:** Holds Managers that handle cross-cutting concerns
62 * (spawning, projectile pooling) and flushes them each frame.
63 * - **Level Management:** Holds the active Level instance with arena bounds.
64 *
65 * ## Usage with GameLoop
66 *
67 * The GameWorld is passed to Systems via UpdateContext. Systems query GameObjects
68 * and their components, while mutations are typically performed through Commands
69 * that are flushed by the CommandBuffer.
70 *
71 * ```cpp
72 * // In a System
73 * void update(UpdateContext& ctx) noexcept override {
74 * for (auto [obj, move] : ctx.gameWorld().find<Move2DComponent>().each()) {
75 * // Process entities with Move2DComponent
76 * }
77 * }
78 * ```
79 *
80 * ## Filtering
81 *
82 * The `find()` methods accept a `GameObjectFilter` bitmask to control which
83 * entities are returned:
84 *
85 * - `GameObjectFilter::Active` — Only active GameObjects
86 * - `GameObjectFilter::ComponentEnabled` — Only objects with enabled components
87 *
88 * Default filter is `Active | ComponentEnabled`.
89 *
90 * @see GameObject
91 * @see GameObjectFilter
92 * @see UpdateContext
93 * @see Manager
94 * @see Level
95 */

392 /**
393 * @brief Finds all GameObjects that have the specified component types.
394 *
395 * @details
396 * Returns a lazy range that filters GameObjects on-the-fly during iteration.
397 * Only GameObjects possessing all specified component types are yielded.
398 *
399 * The query is controlled by a `GameObjectFilter` bitmask:
400 *
401 * | Filter | Effect |
402 * |--------|--------|
403 * | `Active` | Only `obj->isActive() == true` |
404 * | `Inactive` | Only `obj->isActive() == false` |
405 * | `ComponentEnabled` | Only objects where queried components are enabled |
406 * | `ComponentDisabled` | Only objects where queried components are disabled |
407 *
408 * Filters can be combined: `Active | ComponentEnabled` (default).
409 *
410 * Example usage:
411 * ```cpp
412 * // Range-based for loop with default filter (Active + ComponentEnabled)
413 * for (auto* obj : gameWorld.find<Move2DComponent, SceneNodeComponent>()) {
414 * auto* move = obj->get<Move2DComponent>();
415 * move->setVelocity({1.0f, 0.0f, 0.0f});
416 * }
417 *
418 * // Using each() for direct component access (structured bindings)
419 * for (auto [obj, move] : gameWorld.find<Move2DComponent>().each()) {
420 * move.get().setVelocity({1.0f, 0.0f, 0.0f});
421 * }
422 *
423 * // Custom filter: find inactive objects
424 * for (auto* obj : gameWorld.find<HealthComponent>(GameObjectFilter::Inactive)) {
425 * // Process inactive entities (e.g., respawn logic)
426 * }
427 * ```
428 *
429 * @tparam Cs The component types to filter by. GameObjects must have all types.
430 *
431 * @param query Filter bitmask controlling which entities are included.
432 * Default: `Active | ComponentEnabled`.
433 *
434 * @return A GameObjectView object that can be iterated over.
435 *
436 * @see GameObjectFilter
437 * @see GameObjectView
438 */
439 template<class... Cs>
440 [[nodiscard]] auto find(GameObjectFilter query = GameObjectFilter::Active | GameObjectFilter::ComponentEnabled) {
441 return GameObjectRange<Map, Cs...>(gameObjects_, query);
442 }

479 /**
480 * @brief Creates a clone of an existing GameObject.
481 *
482 * @details Clones all Cloneable components from the source GameObject.
483 * The cloned GameObject is initially inactive and receives a new Guid.
484 * Non-cloneable components are skipped with a warning.
485 *
486 * @param gameObject The source GameObject to clone.
487 *
488 * @return Pointer to the newly created clone, or nullptr on failure.
489 *
490 * @note The clone is added to the world in an inactive state.
491 */
492 [[nodiscard]] helios::engine::ecs::GameObject* clone(helios::engine::ecs::GameObject& gameObject) {
493
494
495 auto newGo = std::make_unique<helios::engine::ecs::GameObject>();
496 /**
497 * @todo Optional ClonePolicy where rules are specified?
498 */
499 newGo->setActive(false);
500
501 for (const auto& component : gameObject.components()) {
502 if (const auto* cc = dynamic_cast<const helios::engine::ecs::Cloneable*>(component.get())) {
503 auto cComponent = cc->clone();
504 // use getOrAdd since cloned components may have already added
505 // other components in onAttach(), which we will not defer for now.
506 // we assume that the cloned component that allows for adding
507 // components in onAttach is the source of truth for said components
508 newGo->getOrAdd(std::move(cComponent));
509 } else {
510 logger_.warn("Component not cloneable.");
511 }
512
513 }
514
515 return addGameObject(std::move(newGo));
516 }
