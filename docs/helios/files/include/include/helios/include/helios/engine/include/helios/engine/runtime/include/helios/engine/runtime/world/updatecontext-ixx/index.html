27 /**
28 * @brief Context passed to systems and components during per-frame updates.
29 *
30 * Provides all necessary state for systems to perform their update logic,
31 * including frame timing, input state, command buffer access, and event propagation.
32 *
33 * The UpdateContext serves as the central communication hub within the game loop,
34 * offering two levels of event propagation:
35 *
36 * **Pass-level events** are pushed via pushPass() and become readable in subsequent
37 * passes within the same phase after a commit point is reached.
38 *
39 * **Phase-level events** are pushed via pushPhase() and become readable in subsequent
40 * phases after the current phase commits.
41 *
42 * @see GameLoop
43 * @see Pass
44 */
45 struct UpdateContext {
46
47 private:
48 /**
49 * @brief Time elapsed since the last frame, in seconds.
50 */
51 float deltaTime_ = 0.0f;
52
53 /**
54 * @brief Immutable snapshot of input state for the current frame.
55 */
56 const helios::input::InputSnapshot* inputSnapshot_ = nullptr;
57
58 /**
59 * @brief Buffer for queueing commands to be executed at end of frame.
60 */
61 helios::engine::runtime::messaging::command::CommandBuffer& commandBuffer_;
62
63 /**
64 * @brief Reference to the game world for entity lookups.
65 */
66 helios::engine::runtime::world::GameWorld& gameWorld_;
67
68 /**
69 * @brief Sink for pushing phase-level events during update.
70 *
71 * Used by systems and components to publish events (e.g., collision,
72 * spawn requests) that will be processed in the next phase of the game loop.
73 */
74 helios::engine::runtime::messaging::event::GameLoopEventBus::WriteSink phaseEventSink_;
75
76 /**
77 * @brief Source for reading phase-level events from the previous phase.
78 */
79 const helios::engine::runtime::messaging::event::GameLoopEventBus::ReadSource phaseEventSource_;
80
81 /**
82 * @brief Sink for pushing pass-level events during update.
83 *
84 * Used by systems and components to publish events that will be processed
85 * in subsequent passes within the same phase, after a commit point.
86 */
87 helios::engine::runtime::messaging::event::GameLoopEventBus::WriteSink passEventSink_;
88
89 /**

98 * @brief Constructs an UpdateContext with required dependencies.
99 *
100 * @param commandBuffer Reference to the command buffer for queueing commands.
101 * @param gameWorld Reference to the game world for entity lookups.
102 * @param phaseEventBus Reference to the phase-level event bus for cross-phase communication.
103 * @param passEventBus Reference to the pass-level event bus for cross-pass communication.

105 UpdateContext(

180 /**
181 * @brief Pushes an event to the pass-level event bus.
182 *
183 * Events pushed here become readable in subsequent passes within the same
184 * phase, after a commit point is reached.
185 *
186 * @tparam E The event type to push.
187 * @tparam Args Constructor argument types for the event.
188 *
189 * @param args Arguments forwarded to the event constructor.

199 /**
200 * @brief Pushes an event to the phase-level event bus.
201 *
202 * Events pushed here become readable in subsequent phases,
203 * after the current phase commits.
204 *
205 * @tparam E The event type to push.
206 * @tparam Args Constructor argument types for the event.
207 *
208 * @param args Arguments forwarded to the event constructor.
209 *
210 * @see readPhase()
211 * @see GameLoop
212 */
213 template<typename E, typename... Args>
214 void pushPhase(Args&&... args) {
215 phaseEventSink_.template push<E>(std::forward<Args>(args)...);
216 }
217
218 /**
219 * @brief Reads events from the phase-level event bus.
220 *
221 * Returns events that were pushed during the previous phase via
222 * `pushPhase()`. The phase event bus is swapped at phase boundaries,
223 * configured in GameLoop::phaseCommit().
224 *
225 * @tparam E The event type to read.
226 *
227 * @return A span of const events of type E.
228 *
229 * @see pushPhase()
230 * @see GameLoop
231 */
232 template<typename E>
233 std::span<const E> readPhase() {
234 return phaseEventSource_.template read<E>();
235 }
236
237 /**
238 * @brief Reads events from the pass-level event bus.
239 *
240 * Returns events that were pushed during previous passes within the
241 * current phase via `pushPass()`. The pass event bus is swapped at
242 * commit points, configured via Pass::addCommitPoint().
243 *
244 * @tparam E The event type to read.
245 *
246 * @return A span of const events of type E.
247 *
248 * @see pushPass()
249 * @see Pass::addCommitPoint()
250 */
251 template<typename E>
252 std::span<const E> readPass() {
253 return passEventSource_.template read<E>();
254 }
