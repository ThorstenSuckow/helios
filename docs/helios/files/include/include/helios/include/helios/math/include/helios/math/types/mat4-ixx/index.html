20 /**
21 * @brief Represents a 4x4 matrix, stored in column-major order.
22 *
23 * The `mat4` struct provides a lightweight and efficient way to handle 4D
24 * matrix mathematics for numeric data types.
25 * It stores its components as type `T`. For convenient access, type aliases
26 * `mat4f`, `mat4d` and `mat4i` are available, providing float/double/integer
27 * matrix representatives.
28 *
29 * @tparam T the numeric type of the matrix components.
30 */
31 template<helios::math::Numeric T>
32 struct mat4 {
33 private:
34 /**
35 * @brief Internal array storing matrix components.
36 *
37 * Components are stored in column-major order, that is, each 4 subsequent elements
38 * represent a matrix column.
39 */
40 T m[16];
41
42 public:
43 /**
44 * @brief Default constructor. Initializes all components to 0
45 */
46 explicit constexpr mat4() noexcept : m{} {};
47
48 /**
49 * @brief Creates a diagonal matrix. The diagonal components are initialized
50 * with the value f.
51 *
52 * @param f The scalar value for the diagonal components.
53 */
54 explicit constexpr mat4(const T f) noexcept
55 : m{ f, T{}, T{}, T{},
56 T{}, f, T{}, T{},
57 T{}, T{}, f, T{},
58 T{}, T{}, T{}, f} {}

60 /**
61 * @brief Creates a diagonal matrix with the components of vec3<T> as the
62 * diagonal elements. Element at [4, 4] is set to 1
63 *
64 * @param f The scalar value for the diagonal components.
65 */
66 explicit constexpr mat4(const vec3<T> f) noexcept
67 : m{ f[0], T{}, T{}, T{},
68 T{}, f[1], T{}, T{},
69 T{}, T{}, f[2], T{},
70 T{}, T{}, T{}, static_cast<T>(1)} {}

105 /**
106 * @brief Provides read-only access to a matrix component.
107 *
108 * Elements are _accessed_ in column major order: `m[row + col * 4]`,
109 * while the specified indices represent an usual mxn-matrix access,
110 * i.e. for a given 2x4-matrix
111 * [0, 1, 2, 3,
112 * 4, 5, 6, 7,
113 * 8, 9, 10, 11,
114 * 12, 13, 14, 15]
115 *
116 * a call to row(0, 2) returns "2", while the matrix is internally stored
117 * as (1, 4, 8, 12, 1, 5, 9, 13, 2, 6, 10, 14, 3, 7, 11, 15).
118 *
119 * Bounds checking is performed via `assert` in debug builds.
120 *
121 * @param row The zero based row index.
122 * @param col The zero based col index.
123 *
124 * @return A const ref to the component at the specified position.
125 */
126 constexpr const T& operator()(const size_t row, const size_t col) const {
127 assert(row < 4 && col < 4 && "mat4 - Index out of bounds.");
128 return m[row + col * 4];
129 }

133 * @brief Provides read-write access to a matrix component.
134 *
135 * Elements are _accessed_ in column major order: `m[row + col * 4]`,
136 * while the specified indices represent an usual mxn-matrix access,
137 * i.e. for a given 2x4-matrix
138 * [0, 1, 2, 3,
139 * 4, 5, 6, 7,
140 * 8, 9, 10, 11,
141 * 12, 13, 14, 15]

143 * a call to row(0, 2) returns "2", while the matrix is internally stored

159 /**
160 * @brief Compares this matrix element's with the rgt matrix considering
161 * an epsilon value.
162 * Returns true if for all elements the equation |a-b| <= EPSILON
163 * holds.
164 *
165 * EPSILON is set to 0.00001
166 *
167 * @param rgt The other matix to compare with this matrix for equality.
168 *
169 * @return True is the elements of the matrices are considered equal,
170 * otherwise false.
171 *
172 * @see https://realtimecollisiondetection.net/blog/?p=89
173 *
174 * @todo account for abs (values close to zero) and rel
175 * (larger values), move EPSILON to global constant
176 */
177 constexpr bool same(const mat4<T>& rgt) const {
178
179 static const T EPSILON = static_cast<T>(0.00001);
180
181 const auto* leftPtr = value_ptr(*this);
182 const auto* rgtPtr = value_ptr(rgt);
183
184 for (int i = 0; i < 16; i++) {
185 if (std::fabs(leftPtr[i] - rgtPtr[i]) > EPSILON) {
186 return false;
187 }
188 }
189
190 return true;
191 }
192

194 /**
195 * @brief Strictly compares the elements of this matrix with the elements
196 * of the rgt matrix.
197 *
198 * @param rgt The right matrix to compare for equal values
199 *
200 * @return True if all elements are equal (==), false otherwise.
201 */
202 constexpr bool operator==(const mat4<T>& rgt) const {
203
204 const auto* leftPtr = value_ptr(*this);
205 const auto* rgtPtr = value_ptr(rgt);
206
207 for (int i = 0; i < 16; i++) {
208 if (leftPtr[i] != rgtPtr[i]) {
209 return false;
210 }
211 }
212
213 return true;
214 }

217 /**
218 * @brief Performs matrix-multiplication with another `mat4`.
219 * This matrix is the left operand, while `m` is the right operand.
220 *
221 * @param m The right-hand side `mat4<T>` for multiplication.
222 *
223 * @return A new `mat4<T>`, representing the result of the matrix-multiplication.
224 */
225 constexpr mat4<T> operator*(const mat4<T>& m) const {
226 mat4<T> A{};
227 for (int row = 0; row < 4; row++) {
228 for (int col = 0; col < 4; col++) {
229 T sum = T{};
230 sum += (*this)(row, 0) * m(0, col);
231 sum += (*this)(row, 1) * m(1, col);
232 sum += (*this)(row, 2) * m(2, col);
233 sum += (*this)(row, 3) * m(3, col);
234 A(row, col) = sum;
235 }
236 }
237
238 return A;
239 }

242 /**
243 * @brief Performs matrix-vector-multiplication with a `vec4<T> v`.
244 *
245 * This matrix is the left operand, while `v` is the right operand.
246 *
247 * @param v The right-hand side `vec4<T>` for multiplication.
248 *
249 * @return A new `vec4<T>`, representing the result of the matrix-vector-multiplication.
250 */
251 constexpr vec4<T> operator*(const vec4<T>& v) const {
252 const auto m = *this;
253 return vec4<T>{
254 m(0, 0) * v[0] + m(0, 1) * v[1] + m(0, 2) * v[2] + m(0, 3) * v[3],
255 m(1, 0) * v[0] + m(1, 1) * v[1] + m(1, 2) * v[2] + m(1, 3) * v[3],
256 m(2, 0) * v[0] + m(2, 1) * v[1] + m(2, 2) * v[2] + m(2, 3) * v[3],
257 m(3, 0) * v[0] + m(3, 1) * v[1] + m(3, 2) * v[2] + m(3, 3) * v[3]
258 };
259 }
