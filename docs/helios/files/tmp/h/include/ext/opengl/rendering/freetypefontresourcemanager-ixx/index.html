30 /**
31 * @brief FreeType-based implementation of `FontResourceProvider` for OpenGL.
32 *
33 * `FreeTypeFontResourceManager` uses the FreeType library to load TrueType fonts,
34 * rasterize glyphs, and cache them as OpenGL textures. Each glyph is stored as a
35 * separate texture for simplicity.
36 *
37 * ## Responsibilities
38 *
39 * - **Font Loading:** Parse font files using FreeType and extract glyph metrics.
40 * - **Texture Creation:** Upload glyph bitmaps to GPU as GL_R8 textures.
41 * - **Caching:** Store glyph data per font for efficient repeated access.
42 * - **Resource Cleanup:** Delete OpenGL textures when the manager is destroyed.
43 *
44 * ## Usage
45 *
46 * ```cpp
47 * FreeTypeFontResourceManager fontManager;
48 * fontManager.loadFont(FontId{"roboto"}, "fonts/Roboto-Regular.ttf");
49 *
50 * // Retrieve a glyph for rendering
51 * Glyph g = fontManager.glyph('A', FontId{"roboto"});
52 * ```
53 *
54 * @note Currently loads ASCII characters 0-127 with a fixed pixel size of 24
55 *
56 * @see FontResourceProvider
57 * @see Glyph
58 * @see OpenGLGlyphTextRenderer
59 */
60 class FreeTypeFontResourceManager : public helios::rendering::text::FontResourceProvider {

82 /**
83 * @brief Destructor that cleans up all OpenGL textures.
84 *
85 * Iterates through all cached fonts and deletes the associated glyph textures
86 * to free GPU memory.
87 */
88 ~FreeTypeFontResourceManager() override {
89
90 // Cleanup texture memory for all loaded fonts
91 for (const auto& cache: fontCache_ | std::views::values) {
92 for (const auto& glyph : cache.characters) {
93 // glDeleteTextures ignores 0, so this is safe even if textureId is uninitialized (0)
94 glDeleteTextures(1, &glyph.textureId);
95 }
96 }
97 }

99 /**
100 * @brief Loads a font from a file and caches its glyphs as OpenGL textures.
101 *
102 * Uses FreeType to parse the font file and rasterize ASCII characters 0-127.
103 * Each glyph bitmap is uploaded to the GPU as a GL_R8 texture.
104 *
105 * @param fontId Unique identifier for this font.
106 * @param pathToFont File system path to the font file (e.g., `.ttf`).
107 *
108 * @throws std::runtime_error If FreeType initialization fails.
109 * @throws std::runtime_error If the font file cannot be loaded.
110 * @throws std::runtime_error If a glyph cannot be loaded.
111 *
112 * @pre The font ID must not already be loaded (asserts in debug builds).
113 */
114 void loadFont(
115 helios::engine::core::data::FontId fontId,
116 std::string_view pathToFont
117 ) override {
118
119 const int begin = 0;
120 const int end = 128;
121
122 assert(!fontCache_.contains(fontId) && "FontCache already existing");
123
124 fontCache_.emplace(fontId, FontCache{});
125
126 FT_Library ft{};
127
128 FT_Face face{};
129
130 if (FT_Init_FreeType(&ft)) {
131 throw std::runtime_error("ERROR::FREETYPE: Could not init FreeType Library");
132
133 }
134
135 if (FT_New_Face(ft, pathToFont.data(), 0, &face)) {
136 throw std::runtime_error("ERROR::FREETYPE: Failed to load font");
137 }
138
139 FT_Set_Pixel_Sizes(face, 0, 24);
140 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
141 glPixelStorei(GL_UNPACK_ROW_LENGTH, face->glyph->bitmap.pitch);
142
143 for (unsigned short c = begin; c < end; ++c) {
144
145 if (FT_Load_Char(face, c, FT_LOAD_RENDER)) {
146 throw std::runtime_error("ERROR::FREETYTPE: Failed to load Glyph");
147 }
148
149 unsigned int texture;
150 glGenTextures(1, &texture);
151 glBindTexture(GL_TEXTURE_2D, texture);
152 glTexImage2D(
153 GL_TEXTURE_2D,
154 0,
155 GL_R8,
156 face->glyph->bitmap.width,
157 face->glyph->bitmap.rows,
158 0,
159 GL_RED,
160 GL_UNSIGNED_BYTE,
161 face->glyph->bitmap.buffer
162 );
163 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
164 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
165 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
166 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
167
168 fontCache_[fontId].characters[c - begin] = {
169 texture,
170 {face->glyph->bitmap.width, face->glyph->bitmap.rows},
171 {face->glyph->bitmap_left, face->glyph->bitmap_top},
172 face->glyph->advance.x
173 };
174
175 }
176
177 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
178 FT_Done_Face(face);
179 FT_Done_FreeType(ft);
180 };
181
182 /**
183 * @brief Retrieves the cached glyph data for a character.
184 *
185 * Returns the glyph metrics and texture ID for the specified character
186 * from the font's cache. The font must have been previously loaded via `loadFont()`.
187 *
188 * @param c The character to retrieve the glyph for.
189 * @param fontId The font identifier to use.
190 *
191 * @return The cached `Glyph` data for the character.
192 *
193 * @pre The font must already be loaded (asserts in debug builds).
194 */
195 helios::rendering::text::Glyph glyph(
196 const char c,
197 const helios::engine::core::data::FontId fontId
198 ) override {
199
200 assert(fontCache_.contains(fontId) && "Font cache entry not found for given FontId (font not loaded?)");
201
202 return fontCache_[fontId].characters[c];
203 }
