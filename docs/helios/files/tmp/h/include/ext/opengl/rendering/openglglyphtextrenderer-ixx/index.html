35 /**
36 * @brief Renders and manages text using OpenGL and FreeType.
37 *
38 * This class is responsible for rendering text using cached glyph textures and managing
39 * OpenGL resources for efficient rendering. It leverages FreeType to rasterize font glyphs
40 * and OpenGL to render them onto the screen. Text rendering is performed by positioning
41 * quads (one per character) with texture-mapped glyphs.
42 *
43 * The class provides methods for initializing OpenGL resources, loading fonts into a cache,
44 * and rendering text at specified positions and scales.
45 *
46 * Inherits from the TextRenderer class in the helios rendering system.
47 */
48 class OpenGLGlyphTextRenderer : public helios::rendering::text::TextRenderer {

69 /**
70 * @brief Initializes OpenGL resources (VAO and VBO) for text rendering.
71 *
72 * Creates a vertex array object and vertex buffer for rendering glyph quads.
73 * The VBO is configured for dynamic updates (one quad per character).
74 *
75 * Must be called before any rendering operations.
76 */
77 void init() {
78 glGenVertexArrays(1, &vao_);
79 glGenBuffers(1, &vbo_);
80 glBindVertexArray(vao_);
81 glBindBuffer(GL_ARRAY_BUFFER, vbo_);
82 // 2D quad for a texture requires 6 vertices with 4 floats each => 6*4
83 glBufferData(GL_ARRAY_BUFFER, sizeof(float) * 6*4, nullptr, GL_DYNAMIC_DRAW);
84 glEnableVertexAttribArray(0);
85 glVertexAttribPointer(0, 4, GL_FLOAT, GL_FALSE, 4 * sizeof(float), 0);
86 glBindBuffer(GL_ARRAY_BUFFER, 0);
87 glBindVertexArray(0);
88 }

91 /**
92 * @brief Loads a font and caches glyph textures for the specified character range.
93 *
94 * Uses FreeType to load the font file and render each glyph in the specified
95 * range to an OpenGL texture.
96 *
97 * @param fontId Unique identifier for this font.
98 * @param pathToFont File system path to the font file.
99 * @param begin Start of character range (inclusive).
100 * @param end End of character range (exclusive).
101 *
102 * @throws std::runtime_error If FreeType initialization or font loading fails.
103 */
104 void loadMap(
105 helios::engine::core::data::FontId fontId,
106 std::string_view pathToFont,
107 const unsigned short begin, const unsigned short end) {
108
109 assert(!fontCache_.contains(fontId) && "FontCache already existing");
110
111 fontCache_.emplace(fontId, FontCache{});
112
113 FT_Library ft{};
114
115 FT_Face face{};
116
117 if (FT_Init_FreeType(&ft)) {
118 throw std::runtime_error("ERROR::FREETYPE: Could not init FreeType Library");
119
120 }
121
122 if (FT_New_Face(ft, pathToFont.data(), 0, &face)) {
123 throw std::runtime_error("ERROR::FREETYPE: Failed to load font");
124 }
125
126 FT_Set_Pixel_Sizes(face, 0, 48);
127 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
128 glPixelStorei(GL_UNPACK_ROW_LENGTH, face->glyph->bitmap.pitch);
129
130 for (unsigned short c = begin; c < end; ++c) {
131
132 if (FT_Load_Char(face, c, FT_LOAD_RENDER)) {
133 throw std::runtime_error("ERROR::FREETYTPE: Failed to load Glyph");
134 }
135
136 unsigned int texture;
137 glGenTextures(1, &texture);
138 glBindTexture(GL_TEXTURE_2D, texture);
139 glTexImage2D(
140 GL_TEXTURE_2D,
141 0,
142 GL_R8,
143 face->glyph->bitmap.width,
144 face->glyph->bitmap.rows,
145 0,
146 GL_RED,
147 GL_UNSIGNED_BYTE,
148 face->glyph->bitmap.buffer
149 );
150 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
151 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
152 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
153 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
154
155 fontCache_[fontId].characters[c - begin] = {
156 texture,
157 {face->glyph->bitmap.width, face->glyph->bitmap.rows},
158 {face->glyph->bitmap_left, face->glyph->bitmap_top},
159 face->glyph->advance.x
160 };
161
162 }
163
164 glPixelStorei(GL_UNPACK_ROW_LENGTH, 0);
165 FT_Done_Face(face);
166 FT_Done_FreeType(ft);
167 }

189 /**
190 * @brief Renders text at a specified position using a specified font and scale.
191 *
192 * This method uses OpenGL to render text, character by character, by updating vertex buffer
193 * objects (VBOs) and binding glyph textures. Each glyph is positioned relative to the given
194 * starting position, scaled according to the provided scale factor, and rendered using stored
195 * glyph information from a font cache.
196 *
197 * @param text The text string to render.
198 * @param position The starting (x, y) position for the text, in screen space coordinates.
199 * @param scale The scale factor to apply to the size of each glyph.
200 * @param fontId The font identifier used to select the font from the font cache.
201 */
202 void renderText(
203 const std::string_view text,
204 helios::math::vec2f position,
205 const float scale,
206 const helios::engine::core::data::FontId fontId
207 ) noexcept {
208
209 glActiveTexture(GL_TEXTURE0);
210 glBindVertexArray(vao_);
211 for (const char c : text) {
212 assert(begin_ < 128 && "Glyph range invalid, possible underflow");
213 helios::rendering::text::Glyph ch = fontCache_[fontId].characters[c - begin_];
214 const float xpos = position[0] + ch.bearing[0] * scale;
215 const float ypos = position[1] - ( ch.size[1] - ch.bearing[1]) * scale;
216
217 const float w = ch.size[0] * scale;
218 const float h = ch.size[1] * scale;
219
220 // update the VBO for _each_ character (multiple drawcalls!)
221 const float vertices[6][4] = {
222 {xpos, ypos +h, 0.0f, 0.0f},
223 {xpos, ypos, 0.0f, 1.0f},
224 {xpos + w, ypos, 1.0f, 1.0f},
225
226 {xpos, ypos +h, 0.0f, 0.0f},
227 {xpos + w, ypos, 1.0f, 1.0f},
228 {xpos+w, ypos +h, 1.0f, 0.0f}
229 };
230
231 // render glyph texture over quad
232 glBindTexture(GL_TEXTURE_2D, ch.textureId);
233 // update content of vbo memory
234 glBindBuffer(GL_ARRAY_BUFFER, vbo_);
235 glBufferSubData(GL_ARRAY_BUFFER, 0, sizeof(vertices), vertices);
236 glBindBuffer(GL_ARRAY_BUFFER, 0);
237 //render quad
238 glDrawArrays(GL_TRIANGLES, 0, 6);
239 // advance cursors for next glyph (1/64 pixels)
240 position[0] += (ch.advance >> 6) * scale;
241
242 }
243 glBindVertexArray(0);
244 glBindTexture(GL_TEXTURE_2D, 0);
245 }

250 /**
251 * @brief Destroys the OpenGLGlyphTextRenderer and releases all OpenGL resources.
252 *
253 * Cleans up the Vertex Array Object (VAO), Vertex Buffer Object (VBO),
254 * and all cached glyph textures for every loaded font.
255 *
256 * @note Must be called in a valid OpenGL context.
257 */
258 ~OpenGLGlyphTextRenderer() override {
259
260 if (vao_ != 0) {
261 glDeleteVertexArrays(1, &vao_);
262 }
263 if (vbo_ != 0) {
264 glDeleteBuffers(1, &vbo_);
265 }
266
267 // Cleanup texture memory for all loaded fonts
268 for (const auto& cache: fontCache_ | std::views::values) {
269 for (const auto& glyph : cache.characters) {
270 // glDeleteTextures ignores 0, so this is safe even if textureId is uninitialized (0)
271 glDeleteTextures(1, &glyph.textureId);
272 }
273 }
274 }

285 /**
286 * @brief Registers a font family for text rendering.
287 *
288 * Loads the font from the specified path and pre-renders glyphs for all
289 * characters in the given range. The glyphs are cached as OpenGL textures.
290 *
291 * @param fontId Unique identifier for this font family.
292 * @param path File system path to the font file (TTF, OTF).
293 * @param begin Start of character range to load (inclusive, default: 0).
294 * @param end End of character range to load (exclusive, default: 128).
295 *
296 * @throws std::invalid_argument If `begin > end`.
297 * @throws std::runtime_error If the font cannot be loaded.
298 */
299 void addFontFamily(
300 const helios::engine::core::data::FontId fontId, std::string_view path,
301 const unsigned short begin = 0, const unsigned short end = 128) override {
302
303 if (begin > end) {
304 throw std::invalid_argument("Glyph range is invalid");
305 }
306
307 loadMap(fontId, path, begin, end);
308 }

310 /**
311 * @brief Renders text using the specified render command.
312 *
313 * Processes the `TextRenderCommand`, activates the associated shader,
314 * applies uniform values, and renders the text string.
315 *
316 * @param command The render command containing text and rendering properties.
317 * @param frameUniformValues Frame-level uniforms (e.g., projection matrix).
318 *
319 * @note Requires a valid OpenGL context and initialized renderer.
320 */
321 void render(
322 const helios::rendering::text::TextRenderCommand& command,
323 const helios::rendering::shader::UniformValueMap& frameUniformValues) noexcept override {
324
325 if (auto* textPrototype = command.textRenderPrototype()) {
326
327 const auto& baseShader = textPrototype->shader();
328
329 const auto* shader = static_cast<const helios::ext::opengl::rendering::shader::OpenGLShader*>(&baseShader);
330 assert(shader && "Unexpected failure when casting to OpenGLShader.");
331
332 shader->use();
333
334 shader->applyUniformValues(frameUniformValues);
335 shader->applyUniformValues(command.textUniformValues());;
336
337 const auto& drawProperties = command.drawProperties();
338 const std::string_view text = command.text();
339 const auto screenPosition = drawProperties.position;
340 const auto scale = drawProperties.scale;
341 const auto fontId = drawProperties.fontId;
342
343 renderText(text, screenPosition, scale, fontId);
344 }
