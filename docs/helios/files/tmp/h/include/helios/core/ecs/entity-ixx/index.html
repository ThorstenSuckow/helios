31 /**
32 * @brief Lightweight facade for entity component manipulation.
33 *
34 * @details `Entity` provides a type-safe, convenient interface for
35 * working with entities and their components. It wraps an `EntityHandle`
36 * and a pointer to the `EntityManager`, delegating all operations.
37 *
38 * ## Size and Copy Semantics
39 *
40 * Entity is only 16 bytes (8 bytes EntityHandle + 8 bytes pointer) and
41 * should be **passed by value**, not by reference:
42 *
43 * ```cpp
44 * // Correct - pass by value
45 * void processEntity(Entity entity) { ... }
46 *
47 * // Unnecessary - reference adds indirection
48 * void processEntity(Entity& entity); // Avoid
49 * void processEntity(const Entity& entity); // Avoid
50 * ```
51 *
52 * ## Hierarchy Integration
53 *
54 * When a Entity has a `HierarchyComponent`, activation changes trigger
55 * hierarchy propagation. Calling `setActive()` marks the hierarchy as dirty,
56 * and `HierarchyPropagationSystem` propagates the state to all descendants.
57 *
58 * ## Usage
59 *
60 * ```cpp
61 * auto player = gameWorld.addEntity();
62 *
63 * // Add components
64 * player.add<TransformComponent>(position);
65 * player.add<HealthComponent>(100.0f);
66 *
67 * // Query and access
68 * if (player.has<HealthComponent>()) {
69 * player.get<HealthComponent>()->takeDamage(10.0f);
70 * }
71 *
72 * // Activation state (propagates to children if HierarchyComponent present)
73 * player.setActive(false); // Calls onDeactivate() on components
74 * ```
75 *
76 * @see EntityHandle
77 * @see EntityManager
78 * @see GameWorld
79 * @see HierarchyComponent
80 * @see HierarchyPropagationSystem
81 */
82 template<typename THandle, typename TEntityManager>
83 class Entity {

144 /**
145 * @brief Constructs and attaches a component to this entity.
146 *
147 * @details If the entity is active, calls `onActivate()` on the new component.
148 * If inactive, calls `onDeactivate()`.
149 *
150 * @tparam T The component type to add.
151 * @tparam Args Constructor argument types.
152 *
153 * @param args Arguments forwarded to the component constructor.
154 *
155 * @return Reference to the newly created component.
156 */
157 template<typename T, typename ...Args>
158 T& add(Args&& ...args) {
159
160 bool active = true;
161
162 if (entityManager_->has<components::Active>(entityHandle_)) {
163 active = true;
164 } else {
165 active = false;
166 }
167
168 auto* cmp = entityManager_->emplace<T>(entityHandle_, std::forward<Args>(args)...);
169
170 auto typeId = ComponentTypeId_type::template id<T>();
171 const auto ops = ComponentOpsRegistry_type::ops(typeId);
172 void* raw = entityManager_->raw(entityHandle_, typeId);
173
174 if (active && ops.onActivate) {
175 ops.onActivate(raw);
176 } else if (!active && ops.onDeactivate) {
177 ops.onDeactivate(raw);
178 }
179
180 return *cmp;
181 }
182
183 /**
184 * @brief Returns existing component or creates a new one.
185 *
186 * @tparam T The component type.
187 * @tparam Args Constructor argument types.
188 *
189 * @param args Arguments forwarded to the constructor if creating.
190 *
191 * @return Reference to the existing or newly created component.
192 */
193 template<typename T, typename ...Args>
194 T& getOrAdd(Args&& ...args) {
195 if (entityManager_->has<T>(entityHandle_)) {
196 return *entityManager_->get<T>(entityHandle_);

199 }

289 /**
290 * @brief Sets the activation state of this Entity.
291 *
292 * @details When deactivated:
293 * - An `Inactive` tag component is added
294 * - The `Active` tag component is removed
295 * - `onDeactivate()` is called on components that support it
296 * - If a `HierarchyComponent` is present, it is marked dirty for propagation
297 *
298 * When activated:
299 * - The `Inactive` tag component is removed
300 * - An `Active` tag component is added
301 * - `onActivate()` is called on components that support it
302 * - If a `HierarchyComponent` is present, it is marked dirty for propagation
303 *
304 * @note Does **not** call `enable()`/`disable()` on components.
305 *
306 * @param active True to activate, false to deactivate.
307 *
308 * @see HierarchyComponent_type
309 * @see HierarchyPropagationSystem
310 */
311 void setActive(const bool active) {
312
313 bool isActive = entityManager_->has<components::Active>(entityHandle_);
314 bool isInActive = entityManager_->has<components::Inactive>(entityHandle_);
315
316 if (!isActive && active) {
317 auto* hc = entityManager_->get<HierarchyComponent_type>(entityHandle_);
318 if (hc) {
319 hc->markDirty();
320 }
321
322 entityManager_->remove<components::Inactive>(entityHandle_);
323 entityManager_->emplaceOrGet<components::Active>(entityHandle_);
324 }
325
326 if (!isInActive && !active) {
327 auto* hc = entityManager_->get<HierarchyComponent_type>(entityHandle_);
328 if (hc) {
329 hc->markDirty();
330 }
331
332 entityManager_->emplaceOrGet<components::Inactive>(entityHandle_);
333 entityManager_->remove<components::Active>(entityHandle_);
334 }
335
336 for (auto typeId : componentTypeIds()) {
337 const auto ops = ComponentOpsRegistry_type::ops(typeId);
338 void* raw = entityManager_->raw(entityHandle_, typeId);
339
340 if (active && ops.onActivate) {
341 ops.onActivate(raw);
342 } else if (!active && ops.onDeactivate) {
343 ops.onDeactivate(raw);
344 }
345 }
346 }

388 }
389 }
