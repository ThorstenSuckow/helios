Description​

Fluent configuration for a complete spawn pool with profiles and scheduling.

This file contains SpawnSystemConfigurator, SpawnRuleConfig, SpawnProfileConfig, and SpawnPoolConfig in a single translation unit. These classes form a cyclic done()-chaining dependency (Rule → Profile → Pool → Configurator → Pool) so we will not split them in separate files.

412 std::unique_ptr<helios::engine::runtime::spawn::policy::SpawnRule>>> result;
413 for (auto& rule : rules_) {
414 result.emplace_back(profileId_, rule->build());
415 }
416 return result;
417 }

419 /**
420 * @brief Finalizes the profile and returns to the parent pool config.
421 *
422 * @return Reference to the parent SpawnPoolConfig.
423 */
424 SpawnPoolConfig& done();

428 * @brief ID-centric configuration for a spawn pool with associated profiles.
429 *
430 * @details Bundles a pool ID with its PrefabId, pool size, and one or more
431 * spawn profiles. Each profile can optionally have scheduled rules.
432 * Calling commit() registers the pool, all profiles, and any
433 * scheduled rules with a DefaultSpawnScheduler.
434 *
435 * @see SpawnProfileConfig
436 * @see SpawnRuleConfig
437 * @see SpawnSystemFactory
438 */
439 class SpawnPoolConfig {
440
441 /**
442 * @brief The pool manager to register with.
443 */
444 helios::engine::runtime::pooling::GameObjectPoolManager& poolManager_;
445
446 /**
447 * @brief The spawn manager to register profiles with.
448 */
449 helios::engine::runtime::spawn::SpawnManager& spawnManager_;

453 */
454 helios::engine::core::data::GameObjectPoolId poolId_;
455
456 /**

458 */
459 helios::engine::core::data::PrefabId prefabId_;
460
461 /**
462 * @brief Number of instances to pre-allocate.
463 */

465
466 /**
467 * @brief Profile configurations attached to this pool.
468 */
469 std::vector<std::unique_ptr<SpawnProfileConfig>> profiles_;

474
475 /**
476 * @brief Constructs a SpawnPoolConfig.
477 *
478 * @param poolManager The pool manager to register pools with.
479 * @param spawnManager The spawn manager to register profiles with.
480 * @param poolId Unique identifier for the pool.
481 * @param prefabId Identifier of the prefab template for cloning.
482 * @param poolSize Number of instances to pre-allocate.
483 */
484 SpawnPoolConfig(
485 helios::engine::runtime::pooling::GameObjectPoolManager& poolManager,

501 SpawnProfileConfig& profile(helios::engine::core::data::SpawnProfileId profileId) {

503 *this, spawnManager_, profileId, poolId_
504 ));
505 return *profiles_.back();

508 /**
509 * @brief Commits pool, all profiles, and registers scheduled rules.
510 *
511 * Profiles with scheduled rules are automatically registered
512 * with a DefaultSpawnScheduler per profile.
513 *
514 * @return Configurator for chaining the next pool() call.
515 */
516 SpawnSystemConfigurator commit() {
517 commitPool();
518 commitProfiles(false);
519 return SpawnSystemConfigurator{poolManager_, spawnManager_};
520 }
521
522 /**
523 * @brief Commits pool and profiles without creating any schedulers.
524 *
525 * @details Use this when profiles should be registered but scheduling
526 * is handled separately (e.g., via SchedulerBuilder). Rules attached
527 * via scheduledBy() are silently discarded.
528 *
529 * This enables reuse of the same profile configuration under different
530 * scheduling strategies:
531 *
532 * ```cpp
533 * // Register pool + profiles (no schedulers)
534 * spawns.pool(PoolId, PrefabId, 100)
535 * .profile(LeftId).axisPlacement(...).moveInitializer(...).done()
536 * .profile(RightId).axisPlacement(...).moveInitializer(...).done()
537 * .commitProfilesOnly();
538 *
539 * // Attach scheduling separately
540 * SchedulerBuilder sb(spawnManager);
541 * sb.cyclicScheduler(
542 * SchedulerConfig(LeftId, Rule1).timerCondition(5.0f).fixedAmount(10),
543 * SchedulerConfig(RightId, Rule2).timerCondition(5.0f).fixedAmount(10)
544 * );
545 * ```
546 *
547 * @return Configurator for chaining the next pool() call.
548 */
549 SpawnSystemConfigurator commitProfilesOnly() {
550 commitPool();
551 commitProfiles(true);
552 return SpawnSystemConfigurator{poolManager_, spawnManager_};
553 }
554
555 /**
556 * @brief Commits with all scheduled rules in a single CyclicSpawnScheduler.
557 *
558 * @details Use this when multiple profiles share a cyclic spawn
559 * pattern (e.g., wave-based edge spawning). All rules across all
560 * profiles are bundled into one CyclicSpawnScheduler.
561 *
562 * @tparam N Number of rules in the cyclic scheduler.
563 *
564 * @return Configurator for chaining the next pool() call.
565 */
566 template<std::size_t N>
567 SpawnSystemConfigurator commitCyclic() {
568 commitPool();
569
570 auto scheduler = std::make_unique<
571 helios::engine::runtime::spawn::scheduling::CyclicSpawnScheduler<N>>();
572
573 for (auto& profileConfig : profiles_) {
574 auto rules = profileConfig->commit();
575 for (auto& [profileId, rule] : rules) {
576 scheduler->addRule(profileId, std::move(rule));
577 }
578 }
579
580 spawnManager_.addScheduler(std::move(scheduler));
581 return SpawnSystemConfigurator{poolManager_, spawnManager_};
582 }
583
584 private:
585
586 /**
587 * @brief Registers the pool with the pool manager.
588 */
589 void commitPool() {
590
591 poolManager_.addPoolConfig(
592 std::make_unique<helios::engine::runtime::pooling::GameObjectPoolConfig>(
593 poolId_, prefabId_, size_
594 )
595 );
596 }
597
598 /**
599 * @brief Commits all profiles and optionally creates per-profile schedulers.
600 *
601 * @param skipSchedulers If true, profiles are committed without scheduler creation.
602 */
603 void commitProfiles(bool skipSchedulers) {
604 for (auto& profileConfig : profiles_) {
605 auto rules = profileConfig->commit();
606 if (!skipSchedulers && !rules.empty()) {
607 auto scheduler = std::make_unique<
608 helios::engine::runtime::spawn::scheduling::DefaultSpawnScheduler>();
609 for (auto& [profileId, rule] : rules) {
610 scheduler->addRule(profileId, std::move(rule));
611 }
612 spawnManager_.addScheduler(std::move(scheduler));
613 }
614 }
615 }
