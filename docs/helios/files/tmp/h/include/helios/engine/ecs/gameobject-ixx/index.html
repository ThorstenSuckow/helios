25 /**
26 * @brief Base class for interactive game entities with component-based composition.
27 *
28 * @details GameObject serves as the fundamental building block for all interactive entities
29 * in the helios engine. It provides a type-indexed component storage system that enables
30 * O(1) component access and modification.
31 *
32 * **Component Storage Architecture:**
33 * Components are stored in a vector indexed by `ComponentTypeId`. Each component type
34 * receives a unique compile-time ID, which determines its position in the storage vector.
35 * This approach trades memory (sparse vector with nullptr slots) for speed (direct indexing
36 * without hashing).
37 *
38 * **Lifecycle Management:**
39 * The GameObject manages component lifecycles through callbacks:
40 * - `onAttach()` - Called when a component is added.
41 * - `onActivate()` / `onDeactivate()` - Called when the GameObject's active state changes.
42 * - `onAcquire()` / `onRelease()` - Called during pool-based lifecycle management.
43 *
44 * **Storage Model:**
45 * Components are stored in a `std::vector<std::unique_ptr<Component>>` where the index
46 * corresponds to the component's `ComponentTypeId::value()`. The vector is resized on demand
47 * when new component types are added. Slots for component types that are not attached to
48 * this GameObject remain nullptr.
49 *
50 * **Performance Characteristics:**
51 * - `get<T>()`: O(1) - Direct array access via type ID.
52 * - `has<T>()`: O(1) - Bounds check + nullptr check.
53 * - `add<T>()`: O(1) amortized - May trigger resize if type ID exceeds current capacity.
54 * - `components()`: O(n) on first call after modification (cache rebuild), O(1) thereafter.
55 *
56 * @see Component
57 * @see GameWorld
58 * @see GameObjectPool
59 */

130 *
131 * @details Iterates through `components_` and populates `componentView_` and
132 * `componentConstView_` with non-null entries. This operation is O(n) where
133 * n is the size of the components vector.

135 void updateCache() const {
136
137 if (!needsUpdate_) {
138 return;
139 }
140 componentView_.clear();
141 componentConstView_.clear();
142 componentView_.reserve(components_.size());
143 componentConstView_.reserve(components_.size());
144
145 for (const auto& component : components_) {
146
147 if (component == nullptr) {
148 continue;
149 }
150 componentView_.push_back(component.get());
151 componentConstView_.push_back(component.get());
152 }
153
154 needsUpdate_ = false;
155 }

179 /**
180 * @brief Adds a new component of type T to the GameObject.
181 *
182 * @details The component is created in-place, stored at the index corresponding to
183 * its `ComponentTypeId`, and attached to this GameObject (`onAttach()` is called).
184 * If the type ID exceeds the current vector capacity, the vector is resized.
185 *
186 * **Complexity:** O(1) amortized. Resize operations are O(n) but occur infrequently.
187 *
188 * @tparam T The type of component to add. Must derive from Component.
189 * @tparam Args Argument types for the component's constructor.
190 *
191 * @param args Arguments forwarded to the component's constructor.
192 *
193 * @return Reference to the newly created component.
194 *
195 * @pre No component of type T is already attached (asserts in debug mode).
196 *
197 * @note The component is stored directly at `components_[ComponentTypeId::id<T>().value()]`.
198 */
199 template<typename T, typename... Args>
200 T& add(Args&&... args) {
201 using U = std::remove_cvref_t<T>;
202
203 assert(!has<U>() && "Cannot add, Component is already existing.");
204
205 auto component_ptr = std::make_unique<U>(std::forward<Args>(args)...);
206 U* raw_component_ptr = component_ptr.get();
207
208 const auto componentTypeId = ComponentTypeId::id<U>();
209 if (componentTypeId.value() >= components_.size()) {
210 components_.resize(componentTypeId.value() + 1);
211 }
212
213 components_[componentTypeId.value()] = std::move(component_ptr);
214
215 // make sure component is registered before onAttach is called,
216 // in case onAttach implementations query the gameObject for **this** component
217 // Use raw_component_ptr instead of component_ptr since ownership was moved above
218 finalizeAttach(raw_component_ptr);
219 invalidateCache();
220
221 return *raw_component_ptr;
222 }

224 /**
225 * @brief Adds a pre-constructed component if no component of the same type exists.
226 *
227 * @details If a component with the same `ComponentTypeId` already exists, returns
228 * the existing component without taking ownership of the passed pointer.
229 * Otherwise, stores the component at its type-indexed position and takes ownership.
230 *
231 * **Complexity:** O(1) amortized. May trigger resize if type ID exceeds capacity.
232 *
233 * @param component Unique pointer to the component to add. Ownership is transferred
234 * only if no component of the same type exists.
235 *
236 * @return Pointer to the existing or newly added component, or nullptr if
237 * the input was nullptr.
238 *
239 * @note Uses `component->typeId()` for runtime type identification.
240 */
241 helios::engine::ecs::Component* getOrAdd(
242 std::unique_ptr<helios::engine::ecs::Component> component) {
243
244 assert(component != nullptr && "Unexpected nullptr for component.");
245
246 if (!component) {
247 return nullptr;
248 }
249
250 const auto componentTypeId = component.get()->typeId();
251
252 if (has(componentTypeId)) {
253 return components_[componentTypeId.value()].get();
254 }
255
256 if (componentTypeId.value() >= components_.size()) {
257 components_.resize(componentTypeId.value() + 1);
258 }
259
260 auto* raw_component_ptr = component.get();
261 components_[componentTypeId.value()] = std::move(component);
262
263 // make sure component is registered before onAttach is called,
264 // in case onAttach implementations query the gameObject for **this** component
265 // Use raw_component_ptr instead of component_ptr since ownership was moved above
266 finalizeAttach(raw_component_ptr);
267 invalidateCache();
268 return raw_component_ptr;
269 }

273 * @brief Retrieves an existing component of type T, or creates and adds it if not present.

298 /**
299 * @brief Retrieves a component of type T.
300 *
301 * @details Performs a direct array access using the component's type ID.
302 * If the type ID is out of bounds or the slot is nullptr, returns nullptr.
303 *
304 * **Complexity:** O(1) - Direct index access without hashing.
305 *
306 * @tparam T The type of component to retrieve. cv-qualifiers are stripped.
307 *
308 * @return Pointer to the component if found, nullptr otherwise.
309 */
310 template<typename T>
311 [[nodiscard]] std::remove_cvref_t<T>* get() const {
312 // remove ref (if provided) and return type refered to by T
313 using U = std::remove_cvref_t<T>;
314
315 const auto componentTypeId = ComponentTypeId::id<U>();
316
317 if (componentTypeId.value() >= components_.size()) {
318 return nullptr;
319 }
320
321 auto* cmp = components_[componentTypeId.value()].get();
322
323 if (!cmp) {
324 return nullptr;
325 }
326
327 return static_cast<U*>(cmp);
328 }

549 /**
550 * @brief Finalizes attachment of components by calling lifecycle callbacks.
551 *
552 * @details If a specific component is passed, only that component is finalized.
553 * Otherwise, all unattached components are finalized. Calls `onAttach()` followed
554 * by `onActivate()` or `onDeactivate()` based on the GameObject's active state.
555 *
556 * @param component Optional pointer to a specific component to finalize.
557 * If nullptr, all components are processed.
558 */
559 void finalizeAttach(helios::engine::ecs::Component* component = nullptr) {
560
561 // call attach on all components
562 if (component != nullptr) {
563 if (component->isAttached()) {
564 return;
565 }
566 component->onAttach(this);
567
568 if (isActive_) {
569 component->onActivate();
570 } else {
571 component->onDeactivate();
572 }
573
574 return;
575 }
576
577 // call attach on all components
578 for (const auto& component : components_) {
579 if (component == nullptr) {
580 continue;
581 }
582 if (component->isAttached()) {
583 continue;
584 }
585 component->onAttach(this);
586
587 if (isActive_) {
588 component->onActivate();
589 } else {
590 component->onDeactivate();
591 }
592 }
593
594 }
