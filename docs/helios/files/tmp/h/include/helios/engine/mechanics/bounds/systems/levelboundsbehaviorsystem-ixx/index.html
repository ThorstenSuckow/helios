43 /**
44 * @brief System that handles entity behavior when colliding with level boundaries.
45 *
46 * @details
47 * This system checks if entities with movement components have exceeded the level bounds.
48 * When an entity leaves the bounds, it applies bounce behavior based on the
49 * LevelBoundsBehaviorComponent's restitution coefficient.
50 */
51 class LevelBoundsBehaviorSystem : public helios::engine::ecs::System {
52
53 /**
54 * @brief Result of a bounce calculation against level bounds.
55 */
56 struct BounceResult {
57 /** @brief True if the X-axis boundary was hit. */
58 bool hitX;
59 /** @brief True if the Y-axis boundary was hit. */
60 bool hitY;
61 /** @brief The corrected world position. */
62 helios::math::vec3f translation;
63 /** @brief The new velocity after bounce. */
64 helios::math::vec3f velocity;
65 /** @brief The new direction vector. */
66 helios::math::vec3f direction;
67 };
68
69 public:
70
71 /**
72 * @brief Updates all entities that may have left level bounds.
73 *
74 * @details
75 * For each entity with the required components, checks if its world-space
76 * AABB is within level bounds. If not, applies bounce behavior.
77 *
78 * @param updateContext Context containing deltaTime and other frame data.
79 */
80 void update(helios::engine::runtime::world::UpdateContext& updateContext) noexcept override {
81
82 using namespace helios::engine::modules::physics::collision::types;
83
84 for (auto [entity, m2d, ab, sc, dc, tsc, bc, bbc] : gameWorld_->find<
85 helios::engine::modules::physics::motion::components::Move2DComponent,
86 helios::engine::modules::rendering::model::components::ModelAabbComponent,
87 helios::engine::modules::scene::components::SceneNodeComponent,
88 helios::engine::modules::physics::motion::components::DirectionComponent,
89 helios::engine::modules::spatial::transform::components::TranslationStateComponent,
90 helios::engine::modules::physics::collision::components::AabbColliderComponent,
91 helios::engine::mechanics::bounds::components::LevelBoundsBehaviorComponent
92 >().each()) {
93
94
95 auto& objectBounds = bc->bounds();
96 auto levelBounds = gameWorld_->level().bounds();
97
98 if (!levelBounds.contains(objectBounds)) {
99
100 helios::math::mat4f parentWorldTransform = sc->sceneNode()->parent()->worldTransform();
101 helios::math::vec4f childWorldTranslation = parentWorldTransform * tsc->translation().toVec4(1.0f);
102
103 helios::math::vec3f bouncedWorldTranslation;
104
105 BounceResult bounceResult{};
106
107 if (bbc->collisionBehavior() == CollisionBehavior::Reflect || bbc->collisionBehavior() == CollisionBehavior::Bounce) {
108
109 bounceResult = bounce(
110 childWorldTranslation.toVec3(), objectBounds, levelBounds,
111 bbc->collisionBehavior() == CollisionBehavior::Reflect ? 1.0f : bbc->restitution(),
112 *m2d, *dc
113 );
114
115 bouncedWorldTranslation = bounceResult.translation;
116 updateCollisionResponse(entity, bounceResult, bbc->collisionResponse());
117
118 } else if (bbc->collisionBehavior() == CollisionBehavior::Despawn) {
119 /**
120 * @todo optimize
121 */
122 auto* sbp = entity->get<helios::engine::mechanics::spawn::components::SpawnedByProfileComponent>();
123 assert(sbp && "Unexpected missing SpawnProfile");
124
125 updateContext.commandBuffer().add<helios::engine::runtime::spawn::commands::DespawnCommand>(
126 entity->guid(), sbp->spawnProfileId()
127 );
128
129 }
130
131
132 if (bounceResult.hitX || bounceResult.hitY) {
133 m2d->setVelocity(bounceResult.velocity);
134 dc->setDirection(bounceResult.direction);
135 }
136
137 auto parentTransform_inverse = parentWorldTransform.inverse();
138 auto childLocalTranslation = parentTransform_inverse * bouncedWorldTranslation.toVec4(1.0f);
139
140 tsc->setTranslation(childLocalTranslation.toVec3());
141 bc->setBounds(ab->aabb().applyTransform(
142 parentWorldTransform.withTranslation(bouncedWorldTranslation)
143 ));
144
145 }
146 }
147 }

149 private:
150
151 /**
152 * @brief Updates the entity's state based on the configured collision response.
153 *
154 * @details
155 * If the response is set to AlignHeadingToDirection, this method updates the
156 * SteeringComponent to match the new bounce direction.
157 *
158 * @param go Pointer to the GameObject to update.
159 * @param bounceResult The result data from the bounce calculation.
160 * @param collisionResponse The type of response to apply.
161 */
162 void updateCollisionResponse(
163 helios::engine::ecs::GameObject* go,
164 BounceResult bounceResult,
165 helios::engine::modules::physics::collision::types::CollisionResponse collisionResponse) {
166
167 if (collisionResponse == helios::engine::modules::physics::collision::types::CollisionResponse::AlignHeadingToDirection) {
168 auto* psc = go->get<helios::engine::modules::physics::motion::components::SteeringComponent>();
169
170 const auto direction = bounceResult.direction;
171
172 if (psc) {
173 psc->setSteeringIntent(
174 bounceResult.direction, 1.0f
175 );
176 }
177 }
178 }
179
180 /**
181 * @brief Applies bounce behavior to an entity that has left level bounds.
182 *
183 * @details
184 * Calculates the corrected position and reflected velocity when an entity
185 * collides with level boundaries. The restitution coefficient determines
186 * how much velocity is preserved after the bounce. The DirectionComponent
187 * will be updated with the new direction.
188 *
189 * @param worldTranslation Current world position of the entity.
190 * @param objectBounds World-space AABB of the entity.
191 * @param levelBounds World-space AABB of the level.
192 * @param restitution
193 * @param m2d Reference to the Move2DComponent for velocity modification.
194 * @param dc Reference to the DirectionComponent for direction update.
195 *
196 * @return Corrected world position after bounce.
197 */
198 [[nodiscard]] BounceResult bounce(
199 helios::math::vec3f worldTranslation,
200 helios::math::aabbf objectBounds,
201 helios::math::aabbf levelBounds,
202 const float restitution,
203 helios::engine::modules::physics::motion::components::Move2DComponent& m2d,
204 helios::engine::modules::physics::motion::components::DirectionComponent& dc
205 ) noexcept {
206
207
208 auto velocity = m2d.velocity();
209 bool hitX = false;
210 bool hitY = false;
211 auto translation = worldTranslation;
212
213 // Check left boundary collision
214 if (objectBounds.min()[0] < levelBounds.min()[0]) {
215 translation[0] += (levelBounds.min()[0] - objectBounds.min()[0]);
216 if (velocity[0] < 0) {
217 velocity = (velocity.withX(restitution * -velocity[0]));
218 }
219 hitX = true;
220 // Check right boundary collision
221 } else if (objectBounds.max()[0] > levelBounds.max()[0]) {
222 translation[0] -= (objectBounds.max()[0] - levelBounds.max()[0]) ;
223 if (velocity[0] > 0) {
224 velocity = (velocity.withX(restitution * -velocity[0]));
225 }
226 hitX = true;
227 }
228
229 objectBounds = objectBounds + (translation - worldTranslation);
230
231 // Check bottom boundary collision
232 if (objectBounds.min()[1] < levelBounds.min()[1]) {
233 translation[1] += (levelBounds.min()[1] - objectBounds.min()[1]);
234 if (velocity[1] < 0) {
235 velocity = velocity.withY(restitution * -velocity[1]);
236 }
237 hitY = true;
238 // Check top boundary collision
239 } else if (objectBounds.max()[1] > levelBounds.max()[1]) {
240 translation[1] -= (objectBounds.max()[1] - levelBounds.max()[1]);
241 if (velocity[1] > 0) {
242 velocity = velocity.withY(restitution * -velocity[1]);
243 }
244 hitY = true;
245 }
246
247
248 return {
249 hitX, hitY, translation, velocity, velocity.normalize()
250 };
251 }
