46 /**
47 * @brief Collision detection system using uniform spatial partitioning for Broadphase and
48 * AABB overlaps in the Narrowphase.
49 *
50 * @details This system implements a grid-based spatial partitioning approach for efficient
51 * collision detection, following the principles outlined in Ericson's "Real-Time Collision
52 * Detection" (Chapter 7). The algorithm divides the world into a uniform 3D grid of cells
53 * and assigns each collidable entity to the cells it overlaps.
54 *
55 * ## Detection Phases
56 *
57 * 1 **Broadphase:** Entities are inserted into grid cells based on their AABB. Only entities
58 * sharing the same cell are considered potential collision pairs.
59 * 2 **Narrowphase:** For each cell with multiple candidates, AABB intersection tests
60 * determine actual collisions.
61 *
62 * ## Collision Types
63 *
64 * Collision events are published to the `UpdateContext`'s event sink, distinguishing between:
65 * - **Solid collisions:** Symmetric collisions where both entities can physically interact.
66 * - **Trigger collisions:** Asymmetric collisions for gameplay logic (e.g., pickups, zones).
67 *
68 * ## Hit Policy
69 *
70 * The system respects each entity's HitPolicy setting:
71 * - **OneHit:** Entity receives only its first collision per frame, then skips further checks.
72 * - **All:** Entity receives collision events for all overlapping entities.
73 *
74 * This allows projectiles to stop on first contact while area effects can damage multiple targets.
75 *
76 * ## Layer Filtering
77 *
78 * During the broadphase, the system uses layer masks to filter which entity types
79 * can collide with each other, enabling fine-grained control over collision pairs.
80 *
81 * @see CollisionComponent
82 * @see AabbColliderComponent
83 * @see TriggerCollisionEvent
84 * @see SolidCollisionEvent
85 * @see HitPolicy
86 *
87 * @see [Eri05, Chapter 7]
88 */
89 class GridCollisionDetectionSystem : public helios::engine::ecs::System {

110 /**
111 * @brief Hash functor for pairs of GUIDs.
112 *
113 * Enables the use of GUID pairs as keys in unordered containers. The hash combines
114 * both GUIDs using XOR with a bit-shift to reduce collision probability for symmetric pairs.
115 *
116 * @todo switch to uint32_t once helios/#174 is implemented
117 */
118 struct GuidPairHash {
119
120 /**
121 * @brief Computes a hash value for a GUID pair.
122 *
123 * @param pair The pair of GUIDs to hash.
124 *
125 * @return A combined hash value for both GUIDs.
126 */
127 std::size_t operator()(const std::pair<helios::util::Guid, helios::util::Guid>& pair) const {
128
129 auto g1 = std::hash<helios::util::Guid>{}(pair.first);
130 auto g2 = std::hash<helios::util::Guid>{}(pair.second);
131
132 // compute the hash for the pair - shift g2 one position left, then xor with g1.
133 return g1 ^ (g2 << 1);
134 };
135
136 };

328 /**
329 * @brief Determines the collision type between two entities based on their layer masks.
330 *
331 * Evaluates both solid and trigger collision masks to determine if and how two entities
332 * can collide. Solid collisions are symmetric (both entities must accept collision with
333 * each other), while trigger collisions are asymmetric (either entity can trigger).
334 *
335 * @param cc Collision component of the first entity.
336 * @param matchCC Collision component of the second entity.
337 *
338 * @return CollisionStruct a struct with the requested collision information.
339 */
340 [[nodiscard]] inline CollisionStruct findCollisionType(
341 const CollisionComponent* cc,
342 const CollisionComponent* matchCC
343 ) const noexcept {
344
345 auto isSolidCollision = false;
346 auto isTriggerCollision = false;
347
348 auto aIsCollisionReporter = cc->isCollisionReporter();
349 auto bIsCollisionReporter = matchCC->isCollisionReporter();
350
351 // none of the reports a collision as a pair? skip!
352 if (!aIsCollisionReporter && !bIsCollisionReporter) {
353 return CollisionStruct{};
354 }
355
356 bool aCanSolidCollideWithB = (cc->solidCollisionMask() & matchCC->layerId()) != 0;
357 bool bCanSolidCollideWithA = (matchCC->solidCollisionMask() & cc->layerId()) != 0;
358
359 bool aCanTriggerCollideWithB = (cc->triggerCollisionMask() & matchCC->layerId()) != 0;
360 bool bCanTriggerCollideWithA = (matchCC->triggerCollisionMask() & cc->layerId()) != 0;
361
362 // solid collision is treated symmetric
363 isSolidCollision = aCanSolidCollideWithB && bCanSolidCollideWithA;
364
365 #if HELIOS_DEBUG
366 // detect asymmetric solid collision
367 if (aCanSolidCollideWithB && !bCanSolidCollideWithA) {
368 logger_.warn("Collision Asymmetry detected!");
369 }
370 #endif
371
372 // trigger collision is treated asymmetric
373 isTriggerCollision = aCanTriggerCollideWithB || bCanTriggerCollideWithA;
374
375 return CollisionStruct {
376 isSolidCollision,
377 isTriggerCollision,
378 aIsCollisionReporter,
379 bIsCollisionReporter,
380 isSolidCollision
381 ? cc->solidCollisionBehavior(matchCC->layerId()) : cc->triggerCollisionBehavior(matchCC->layerId()),
382 isSolidCollision
383 ? matchCC->solidCollisionBehavior(cc->layerId()) : matchCC->triggerCollisionBehavior(cc->layerId()),
384 // use the layerId of cc
385 cc->layerId(),
386 matchCC->layerId()
387
388 };
389 }

393 /**
394 * @brief Constructs a GridCollisionDetectionSystem with specified bounds and cell size.
395 *
396 * The grid is initialized to cover the given world bounds. Cell size determines
397 * the granularity of spatial partitioning; smaller cells reduce false positives
398 * but increase memory and insertion overhead.
399 *
400 * @param bounds World-space AABB defining the region where collision detection is active.
401 * @param cellSize Size of each grid cell in world units. Must be greater than zero.
402 *
403 * @note Given n objects, the 1/3 rule should be applied for determining the number of cells for partitioning,
404 * i.e. with n objects, the cube should consist of k x k x k cells, with k = n^{1/3} - that is, on average,
405 * one cell fits one object that should be tested for collision. Dimensions of said cells
406 * are derived by taking the reference space into account, e.g. WorldBounds.size/k (see [HS99]).
407 */
408 explicit GridCollisionDetectionSystem(
409 const helios::math::aabbf& bounds,
410 const float cellSize
411 ) : gridBounds_(bounds),
412 cellSize_(cellSize) {
413 assert(cellSize_ > 0.0f && "cellSize must not be <= 0.0f");
414 initGrid();
415 }
416
417 /**
418 * @brief Performs collision detection for all active entities.
419 *
420 * Executes the complete collision detection pipeline each frame:
421 * 1 Clears all grid cells and resets solved collision pairs.
422 * 2 Iterates all active GameObjects with CollisionComponent and AabbColliderComponent.
423 * 3 Inserts each entity into all grid cells its AABB overlaps.
424 * 4 For each cell with multiple candidates, runs narrow-phase AABB intersection tests.
425 * 5 Publishes collision events (TriggerCollisionEvent, SolidCollisionEvent) to the event queue.
426 *
427 * @param updateContext The update context providing access to GameWorld and event queue.
428 */
429 void update(helios::engine::runtime::world::UpdateContext& updateContext) noexcept override {
430
431 prepareCollisionDetection();
432
433 // only consider enabled CollisionSettingsComponents
434 constexpr auto filter = helios::engine::ecs::query::GameObjectFilter::Active | helios::engine::ecs::query::GameObjectFilter::ComponentEnabled;
435
436 for (auto [entity, cc, csc, acc] : gameWorld_->find<
437 CollisionComponent,
438 CollisionStateComponent,
439 AabbColliderComponent
440
441 >(filter).each()) {
442
443 if (!acc->boundsInitialized()) {
444 continue;
445 }
446
447 if (!cc->isEnabled() || (cc->triggerCollisionMask() == 0 && cc->solidCollisionMask() == 0)) {
448 continue;
449 }
450
451 const auto& worldBounds = acc->bounds();
452
453 updateCollisionCandidate(
454 entity,
455 worldBoundsToGridBounds(worldBounds),
456 acc,
457 cc,
458 csc
459 );
460 }
461
462 for (const auto idx : trackedCells_) {
463
464 if (cells_[idx].collisionCandidates.size() < 2) {
465 continue;
466 }
467
468 solveCell(cells_[idx], updateContext);
469 }
470 }

473 /**
474 * @brief Converts world-space AABB bounds to grid cell indices.
475 *
476 * Transforms a floating-point AABB in world space to integer cell coordinates,
477 * clamped to the valid grid range. Used to determine which cells an entity occupies.
478 *
479 * @param aabbf The world-space AABB to convert.
480 *
481 * @return An integer AABB representing the range of grid cell indices the entity spans.
482 */
483 [[nodiscard]] helios::math::aabbi worldBoundsToGridBounds(const helios::math::aabbf& aabbf) const noexcept {
484
485 helios::math::vec3f min = aabbf.min() - gridBounds_.min();
486 helios::math::vec3f max = aabbf.max() - gridBounds_.min();
487
488 int xMin = static_cast<int>(std::floor(min[0] / cellSize_));
489 int yMin = static_cast<int>(std::floor(min[1] / cellSize_));
490 int zMin = static_cast<int>(std::floor(min[2] / cellSize_));
491
492 int xMax = static_cast<int>(std::floor(max[0] / cellSize_));
493 int yMax = static_cast<int>(std::floor(max[1] / cellSize_));
494 int zMax = static_cast<int>(std::floor(max[2] / cellSize_));
495
496 return helios::math::aabbi{
497 std::clamp(xMin, 0, static_cast<int>(cellsX_ - 1)),
498 std::clamp(yMin, 0, static_cast<int>(cellsY_ - 1)),
499 std::clamp(zMin, 0, static_cast<int>(cellsZ_ - 1)),
500 std::clamp(xMax, 0, static_cast<int>(cellsX_ - 1)),
501 std::clamp(yMax, 0, static_cast<int>(cellsY_ - 1)),
502 std::clamp(zMax, 0, static_cast<int>(cellsZ_ - 1)),
503 };
504 }

507 /**
508 * @brief Inserts a collision candidate into all grid cells it overlaps.
509 *
510 * Iterates through all cells within the given bounds and adds the candidate
511 * to each cell's collision candidate list for subsequent narrow-phase testing.
512 *
513 * @param go Pointer to the GameObject entity.
514 * @param bounds Grid cell index bounds (integer AABB) the entity spans.
515 * @param aabbColliderComponent Pointer to the entity's AABB collider component.
516 * @param collisionComponent Pointer to the entity's collision component.
517 */
518 inline void updateCollisionCandidate(
519 helios::engine::ecs::GameObject* go,
520 const helios::math::aabbi& bounds,
521 AabbColliderComponent* aabbColliderComponent,
522 CollisionComponent* collisionComponent,
523 CollisionStateComponent* collisionStateComponent
524 ) {
525 const auto xMin = bounds.min()[0];
526 const auto xMax = bounds.max()[0];
527 const auto yMin = bounds.min()[1];
528 const auto yMax = bounds.max()[1];
529 const auto zMin = bounds.min()[2];
530 const auto zMax = bounds.max()[2];
531
532 for (int x = xMin; x <= xMax; x++) {
533 for (int y = yMin; y <= yMax; y++) {
534 for (int z = zMin; z <= zMax; z++) {
535 auto& [collisionCandidates] = cell(x, y, z);
536
537 collisionCandidates.push_back(
538 CollisionCandidate{
539 go,
540 aabbColliderComponent,
541 collisionComponent,
542 collisionStateComponent
543 }
544 );
545
546 // only consider the first added to prevent dups
547 if (collisionCandidates.size() == 1) {
548 const auto idx = cellIndex(x, y, z);
549 trackedCells_.push_back(idx);
550 }
551 }
552 }
553 }
554
555 }

558 /**
559 * @brief Performs narrow-phase collision detection for all candidates in a cell.
560 *
561 * Tests all unique pairs of collision candidates within the cell using AABB intersection.
562 * For each collision detected, determines collision type (solid/trigger) based on layer
563 * masks and publishes appropriate events. Uses a solved-set to avoid duplicate events
564 * when entities span multiple cells.
565 *
566 * @param cell The grid cell containing collision candidates to test.
567 * @param updateContext Context for pushing collision events to the event queue.
568 */
569 inline void solveCell(GridCell& cell, helios::engine::runtime::world::UpdateContext& updateContext) {
570
571 auto& candidates = cell.collisionCandidates;
572
573 // we will skip this cell if none of the candidates reports a collision
574 bool isCollisionReporter = false;
575 for (const auto& candidate : candidates) {
576 if (candidate.collisionComponent->isCollisionReporter()) {
577 isCollisionReporter = true;
578 break;
579 }
580 }
581
582 if (!isCollisionReporter) {
583 return;
584 }
585
586 for (size_t i = 0; i < candidates.size(); i++) {
587
588 CollisionCandidate& candidate = candidates[i];
589 CollisionComponent* cc = candidate.collisionComponent;
590 CollisionStateComponent* csc = candidate.collisionStateComponent;
591
592 auto hitPolicy = cc->hitPolicy();
593 if (hitPolicy == helios::engine::modules::physics::collision::types::HitPolicy::OneHit && csc->hasCollision()) {
594 continue;
595 }
596
597 const helios::math::aabbf& aabbCandidate = candidate.aabbColliderComponent->bounds();
598
599 for (size_t j = i+1; j < candidates.size(); j++) {
600
601 auto& [gameObject, aabbColliderComponent, collisionComponent, collisionStateComponent] = candidates[j];
602
603 CollisionComponent* matchCC = collisionComponent;
604 CollisionStateComponent* matchCSC = collisionStateComponent;
605
606 const auto collisionStruct = findCollisionType(cc, matchCC);
607
608 if (!collisionStruct.hasAnyInteraction()) {
609 continue;
610 }
611
612 const helios::math::aabbf& aabbMatch = aabbColliderComponent->bounds();
613 if (!aabbCandidate.intersects(aabbMatch)) {
614 continue;
615 }
616
617 auto lGuid = candidate.gameObject->guid();
618 auto rGuid = gameObject->guid();
619
620 if (lGuid > rGuid) {
621 std::swap(lGuid, rGuid);
622 }
623
624 // if we have already processed a collision, do not add this collision again.
625 if (solvedCollisions_.contains({lGuid, rGuid})) {
626 continue;
627 }
628
629 solvedCollisions_.insert({lGuid, rGuid});
630
631 postEvent(
632 candidate.gameObject, gameObject,
633 helios::math::overlapCenter(aabbCandidate, aabbMatch),
634 collisionStruct,
635 updateContext, csc, matchCSC
636 );
637
638 if (hitPolicy == helios::engine::modules::physics::collision::types::HitPolicy::OneHit) {
639 break;
640 }
641 }
642 }
643 }
