163 /**
164 * @brief Updates the text content if dirty.
165 *
166 * Applies the template to the current value and updates the renderable.
167 */
168 void update() {
169
170 if (!isDirty_) {
171 return;
172 }
173
174 if (template_ == "{0}" || template_.empty()) {
175 renderable_->setText(std::to_string(doubleValue_));
176 needsResize_ = true;
177 isDirty_ = false;
178 return;
179 }
180
181 renderable_->setText(std::vformat(std::string_view(template_), std::make_format_args(doubleValue_)));
182 needsResize_ = true;
183 isDirty_ = false;
184 }
