33 /**
34 * @brief High-level manager for GameObject pooling operations.
35 *
36 * @details GameObjectPoolManager provides a unified interface for managing
37 * multiple GameObjectPools. It handles pool configuration, initialization,
38 * and the acquire/release lifecycle of pooled GameObjects.
39 *
40 * The manager integrates with the GameWorld to clone prefabs during pool
41 * initialization and to look up GameObjects by their GUID during
42 * acquire/release operations.
43 *
44 * ## Lifecycle
45 *
46 * 1 **Configuration**: Add pool configurations via `addPoolConfig()`.
47 * 2 **Initialization**: Call `init()` to create pools and populate them
48 * with cloned prefab instances.
49 * 3 **Runtime**: Use `acquire()` and `release()` to manage entity lifecycle.
50 *
51 * ## Example
52 *
53 * ```cpp
54 * GameObjectPoolManager poolManager;
55 *
56 * auto config = std::make_unique<GameObjectPoolConfig>(
57 * bulletPoolId, bulletPrefab, 100
58 * );
59 * poolManager.addPoolConfig(std::move(config));
60 *
61 * poolManager.init(gameWorld);
62 *
63 * if (auto* bullet = poolManager.acquire(bulletPoolId)) {
64 * bullet->get<ComposeTransformComponent>()->setTranslation(spawnPos);
65 * }
66 *
67 * poolManager.release(bulletPoolId, bullet->guid());
68 * ```
69 *
70 * @see GameObjectPool
71 * @see GameObjectPoolRegistry
72 * @see Manager
73 */
74 class GameObjectPoolManager : public helios::engine::runtime::world::Manager {

76 /**
77 * @brief Registry of GameObjectPools for entity recycling.
78 *
79 * @details Pools enable efficient reuse of GameObjects without repeated
80 * allocation/deallocation. Each pool is identified by a GameObjectPoolId.
81 */
82 helios::engine::runtime::pooling::GameObjectPoolRegistry pools_{};
83
84 /**
85 * @brief Non-owning pointer to the associated GameWorld.
86 *
87 * @details Set during `init()`. Used for cloning prefabs and looking up
88 * GameObjects by GUID.
89 */
90 helios::engine::runtime::world::GameWorld* gameWorld_ = nullptr;
91
92 /**
93 * @brief Pending pool configurations awaiting initialization.
94 *
95 * @details Configurations are consumed during `init()` to create and
96 * populate the actual pools.
97 */
98 std::unordered_map<
99 helios::engine::core::data::GameObjectPoolId,
100 std::unique_ptr<GameObjectPoolConfig>> poolConfigs_;

102 /**
103 * @brief Populates a pool with cloned prefab instances.
104 *
105 * @details Clones the prefab GameObject until the pool reaches its
106 * configured capacity. Each clone is immediately deactivated and
107 * marked as inactive in the pool.
108 *
109 * @param gameObjectPoolId The pool to fill.
110 * @param gameObjectPrefab The prefab to clone.
111 */
112 void fillPool(
113 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId,
114 const helios::engine::ecs::GameObject& gameObjectPrefab
115 ) {
116 helios::util::Guid guid{helios::core::types::no_init};
117
118 auto* gameObjectPool = pool(gameObjectPoolId);
119
120 const size_t used = gameObjectPool->activeCount() + gameObjectPool->inactiveCount();
121 const size_t space = used < gameObjectPool->size() ? gameObjectPool->size() - used : 0;
122
123 for (size_t i = 0; i < space; i++) {
124 helios::engine::ecs::GameObject* go = gameWorld_->clone(gameObjectPrefab);
125 if (go) {
126 go->setActive(false);
127 go->onRelease();
128 gameObjectPool->addInactive(go->guid());
129 }
130 }
131 }

135 /**
136 * @brief Registers a pool configuration for later initialization.
137 *
138 * @details The configuration specifies the pool ID, capacity, and prefab
139 * to use for cloning. Configurations are processed during `init()`.
140 *
141 * @param gameObjectPoolConfig The pool configuration to register.
142 *
143 * @return Reference to this manager for method chaining.
144 *
145 * @throws std::runtime_error If a pool with the same ID is already registered.
146 */
147 GameObjectPoolManager& addPoolConfig(std::unique_ptr<GameObjectPoolConfig> gameObjectPoolConfig) {
148
149 if (poolConfigs_.contains(gameObjectPoolConfig->gameObjectPoolId)) {
150 throw std::runtime_error("Pool already registered with this manager");
151 }
152
153 poolConfigs_[gameObjectPoolConfig->gameObjectPoolId] = std::move(gameObjectPoolConfig);
154
155 return *this;
156 }

178 /**
179 * @brief Releases a GameObject back to its pool.
180 *
181 * @details Marks the entity as inactive in both the pool and the
182 * GameWorld. Calls `onRelease()` on the GameObject to allow cleanup.
183 *
184 * @param gameObjectPoolId The pool that owns this entity.
185 * @param entityId The GUID of the entity to release.
186 *
187 * @return Pointer to the released GameObject, or nullptr if not found.
188 */
189 helios::engine::ecs::GameObject* release(
190 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId,
191 const helios::util::Guid& entityId
192 ) {
193 auto* gameObjectPool = pool(gameObjectPoolId);
194
195 helios::engine::ecs::GameObject* worldGo = gameWorld_->find(entityId);
196
197 if (worldGo) {
198 if (gameObjectPool->release(entityId)) {
199 worldGo->onRelease();
200 worldGo->setActive(false);
201 }
202 }
203
204 return worldGo;
205 }
206
207 /**
208 * @brief Acquires an inactive GameObject from the pool.
209 *
210 * @details Retrieves the next available inactive entity, activates it,
211 * and calls `onAcquire()` to prepare it for use.
212 *
213 * If an entity's GUID is no longer valid in the GameWorld, it is removed
214 * from the pool and the next entity is tried.
215 *
216 * @param gameObjectPoolId The pool to acquire from.
217 *
218 * @return Pointer to the acquired GameObject, or nullptr if pool exhausted.
219 */
220 [[nodiscard]] helios::engine::ecs::GameObject* acquire(
221 const helios::engine::core::data::GameObjectPoolId gameObjectPoolId
222 ) {
223 helios::util::Guid guid{helios::core::types::no_init};
224
225 auto* gameObjectPool = pool(gameObjectPoolId);
226
227 while (gameObjectPool->acquire(guid)) {
228
229 auto* worldGo = gameWorld_->find(guid);
230
231 if (worldGo) {
232 worldGo->onAcquire();
233 return worldGo;
234 }
235
236 // we assume the pool is owned by this gameWorld,
237 // so removing this guid does not impact another gameWorld that is
238 // using this pool
239 gameObjectPool->releaseAndRemove(guid);
240 }
241
242 return nullptr;
243
244 }

246 /**
247 * @brief Initializes all registered pools.
248 *
249 * @details Creates GameObjectPool instances from the pending configurations
250 * and populates them by cloning the specified prefabs into the GameWorld.
251 *
252 * @param gameWorld The GameWorld to associate with this manager.
253 */
254 void init(helios::engine::runtime::world::GameWorld& gameWorld) override {
255
256 gameWorld_ = &gameWorld;
257
258 for (const auto& [gameObjectPoolId, poolConfig] : poolConfigs_) {
259
260 auto pool = std::make_unique<helios::engine::runtime::pooling::GameObjectPool>(
261 poolConfig->amount);
262
263 pools_.addPool(gameObjectPoolId, std::move(pool));
264 fillPool(gameObjectPoolId, *poolConfig->prefab);
265 }
266
267 };
