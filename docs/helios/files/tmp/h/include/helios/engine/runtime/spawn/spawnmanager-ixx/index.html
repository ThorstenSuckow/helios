47 /**
48 * @brief Manager for processing spawn and despawn commands.
49 *
50 * @details SpawnManager is a Manager that handles the lifecycle of pooled
51 * GameObjects. It receives commands via SpawnCommandHandler::submit() and
52 * queues them for deferred processing during the manager flush phase.
53 *
54 * ## Command Processing Order
55 *
56 * During flush(), commands are processed in this order:
57 * 1 **Despawn commands** — Entities are returned to their pools
58 * 2 **Spawn commands** — Single entities are acquired and initialized
59 * 3 **Scheduled spawn plan commands** — Batch spawns from scheduler
60 *
61 * ## Spawn Profiles
62 *
63 * Each spawn profile defines:
64 * - Which pool to acquire entities from
65 * - How to position spawned entities (SpawnPlacer)
66 * - How to initialize spawned entities (SpawnInitializer)
67 *
68 * ## Integration
69 *
70 * SpawnManager registers itself with the GameWorld for each profile ID,
71 * allowing commands to route to the correct handler.
72 *
73 * Example:
74 * ```cpp
75 * auto spawnManager = std::make_unique<SpawnManager>();
76 *
77 * spawnManager->addSpawnProfile(bulletProfileId, std::make_unique<SpawnProfile>(
78 * SpawnProfile{
79 * .gameObjectPoolId = bulletPoolId,
80 * .spawnPlacer = std::make_unique<EmitterSpawnPlacer>(),
81 * .spawnInitializer = std::make_unique<EmitterInitializer>()
82 * }
83 * ));
84 *
85 * gameWorld.addManager(std::move(spawnManager));
86 * ```
87 *
88 * @see SpawnCommand
89 * @see DespawnCommand
90 * @see ScheduledSpawnPlanCommand
91 * @see SpawnProfile
92 * @see SpawnCommandHandler
93 */
94 class SpawnManager : public helios::engine::runtime::world::Manager,

186

188

193 assert(false && "SpawnManager does not manage this SpawnProfileId");

195
196
197
198 const auto spawnCount = std::min(amount, poolSnapshot.inactiveCount);
199 for (size_t i = 0; i < spawnCount; i++) {
200

202

226 spawnContext

426 SpawnManager& addSpawnProfile(
