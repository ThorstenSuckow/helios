43 /**
44 * @brief Central registry for game entities, managers, pools, and the active level.
45 *
46 * @details
47 * The GameWorld is the root container for the game state. It manages the lifecycle
48 * of GameObjects, coordinates Managers, and holds the current Level.
49 *
50 * ## Key Responsibilities
51 *
52 * - **Entity Management:** Owns all GameObjects and provides amortized O(1) lookup by Guid
53 * via `std::unordered_map`.
54 * - **Component Queries:** Efficient iteration over GameObjects with specific components
55 * via `find<Components...>()` with optional filtering by active/enabled state.
56 * - **Pool Management:** Registers and provides access to GameObjectPools for entity recycling.
57 * - **Manager Coordination:** Holds Managers that handle cross-cutting concerns
58 * (spawning, projectile pooling) and flushes them each frame.
59 * - **Level Management:** Holds the active Level instance with arena bounds.
60 *
61 * ## Usage with GameLoop
62 *
63 * The GameWorld is passed to Systems via UpdateContext. Systems query GameObjects
64 * and their components, while mutations are typically performed through Commands
65 * that are flushed by the CommandBuffer.
66 *
67 * ```cpp
68 * // In a System
69 * void update(UpdateContext& ctx) noexcept override {
70 * for (auto [obj, move] : ctx.gameWorld().find<Move2DComponent>().each()) {
71 * // Process entities with Move2DComponent
72 * }
73 * }
74 * ```
75 *
76 * ## Filtering
77 *
78 * The `find()` methods accept a `GameObjectFilter` bitmask to control which
79 * entities are returned:
80 *
81 * - `GameObjectFilter::Active` — Only active GameObjects
82 * - `GameObjectFilter::ComponentEnabled` — Only objects with enabled components
83 *
84 * Default filter is `Active | ComponentEnabled`.
85 *
86 * @see GameObject
87 * @see GameObjectFilter
88 * @see UpdateContext
89 * @see Manager
90 * @see Level
91 */

381 /**
382 * @brief Finds all GameObjects that have the specified component types.
383 *
384 * @details
385 * Returns a lazy range that filters GameObjects on-the-fly during iteration.
386 * Only GameObjects possessing all specified component types are yielded.
387 *
388 * The query is controlled by a `GameObjectFilter` bitmask:
389 *
390 * | Filter | Effect |
391 * |--------|--------|
392 * | `Active` | Only `obj->isActive() == true` |
393 * | `Inactive` | Only `obj->isActive() == false` |
394 * | `ComponentEnabled` | Only objects where queried components are enabled |
395 * | `ComponentDisabled` | Only objects where queried components are disabled |
396 *
397 * Filters can be combined: `Active | ComponentEnabled` (default).
398 *
399 * Example usage:
400 * ```cpp
401 * // Range-based for loop with default filter (Active + ComponentEnabled)
402 * for (auto* obj : gameWorld.find<Move2DComponent, SceneNodeComponent>()) {
403 * auto* move = obj->get<Move2DComponent>();
404 * move->setVelocity({1.0f, 0.0f, 0.0f});
405 * }
406 *
407 * // Using each() for direct component access (structured bindings)
408 * for (auto [obj, move] : gameWorld.find<Move2DComponent>().each()) {
409 * move.get().setVelocity({1.0f, 0.0f, 0.0f});
410 * }
411 *
412 * // Custom filter: find inactive objects
413 * for (auto* obj : gameWorld.find<HealthComponent>(GameObjectFilter::Inactive)) {
414 * // Process inactive entities (e.g., respawn logic)
415 * }
416 * ```
417 *
418 * @tparam Cs The component types to filter by. GameObjects must have all types.
419 *
420 * @param query Filter bitmask controlling which entities are included.
421 * Default: `Active | ComponentEnabled`.
422 *
423 * @return A GameObjectView object that can be iterated over.
424 *
425 * @see GameObjectFilter
426 * @see GameObjectView
427 */
428 template<class... Cs>
429 [[nodiscard]] auto find(GameObjectFilter query = GameObjectFilter::Active | GameObjectFilter::ComponentEnabled) {
430 return GameObjectRange<Map, Cs...>(gameObjects_, query);
431 }

473 /**
474 * @brief Creates a clone of an existing GameObject.
475 *
476 * @details Clones all Cloneable components from the source GameObject.
477 * The cloned GameObject is initially inactive and receives a new Guid.
478 * Non-cloneable components are skipped with a warning.
479 *
480 * @param gameObject The source GameObject to clone.
481 *
482 * @return Pointer to the newly created clone, or nullptr on failure.
483 *
484 * @note The clone is added to the world in an inactive state.
485 */
486 [[nodiscard]] helios::engine::ecs::GameObject* clone(const helios::engine::ecs::GameObject& gameObject) {
487
488
489 auto newGo = std::make_unique<helios::engine::ecs::GameObject>();
490 /**
491 * @todo Optional ClonePolicy where rules are specified?
492 */
493 newGo->setActive(false);
494
495 for (const auto* component : gameObject.components()) {
496 if (const auto* cc = dynamic_cast<const helios::engine::ecs::Cloneable*>(component)) {
497 auto cComponent = cc->clone();
498 // use getOrAdd since cloned components may have already added
499 // other components in onAttach(), which we will not defer for now.
500 // we assume that the cloned component that allows for adding
501 // components in onAttach is the source of truth for said components
502 newGo->getOrAdd(std::move(cComponent));
503 } else {
504 logger_.warn("Component not cloneable.");
505 }
506
507 }
508
509 return addGameObject(std::move(newGo));
510 }

513 /**
514 * @brief Removes a GameObject from the world and transfers ownership to the caller.
515 *
516 * @param gameObject Reference to the GameObject to remove.
517 *
518 * @return Unique pointer to the removed GameObject, or nullptr if not found.
519 *
520 * @retval std::unique_ptr<GameObject> Ownership of the removed object
521 * @retval nullptr If the GameObject was not found in the world
522 *
523 * @note After successful removal, all non-owning pointers to the GameObject
524 * become invalid unless the caller maintains the returned unique_ptr.
525 * @note Attempting to remove a GameObject that doesn't exist returns nullptr
526 * and logs a warning.
527 */
528 [[nodiscard]] std::unique_ptr<helios::engine::ecs::GameObject> removeGameObject(const helios::engine::ecs::GameObject& gameObject) {
529 auto node = gameObjects_.extract(gameObject.guid());
530
531 if (node.empty()) {
532 logger_.warn(std::format("Attempted to remove non-existent GameObject with Guid {} from GameWorld",
533 gameObject.guid().value()));
534 return nullptr;
535 }
536
537 return std::move(node.mapped());
538 }
