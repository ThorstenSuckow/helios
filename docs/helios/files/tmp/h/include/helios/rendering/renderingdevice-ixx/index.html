18 /**
19 * @brief Abstract interface for a low-level rendering device.
20 *
21 * This pure virtual class provides the fundamental API for managing
22 * the rendering pipeline and respective contracts for concrete
23 * implementations using a specific rendering backend (e.g. OpenGL,
24 * Vulkan...).
25 *
26 * Implementations of `RenderingDevice` are responsible for managing
27 * render passes and configuring the viewport.
28 */
29 class RenderingDevice {
30
31 protected:
32 /**
33 * @brief Reflects the initialization state of this device.
34 */
35 bool initialized_ = false;
36
37 /**
38 * @brief The logger used with this RenderingDevice class.
39 *
40 * Defaults to HELIOS_LOG_SCOPE.
41 */
42 inline static const helios::util::log::Logger& logger_ = helios::util::log::LogManager::loggerForScope(
43 HELIOS_LOG_SCOPE
44 );
45
46 public:
47 virtual ~RenderingDevice() = default;
48
49 /**
50 * @brief Initializes this rendering device.
51 *
52 * Should be called when the graphics context was created and before
53 * any rendering operations are performed.
54 *
55 * Implementing classes should load pointers to underlying GL functions
56 * or provide similar API-specific setup.
57 *
58 * @throws std::runtime_error if initialization of the device fails.
59 */
60 virtual void init() = 0;
61
62 /**
63 * @brief Signals this device that the application is now ready to begin a new render pass.
64 *
65 * Implementing classes should prepare the rendering surface at this point and
66 * clear specific buffers.
67 *
68 * This method should be called at the start of each frame before any drawing occurs.
69 *
70 * @param renderPass The RenderPass that is about to be processed by the rendering device.
71 */
72 virtual void beginRenderPass(helios::rendering::RenderPass& renderPass) const noexcept = 0;
73
74
75 /**
76 * @brief Advises this RenderingDevice to process the specified RenderPass.
77 *
78 * A `RenderPass` consists of `RenderCommand`s this RenderDevice executes, i.e.
79 * binding meshes, materials and shaders as well as drawing geometry.
80 *
81 * @param renderPass A reference to the RenderPass that holds the `RenderCommand`s to be executed.
82 */
83 virtual void doRender(helios::rendering::RenderPass& renderPass) const noexcept = 0;
84
85
86 /**
87 * @brief Ends the specified render pass of this RenderingDevice.
88 *
89 * This method should be called when the renderPass was processed by the RenderingDevice,
90 * before the window swaps the buffers.
91 *
92 * Implementing classes should take care of unbinding frame buffers and/or flushing
93 * command buffers.
94 *
95 * @param renderPass The RenderPass that was processed by the rendering device.
96 */
97 virtual void endRenderPass(helios::rendering::RenderPass& renderPass) const noexcept = 0;
98
99 /**
100 * @brief Convenience method to subsequently call `beginRenderPass`, `doRender`, and `endRenderPass` (in this order) with the specified `RenderPass`.
101 *
102 * @param renderPass The RenderPass to process by this RenderingDevice.
103 */
104 void render(helios::rendering::RenderPass& renderPass) const {
105 beginRenderPass(renderPass);
106 doRender(renderPass);
107 endRenderPass(renderPass);
108 }
109
110
111
112 /**
113 * @brief Returns the initialized state of this rendering device.
114 *
115 * This method is guaranteed to return `true` if the device was successfully
116 * initialized, otherwise `false`.
117 *
118 * @return `true` if the device was successfully initialized, `false` otherwise.
119 */
120 [[nodiscard]] bool initialized() const noexcept{
121 return initialized_;
122 };
123 };

125}
