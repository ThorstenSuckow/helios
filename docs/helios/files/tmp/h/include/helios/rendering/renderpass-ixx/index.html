23 /**
24 * @brief Represents a single RenderPass under a set of specific uniform values.
25 *
26 * A `RenderPass` holds a `RenderQueue` that contains all `RenderCommands` that need
27 * to be processed in the subsequent rendering. The associated uniform values specified
28 * with this `RenderPass` remain constant during processing of the `RenderCommand`s.
29 *
30 * `RenderPass`es are typically processed sequentially by an API-specific implementation
31 * of the pure virtual `RenderingDevice` class.
32 *
33 * @note A `RenderPass` should be configurable with other properties relevant to the underlying
34 * rendering backend, e.g. depth testing, draw mode etc.
35 */
36 class RenderPass {
37
38 private:
39 /**
40 * @brief An owning unique pointer to the `RenderQueue` for this pass.
41 */
42 std::unique_ptr<helios::rendering::RenderQueue> renderQueue_;
43
44 /**
45 * @brief An owning unique pointer to the uniform values specific to the current pass (i.e.,
46 * the rendered "frame"). This map contains uniforms that change once per frame, such as the view and the
47 * projection matrix.
48 */
49 std::unique_ptr<const helios::rendering::shader::UniformValueMap> frameUniformValues_;
50
51 /**
52 * @brief Shared ownership of the `Viewport` processed by this pass.
53 */
54 std::shared_ptr<const helios::rendering::Viewport> viewport_;
55
56 protected:
57 /**
58 * @brief Shared logger instance for all RenderPass objects.
59 */
60 inline static const helios::util::log::Logger& logger_ = helios::util::log::LogManager::loggerForScope(
61 HELIOS_LOG_SCOPE
62 );

89 /**
90 * @brief Create a new `RenderPass` with the specified `Viewport`, `RenderQueue` and the `UniformValueMap`.
91 * Ownership of Viewport is shared with this instance, the latter objects are transferred to **this** `RenderPass`.
92 *
93 * @param viewport The viewport this RenderPass is processing.
94 * @param renderQueue A unique ptr to the `RenderQueue` that should be processed with this pass.
95 * @param frameUniformValues A unique ptr to the `UniformValueMap` associated with this pass, i.e.
96 * frame-specific uniform values. If no values are specified, a default empty set is created.
97 */
98 explicit RenderPass(
99 std::shared_ptr<const helios::rendering::Viewport> viewport,
100 std::unique_ptr<helios::rendering::RenderQueue> renderQueue,
101 std::unique_ptr<const helios::rendering::shader::UniformValueMap> frameUniformValues = nullptr
102 ) noexcept
103 :
104 viewport_(std::move(viewport)),
105 renderQueue_(std::move(renderQueue)),
106 frameUniformValues_(std::move(frameUniformValues)) {
107
108 if (!viewport_) {
109 logger_.error("Constructor received a nullptr Viewport. Creating a default empty Viewport.");
110 viewport_ = std::make_shared<Viewport>();
111 }
112
113 if (!renderQueue_) {
114 logger_.error("Constructor received a nullptr RenderQueue. Creating a default empty queue.");
115 renderQueue_ = std::make_unique<helios::rendering::RenderQueue>();
116 }
117
118 if (!frameUniformValues_) {
119 logger_.warn("Constructor received a nullptr UniformValueMap. Creating a default empty map.");
120 frameUniformValues_ = std::make_unique<helios::rendering::shader::UniformValueMap>();
121 }
122 }
123
124 /**

143 * @brief Sets this `RenderPass`' `UniformValueMap` containing per-frame specific uniform values.
144 * Transfers ownership of the provided UniformValueMap to this `RenderPass`.
145 *
146 * @param frameUniformValues A unique ptr to the `UniformValueMap` for frame specific uniform values.
147 */
148 void setFrameUniformValues(std::unique_ptr<const helios::rendering::shader::UniformValueMap> frameUniformValues) noexcept {

150 }
