23 /**
24 * @brief Manages collections of render commands for geometry and text.
25 *
26 * `RenderQueue` acts as a container for all render commands in a single rendering pass.
27 * It manages two separate command lists:
28 *
29 * - **RenderCommands:** For geometry rendering (meshes, materials).
30 * - **TextRenderCommands:** For text rendering (glyphs, fonts).
31 *
32 * A `RenderQueue` holds unique ownership of geometry render commands and value ownership
33 * of text render commands. It can be cleared after processing for reuse in subsequent passes.
34 *
35 * ## Design
36 *
37 * - **Non-copyable, non-movable:** Ensures unique queue identity during processing.
38 * - **Separate command lists:** Allows different rendering strategies for geometry and text.
39 *
40 * @see RenderCommand
41 * @see TextRenderCommand
42 * @see RenderPass
43 */
44 class RenderQueue {

90 /**
91 * @brief Adds a `RenderCommand` to this `RenderQueue`.
92 * Ownership of the `RenderCommand` is transferred to this `RenderQueue`.
93 *
94 * @param renderCommand A unique_ptr to the RenderCommand. This
95 * instance takes ownership of the RenderCommand.
96 *
97 * @todo prevent adding renderables while rendering
98 */
99 void add(std::unique_ptr<const helios::rendering::RenderCommand> renderCommand) {
100 assert(renderCommand && "Received nullptr RenderCommand.");
101 if (!renderCommand) {
102 logger_.error("Attempted to add a nullptr RenderCommand to the queue.");
103 return; // silently skip nullptr to avoid crashes
104 }
105 renderCommands_.emplace_back(std::move(renderCommand));
106 }
107
108 /**
109 * @brief Adds a `TextRenderCommand` to this `RenderQueue`.
110 *
111 * The command is moved into the queue. Text commands are processed
112 * separately from geometry commands during rendering.
113 *
114 * @param renderCommand The text render command to add (moved).
115 */
116 void add(helios::rendering::text::TextRenderCommand renderCommand) {
117 textRenderCommands_.emplace_back(std::move(renderCommand));
118 }
