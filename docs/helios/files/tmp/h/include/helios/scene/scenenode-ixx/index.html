42 /**
43 * @brief Represents a SceneNode within a SceneGraph.
44 *
45 * A SceneNode holds its local transformation and a (shared) pointer to
46 * a Renderable object (the representative of a visible element).
47 * Furthermore, a SceneNode can have multiple child nodes uniquely owned by
48 * **this** SceneNode.
49 *
50 * A SceneNode holds a global unique identifier and tracks a dirty state (`needsUpdate_`)
51 * related to any transformations that occurred on the SceneNode.
52 * Ownership is managed by a collection of unique pointers of the child nodes.
53 * SceneNodes are designed to be not copyable and not assignable to prevent duplication
54 * and to enforce the hierarchical tree structure.
55 *
56 * The `worldTransform_` is derived from a SceneNode's `localTransform_` and the SceneNode's
57 * parent's `worldTransform`s. A call to `worldTransform()` will update the particular subtree
58 * of the SceneNode bottom-up to make sure the actual values are available with the SceneNode's
59 * `worldTransform`. A Scene might also propagate changes top-bottom - see `Scene::updateNodes()`.
60 *
61 */

356 /**
357 * @brief Sets the world transform for this SceneNode.
358 *
359 * @details
360 * Does nothing if the current SceneNode's world transform is considered
361 * equal to the specified world transform.
362 *
363 * @param wf The worldTransform to use for this SceneNode.
364 * @param sceneGraphKey The sceneGraphKey as constructed by the `Scene`. Serves
365 * as the passkey so this method is not callable from outside.
366 *
367 * @return true if the world transform was updated, otherwise false.
368 */
369 [[nodiscard]] bool applyWorldTransform(
370 const helios::math::mat4f& parentWorldTransform, helios::scene::SceneGraphKey sceneGraphKey
371 ) noexcept {
372
373 const auto newWt = inheritWorldTransform(parentWorldTransform);
374
375 if (!needsUpdate() && worldTransform_.same(newWt)) {
376 return false;
377 }
378
379 worldTransform_ = newWt;
380 needsUpdate_ = false;
381
382
383 onWorldTransformUpdate();
384 return true;
385 }

442 /**
443 * @brief Updates the world transform and axis-aligned bounding box of this SceneNode.
444 *
445 * Recomputes the `worldTransform_` based on the parent's world transform and
446 * this node's `localTransform_`. Also updates the `aabb_` to reflect the
447 * current world-space bounds.
448 */
449 void update() noexcept {
450 needsUpdate_ = false;
451
452 if (parent_ == nullptr) {
453 worldTransform_ = localTransform_.transform();
454 } else {
455 worldTransform_ = inheritWorldTransform(parent_->worldTransform());
456 }
457

459 }

461 /**
462 * @brief Virtual callback invoked after the world transform has been updated.
463 *
464 * Derived classes can override this method to perform additional processing
465 * when the node's world transform changes, such as updating dependent
466 * resources or triggering view matrix recalculations in camera nodes.
467 */
468 virtual void onWorldTransformUpdate() noexcept {
469 if (renderable_) {
470 if (const auto prototype = renderable_->renderPrototype()) {
471 const auto& localAABB = prototype->mesh().aabb();
472 aabb_ = localAABB.applyTransform(worldTransform_);
473 }
474 }
475 }

477 /**
478 * @brief Filters a parent world transform based on the node's inheritance flags.
479 *
480 * This method extracts only the transform components (Translation, Rotation, Scale)
481 * that this node is configured to inherit from its parent. Components not included
482 * in `inheritance_` are replaced with identity values.
483 *
484 * @param parentWorldTransform The parent's world transform matrix.
485 *
486 * @return A filtered transform matrix containing only the inherited components.
487 *
488 * @see setInheritance()
489 * @see helios::math::TransformType
490 */
491 helios::math::mat4f inheritWorldTransform(const helios::math::mat4f& parentWorldTransform) noexcept {
492 using namespace helios::math;
493
494 if (inheritance_ == helios::math::TransformType::All) {
495 return parentWorldTransform * localTransform_.transform();
496 }
497
498 auto id = parentWorldTransform.decompose(inheritance_);
499
500 return id * localTransform_.transform();
501 }
